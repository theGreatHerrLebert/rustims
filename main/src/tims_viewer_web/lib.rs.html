119/// wasm entry point — Trunk/wasm-bindgen call this on load.
120#[wasm_bindgen(start)]
121pub fn start() {
122    // Custom panic hook: keep the console output AND forward the panic to the server's /telemetry
123    // (PROD only, gated on TELEMETRY_ON once /clientconfig confirms it's enabled — no-op in dev).
124    std::panic::set_hook(Box::new(|info| {
125        console_error_panic_hook::hook(info);
126        if TELEMETRY_ON.load(std::sync::atomic::Ordering::Relaxed) {
127            fire_event("/telemetry", telemetry_json("wasm-panic", &info.to_string()));
128        }
129    }));
130    // Off the main thread (the clustering Web Worker re-inits this same wasm) there is no DOM —
131    // `window()` is None. Skip the renderer; the worker only calls `cluster_dbscan_flat`.
132    if web_sys::window().is_none() {
133        return;
134    }
135    let _ = console_log::init_with_level(log::Level::Info);
136    wasm_bindgen_futures::spawn_local(async {
137        if let Err(e) = run().await {
138            log::error!("{e}");
139            hide_loading(); // don't leave the veil covering the error
140            show_status(&format!("tims-viewer could not start: {e}"));
141        }
142    });
143    // Probe /clientconfig; wire error forwarding + the feedback widget only if the server enabled them.
144    wasm_bindgen_futures::spawn_local(init_telemetry_and_feedback());
145}

156/// Minimal JSON string encoder (escape quotes/backslashes/controls), so we can hand-build event
157/// bodies without pulling in serde on the wasm side.
158fn json_str(s: &str) -> String {
159    let mut out = String::with_capacity(s.len() + 2);
160    out.push('"');
161    for c in s.chars() {
162        match c {
163            '"' => out.push_str("\\\""),
164            '\\' => out.push_str("\\\\"),
165            '\n' => out.push_str("\\n"),
166            '\r' => out.push_str("\\r"),
167            '\t' => out.push_str("\\t"),
168            c if (c as u32) < 0x20 => out.push_str(&format!("\\u{:04x}", c as u32)),
169            c => out.push(c),
170        }
171    }
172    out.push('"');
173    out
174}

176/// A telemetry event body: kind + message + light context (dataset, page URL, user agent).
177fn telemetry_json(kind: &str, message: &str) -> String {
178    let ds = dataset_id();
179    let win = web_sys::window();

184        json_str(kind), json_str(message), json_str(&href), json_str(&ua)

188/// Fire-and-forget POST of a JSON body to a same-server endpoint. A plain-string body stays a CORS
189/// "simple request" (no preflight); failures (e.g. telemetry disabled -> 404) are ignored on purpose.
190fn fire_event(path: &str, body: String) {
191    let Some(window) = web_sys::window() else { return };
192    let url = format!("{}{path}", server_base());
193    let opts = web_sys::RequestInit::new();
194    opts.set_method("POST");
195    opts.set_body(&wasm_bindgen::JsValue::from_str(&body));
196    let _ = window.fetch_with_str_and_init(&url, &opts);
197}

199/// Ask the server (GET /clientconfig) whether error telemetry + the feedback widget are enabled
200/// (PROD only), and wire them if so. In dev the endpoint 404s / returns false and this is a no-op.
201async fn init_telemetry_and_feedback() {
202    let Some(window) = web_sys::window() else { return };
203    let url = format!("{}/clientconfig", server_base());
204    let Ok(resp_val) = wasm_bindgen_futures::JsFuture::from(window.fetch_with_str(&url)).await else { return };
205    let Ok(resp) = resp_val.dyn_into::<web_sys::Response>() else { return };
206    if !resp.ok() { return; }
207    let Ok(text_promise) = resp.text() else { return };
208    let Ok(text_val) = wasm_bindgen_futures::JsFuture::from(text_promise).await else { return };
209    let Some(text) = text_val.as_string() else { return };
210    let Ok(v) = js_sys::JSON::parse(&text) else { return };
211    if jget(&v, "telemetry").as_bool().unwrap_or(false) {
212        TELEMETRY_ON.store(true, std::sync::atomic::Ordering::Relaxed);
213        install_error_listeners();
214        log::info!("error telemetry enabled -> {}/telemetry", server_base());
215    }
216    if jget(&v, "feedback").as_bool().unwrap_or(false) {
217        set_body_class("feedback-on", true); // reveals the 💬 button (hidden by default)
218        wire_feedback_ui();
219    }
220}

222/// Forward uncaught JS errors + unhandled promise rejections to /telemetry (wasm panics go via the
223/// panic hook). Only installed when telemetry is enabled.
224fn install_error_listeners() {
225    let Some(window) = web_sys::window() else { return };
226    let on_err = Closure::wrap(Box::new(move |e: web_sys::Event| {
227        let msg = e
228            .dyn_ref::<web_sys::ErrorEvent>()
229            .map(|ee| ee.message())
230            .unwrap_or_else(|| "js error".to_string());
231        fire_event("/telemetry", telemetry_json("js-error", &msg));
232    }) as Box<dyn FnMut(web_sys::Event)>);
233    let _ = window.add_event_listener_with_callback("error", on_err.as_ref().unchecked_ref());
234    on_err.forget();
235    let on_rej = Closure::wrap(Box::new(move |_e: web_sys::Event| {
236        fire_event("/telemetry", telemetry_json("unhandledrejection", "unhandled promise rejection"));
237    }) as Box<dyn FnMut(web_sys::Event)>);
238    let _ = window.add_event_listener_with_callback("unhandledrejection", on_rej.as_ref().unchecked_ref());
239    on_rej.forget();
240}

242/// Wire the feedback widget: open/close the panel and POST the form to /feedback with light context.
243fn wire_feedback_ui() {
244    if let Some(btn) = by_id::<web_sys::HtmlElement>("feedback-open") {

248        add_listener(btn.as_ref(), "click", move |_e: web_sys::Event| set_body_class("feedback-open", false));
249    }
250    if let Some(btn) = by_id::<web_sys::HtmlElement>("feedback-send") {
251        add_listener(btn.as_ref(), "click", move |_e: web_sys::Event| submit_feedback());
252    }
253}

255fn submit_feedback() {
256    let kind = by_id::<web_sys::HtmlSelectElement>("feedback-type").map(|s| s.value()).unwrap_or_default();
257    let text = by_id::<web_sys::HtmlTextAreaElement>("feedback-text").map(|t| t.value()).unwrap_or_default();

262    }
263    let ds = dataset_id();
264    let backend = by_id::<web_sys::HtmlElement>("hud-backend").map(|e| e.inner_text()).unwrap_or_default();
265    let body = format!(
266        "{{\"type\":{},\"text\":{},\"email\":{},\"dataset\":{ds},\"backend\":{}}}",
267        json_str(&kind), json_str(&text), json_str(&email), json_str(&backend)
268    );
269    fire_event("/feedback", body);
270    if let Some(t) = by_id::<web_sys::HtmlTextAreaElement>("feedback-text") {
271        t.set_value("");
272    }
273    set_body_class("feedback-open", false);
274    show_status("thanks — your feedback was sent");
275}

277/// DBSCAN over a flat `[x,y,z, x,y,z, …]` buffer, returning per-point labels (`-1` = noise). The
278/// Web Worker calls this; exported so its wasm instance has it too. `progress` (if a JS function) is
279/// called `(visited, total)` ~every 1% so the worker can post progress back to the main thread.
280#[wasm_bindgen]
281pub fn cluster_dbscan_flat(flat: &[f32], min_pts: usize, eps: f32, progress: &JsValue) -> Vec<i32> {
282    // Reinterpret the flat xyz buffer as [f32;3] triples with no copy ([f32;3] and f32 share align 4).
283    // Truncate any trailing partial triple first — matches the old chunks_exact drop-remainder
284    // behavior and avoids a cast_slice length panic on malformed input.
285    let n = flat.len() - flat.len() % 3;
286    let pts: &[[f32; 3]] = bytemuck::cast_slice(&flat[..n]);
287    let total = pts.len() as f64;
288    let cb = progress.dyn_ref::<js_sys::Function>();
289    dbscan_with_progress(pts, eps, min_pts, |visited| {
290        if let Some(f) = cb {
291            let _ =
292                f.call2(&JsValue::NULL, &JsValue::from_f64(visited as f64), &JsValue::from_f64(total));
293        }
294    })
295    .0
296}

495    /// Match the surface (and depth texture) to the canvas's CSS size × device-pixel-ratio, so
496    /// the cloud stays crisp on HiDPI displays and tracks window resizes.
497    fn resize_to_display(&mut self) {
498        let (w, h) = display_size(&self.canvas);
499        if w == 0 || h == 0 || (w == self.config.width && h == self.config.height) {
500            return;
501        }
502        self.canvas.set_width(w);
503        self.canvas.set_height(h);
504        self.config.width = w;
505        self.config.height = h;
506        self.surface.configure(&self.device, &self.config);
507        self.depth_view = create_depth(&self.device, w, h);
508        self.needs_redraw = true; // a resize changes the surface — redraw next frame
509    }
510
511    /// Project each axis label's cube position to the canvas and position its DOM element.
512    fn update_labels(&self, aspect: f32) {
513        if self.labels.is_empty() {
514            return;
515        }
516        let vp = self.camera.view_proj(aspect); // glam::Mat4, with the axis-roll baked in
517        let (cw, ch) = (self.canvas.client_width() as f32, self.canvas.client_height() as f32);
518        for (el, w) in &self.labels {
519            let clip = vp * glam::Vec4::new(w[0], w[1], w[2], 1.0);
520            let style = el.style();
521            if clip.w <= 0.0 {
522                let _ = style.set_property("display", "none"); // behind the camera
523                continue;
524            }
525            let x = (clip.x / clip.w * 0.5 + 0.5) * cw;
526            let y = (1.0 - (clip.y / clip.w * 0.5 + 0.5)) * ch;
527            let _ = style.set_property("display", "block");
528            let _ = style.set_property(
529                "transform",
530                &format!("translate({x:.1}px,{y:.1}px) translate(-50%,-50%)"),
531            );
532        }
533    }
534
535    /// Rebuild the density grid from `cpu_points` if stale, upload it, and refresh the volume
536    /// uniform for this frame. No-op when volume is unsupported.
537    fn ensure_volume(&mut self, aspect: f32) {
538        if self.volume.is_none() {
539            return;
540        }
541        if self.vol_grid.is_none() {
542            self.vol_grid = Some(VolumeGrid::new(VOLUME_DIMS));
543            self.vol_needs_grid = true;
544        }
545        let grid = self.vol_grid.as_mut().unwrap();
546        if self.vol_needs_grid {
547            // Density of all resident points, filtered by the current MS mask (the spatial crop is
548            // applied in the shader via box_min/max). Deposit intensity*weight so the density is
549            // independent of the downsample ratio.
550            grid.clear();
551            let ms = self.params.ms_mask;
552            let floor = self.params.filter_min[3]; // intensity floor: dim points don't deposit
553            for pt in &self.cpu_points {
554                if pt.intensity < floor {
555                    continue;
556                }
557                let is_ms2 = pt.flags & GpuPoint::MS2_FLAG != 0;
558                if if is_ms2 { ms & 0b10 != 0 } else { ms & 0b01 != 0 } {
559                    grid.deposit(pt.pos, pt.intensity * pt.weight);
560                }
561            }
562            // Auto-range the density transfer to the grid's own percentiles (median..p99.9) — the
563            // point intensity range is a different scale, which is why the volume looked flat.
564            let (lo, hi) = grid.density_percentiles();
565            self.vol_i_min = lo;
566            self.vol_i_max = hi;
567            self.volume.as_ref().unwrap().upload(&self.queue, grid.to_f16_scaled());
568            self.vol_needs_grid = false;
569            show_status(""); // clear the "building volume…" notice
570        }
571        let p = &self.params;
572        let inv = self.camera.view_proj(aspect).inverse().to_cols_array_2d();
573        let vu = VolumeUniform {
574            inv_view_proj: inv,
575            box_min: [p.filter_min[0], p.filter_min[1], p.filter_min[2], 0.0],
576            box_max: [p.filter_max[0], p.filter_max[1], p.filter_max[2], 0.0],
577            // Log over the density percentiles + exposure 1.0 (mirrors the native auto-range).
578            transfer: [2.0, self.vol_i_min, self.vol_i_max, 1.0],
579            steps: self.vol_steps.max(1),
580            style: self.vol_style,
581            colormap_id: p.colormap_id,
582            n_colormaps: p.n_colormaps.max(1),
583            density_scale: grid.density_scale(),
584            focus: 0.0,
585            _pad: [0.0; 2],
586        };
587        self.volume.as_ref().unwrap().update_uniform(&self.queue, &vu);
588    }

590    /// Render one frame. `now` is the RAF timestamp (ms). Returns `false` if the loop should stop.
591    fn frame(&mut self, now: f64) -> bool {
592        // Smoothed FPS, pushed to the HUD a few times a second.
593        if self.fps_t > 0.0 {
594            let dt = (now - self.fps_t) as f32;
595            if dt > 0.0 {
596                let inst = 1000.0 / dt;
597                self.fps_ema = if self.fps_ema == 0.0 { inst } else { self.fps_ema * 0.9 + inst * 0.1 };
598            }
599        }
600        self.fps_t = now;
601        self.hud_tick = self.hud_tick.wrapping_add(1);
602        if self.hud_tick % 12 == 0 {
603            set_text("hud-fps", &format!("{:.0} fps", self.fps_ema));
604        }
605        // Recompute the displayed (filter-surviving) count when the filter/draw-count changed.
606        if self.filter_dirty {
607            self.needs_redraw = true; // a filter/param/load change must trigger a redraw
608            let shown = recount_displayed(self);
609            let resident = self.renderer.resident() as usize;
610            let txt = if self.points_base.is_empty() {
611                format!("{} · demo", group_short(shown))
612            } else if shown >= resident {
613                group_short(resident) // nothing hidden — show the loaded pool
614            } else {
615                format!("{} / {}", group_short(shown), group_short(resident))
616            };
617            set_text("hud-points", &txt);
618            self.filter_dirty = false;
619        }
620
621        // Idle skip: if nothing changed and nothing is animating, keep the rAF loop alive but skip
622        // the (multi-million-point) draw — idle GPU ~0. IDLE_HEARTBEAT forces a periodic redraw so a
623        // missed wake-source recovers in a fraction of a second rather than freezing the view.
624        let animating = self.auto_rotate || self.dragging.is_some() || self.acq_playing;
625        if !self.needs_redraw && !animating && self.hud_tick % IDLE_HEARTBEAT != 0 {
626            return true;
627        }
628
629        if self.auto_rotate {
630            self.camera.yaw += 0.005;
631        }
632        let (w, h) = (self.config.width as f32, self.config.height as f32);
633        let cam = self.camera.to_uniform(w / h, [w, h]);
634        self.renderer.update_camera(&self.queue, &cam);
635        self.axes.update_camera(&self.queue, &cam);
636        self.windows.update_camera(&self.queue, &cam);
637        if self.show_windows {
638            // Cull the windows to the live crop box (mirror the point filter), like the native overlay.
639            self.windows
640                .update_filter(&self.queue, self.params.filter_min, self.params.filter_max, 0.0);
641        }
642        self.update_labels(w / h);
643        let volume_view = self.view_mode == ViewMode::Volume && self.volume.is_some();
644        if volume_view {
645            self.ensure_volume(w / h);
646        } else {
647            self.renderer.update_params(&self.queue, &self.params);
648        }
649
650        let surface_tex = match self.surface.get_current_texture() {
651            Ok(t) => t,
652            // Transient: just skip this frame.
653            Err(wgpu::SurfaceError::Timeout) => return true,
654            // Stale/lost (e.g. resize, tab restore): reconfigure and retry once next frame.
655            Err(wgpu::SurfaceError::Outdated) | Err(wgpu::SurfaceError::Lost) => {
656                self.surface.configure(&self.device, &self.config);
657                return true;
658            }
659            Err(wgpu::SurfaceError::OutOfMemory) => {
660                log::error!("surface out of memory — stopping render loop");
661                show_status("GPU out of memory — render loop stopped.");
662                return false;
663            }
664        };
665        let view = surface_tex.texture.create_view(&Default::default());
666        let mut enc = self.device.create_command_encoder(&Default::default());
667        if !volume_view {
668            self.renderer.prepare(&self.queue, &mut enc);
669        }
670        {
671            let mut rpass = enc.begin_render_pass(&wgpu::RenderPassDescriptor {
672                label: Some("scene"),
673                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
674                    view: &view,
675                    resolve_target: None,
676                    ops: wgpu::Operations {
677                        load: wgpu::LoadOp::Clear(wgpu::Color {
678                            r: 0.02,
679                            g: 0.02,
680                            b: 0.05,
681                            a: 1.0,
682                        }),
683                        store: wgpu::StoreOp::Store,
684                    },
685                })],
686                depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment {
687                    view: &self.depth_view,
688                    depth_ops: Some(wgpu::Operations {
689                        load: wgpu::LoadOp::Clear(1.0),
690                        store: wgpu::StoreOp::Store,
691                    }),
692                    stencil_ops: None,
693                }),
694                timestamp_writes: None,
695                occlusion_query_set: None,
696            });
697            if volume_view {
698                self.volume.as_ref().unwrap().render(&mut rpass);
699            } else {
700                // Cluster hues mush under additive blending — force opaque while cluster-colored.
701                let pm = if self.params.color_mode >= 1 {
702                    PointMode::StructuralOpaque
703                } else {
704                    self.point_mode
705                };
706                self.renderer.render(&mut rpass, pm);
707            }
708            self.axes.render(&mut rpass);
709            if self.show_windows {
710                self.windows.render(&mut rpass);
711            }
712        }
713        self.queue.submit(std::iter::once(enc.finish()));
714        surface_tex.present();
715        self.needs_redraw = false; // drawn this frame; go idle until the next change/interaction
716        true
717    }
718}

736    // The surface is created only after a backend wins, so the canvas context matches it.

1083/// Create a wgpu surface + device, preferring WebGPU and falling back to WebGL2 if its
1084/// `requestDevice` fails. WebGPU enumerates its adapter globally (no canvas context yet), so we only
1085/// create the WebGPU surface once the device succeeds; WebGL2 instead needs the canvas surface to
1086/// find its adapter at all.
1087///
1088/// KNOWN ISSUE (deferred): on current Chrome/Firefox the WebGPU `requestDevice` always fails with
1089/// `The limit "maxInterStageShaderComponents" ... is not recognized`, so we silently run on WebGL2.
1090/// wgpu 22.1.0 sends that spec-removed limit and the value can't be suppressed via its API; the fix
1091/// requires bumping to wgpu 23+ (latest 29), which forces a coordinated egui/egui-wgpu/egui-winit
1092/// (0.29 -> matching) + winit upgrade across the whole render stack and the native GUI. WebGL2 is
1093/// fully functional, so this is a post-demo upgrade, not a blocker.
1094async fn init_gpu(canvas: &web_sys::HtmlCanvasElement) -> Result<Gpu, String> {
1095    // 1) WebGPU — global adapter; create the surface only if the device is actually obtained.
1096    let wgpu_inst = wgpu::Instance::new(wgpu::InstanceDescriptor {
1097        backends: wgpu::Backends::BROWSER_WEBGPU,
1098        ..Default::default()
1099    });
1100    let webgpu_err: Option<String> = match wgpu_inst
1101        .request_adapter(&wgpu::RequestAdapterOptions {
1102            power_preference: wgpu::PowerPreference::HighPerformance,
1103            force_fallback_adapter: false,
1104            compatible_surface: None,
1105        })
1106        .await
1107    {
1108        Some(adapter) => {
1109            let limits = web_limits(&adapter, false);
1110            match adapter.request_device(&device_desc(limits), None).await {
1111                // Surface creation can also fail here — treat that like any WebGPU failure and
1112                // fall through to WebGL2 rather than giving up (the canvas has no context yet).
1113                Ok((device, queue)) => match wgpu_inst
1114                    .create_surface(wgpu::SurfaceTarget::Canvas(canvas.clone()))
1115                {
1116                    Ok(surface) => return Ok((wgpu_inst, surface, adapter, device, queue, false)),
1117                    Err(e) => Some(format!("WebGPU create_surface: {e}")),
1118                },
1119                Err(e) => Some(format!("WebGPU requestDevice: {e:?}")),
1120            }
1121        }
1122        None => Some("no WebGPU adapter".to_string()),
1123    };
1124    if let Some(e) = &webgpu_err {
1125        log::warn!("{e}; falling back to WebGL2");
1126    }
1127
1128    // 2) WebGL2 — needs the canvas surface to enumerate its adapter.
1129    let gl_inst = wgpu::Instance::new(wgpu::InstanceDescriptor {
1130        backends: wgpu::Backends::GL,
1131        ..Default::default()
1132    });
1133    let surface = gl_inst
1134        .create_surface(wgpu::SurfaceTarget::Canvas(canvas.clone()))
1135        .map_err(|e| format!("create_surface (webgl2): {e}"))?;
1136    let adapter = gl_inst
1137        .request_adapter(&wgpu::RequestAdapterOptions {
1138            power_preference: wgpu::PowerPreference::HighPerformance,
1139            force_fallback_adapter: false,
1140            compatible_surface: Some(&surface),
1141        })
1142        .await
1143        .ok_or_else(|| format!("no WebGL2 adapter (after {webgpu_err:?})"))?;
1144    let limits = web_limits(&adapter, true);
1145    let (device, queue) = adapter
1146        .request_device(&device_desc(limits), None)
1147        .await
1148        .map_err(|e| format!("WebGL2 requestDevice: {e:?} (after {webgpu_err:?})"))?;
1149    Ok((gl_inst, surface, adapter, device, queue, true))
1150}

1330/// Decode one 24-byte wire record into a normalized `GpuPoint` (`_pad[0]` = peptide_id, `flags` bit0 =
1331/// fragment). `c` must be exactly 24 bytes.
1332fn decode_acq_record(c: &[u8], m: &AcqMeta) -> GpuPoint {
1333    let f = |o: usize| f32::from_le_bytes([c[o], c[o + 1], c[o + 2], c[o + 3]]);
1334    let u = |o: usize| u32::from_le_bytes([c[o], c[o + 1], c[o + 2], c[o + 3]]);
1335    GpuPoint {
1336        pos: [
1337            m.mz.normalize(f(0) as f64),
1338            m.im.normalize(f(4) as f64),
1339            m.rt.normalize(f(8) as f64),
1340        ],
1341        intensity: f(12),
1342        weight: 1.0,
1343        flags: u(20),
1344        _pad: [u(16), 0],
1345    }
1346}

1348/// GET a binary acquisition endpoint and return the raw bytes. Aborts after 20s so a hung/half-open
1349/// sidecar can't leave a prefetch in flight forever (which would stall the play-loop).
1350async fn fetch_acq_bytes(url: &str) -> Result<Vec<u8>, String> {
1351    let window = web_sys::window().ok_or("no window")?;
1352    let opts = web_sys::RequestInit::new();
1353    if let Ok(controller) = web_sys::AbortController::new() {
1354        opts.set_signal(Some(&controller.signal()));
1355        let cb = wasm_bindgen::closure::Closure::once_into_js(move || controller.abort());
1356        let _ = window.set_timeout_with_callback_and_timeout_and_arguments_0(cb.unchecked_ref(), 20_000);
1357    }
1358    let resp_val = wasm_bindgen_futures::JsFuture::from(window.fetch_with_str_and_init(url, &opts))
1359        .await
1360        .map_err(|e| format!("fetch failed: {e:?}"))?;
1361    let resp: web_sys::Response = resp_val.dyn_into().map_err(|_| "not a Response".to_string())?;
1362    if !resp.ok() {
1363        return Err(format!("HTTP {}", resp.status()));
1364    }
1365    let buf = wasm_bindgen_futures::JsFuture::from(
1366        resp.array_buffer().map_err(|e| format!("array_buffer: {e:?}"))?,
1367    )
1368    .await
1369    .map_err(|e| format!("body read failed: {e:?}"))?;
1370    Ok(js_sys::Uint8Array::new(&buf).to_vec())
1371}

1373/// Fetch a single frame (`<base>/acq/frame?id=N`, `id` a 0-based frame index) as `GpuPoint`s.
1374#[allow(dead_code)] // kept for debugging / single-frame probes; playback uses the batch path
1375async fn fetch_acq_frame(base: &str, id: u32, m: &AcqMeta) -> Result<Vec<GpuPoint>, String> {
1376    let bytes = fetch_acq_bytes(&format!("{base}/acq/frame?id={id}")).await?;
1377    if bytes.len() % 24 != 0 {
1378        return Err(format!("acq frame body not 24-byte aligned ({} bytes)", bytes.len()));
1379    }
1380    Ok(bytes.chunks_exact(24).map(|c| decode_acq_record(c, m)).collect())
1381}

1383/// Fetch a parallel batch (`<base>/acq/frames?start=N&count=K`) and split it into per-frame
1384/// `GpuPoint` lists. Wire layout: `u32 count`, `count × [u32 frame_index, u32 n_peaks]`, then the
1385/// concatenated 24-byte records for each frame in order.
1386async fn fetch_acq_frames(base: &str, start: u32, count: u32, m: &AcqMeta) -> Result<Vec<Vec<GpuPoint>>, String> {
1387    let bytes = fetch_acq_bytes(&format!("{base}/acq/frames?start={start}&count={count}")).await?;
1388    if bytes.len() < 4 {
1389        return Err("acq batch response too short".into());
1390    }
1391    let rd_u32 = |o: usize| u32::from_le_bytes([bytes[o], bytes[o + 1], bytes[o + 2], bytes[o + 3]]);
1392    // The server returns exactly the frames we asked for; validating up front bounds `n` (so the
1393    // header arithmetic below can't overflow) and stops a malformed/short batch from desyncing the
1394    // fetch frontier (we advanced `acq_fetch_next` by `count` already).
1395    let n = rd_u32(0) as usize;
1396    if n != count as usize {
1397        return Err(format!("acq batch returned {n} frames, requested {count}"));
1398    }
1399    let table_end = 4 + n * 8; // n == count <= ACQ_BATCH, so no overflow
1400    if bytes.len() < table_end {
1401        return Err("acq batch header truncated".into());
1402    }
1403    let mut counts = Vec::with_capacity(n);
1404    for i in 0..n {
1405        let idx = rd_u32(4 + i * 8);
1406        if idx != start + i as u32 {
1407            return Err(format!("acq batch frame_index {idx} != expected {}", start + i as u32));
1408        }
1409        counts.push(rd_u32(4 + i * 8 + 4) as usize);
1410    }
1411    let recs = &bytes[table_end..];
1412    let total: usize = counts.iter().sum();
1413    if total.checked_mul(24) != Some(recs.len()) {
1414        return Err(format!("acq batch body {} B != {total} records", recs.len()));
1415    }
1416    let mut frames = Vec::with_capacity(n);
1417    let mut off = 0usize;
1418    for c in counts {
1419        let mut pts = Vec::with_capacity(c);
1420        for k in 0..c {
1421            pts.push(decode_acq_record(&recs[off + k * 24..off + k * 24 + 24], m));
1422        }
1423        off += c * 24;
1424        frames.push(pts);
1425    }
1426    Ok(frames)
1427}

1429/// Await `ms` milliseconds (wasm has no blocking sleep): a `setTimeout`-backed Promise.

1438/// Enter live-acquisition playback: clear the cloud, colour by peptide id (`_pad[0]`), and start the
1439/// async play-loop. Toggling again stops it. (Re-uses the cluster-colour shader path; the cube bounds
1440/// stay as-is for now — points are already normalized to the acq ranges by `fetch_acq_frame`.)
1441fn toggle_acquisition(gfx: &Rc<RefCell<Gfx>>) {
1442    let start = {
1443        let g = &mut *gfx.borrow_mut();
1444        if g.acq_meta.is_none() {
1445            return;
1446        }
1447        g.acq_playing = !g.acq_playing;
1448        if g.acq_playing {
1449            // Fresh ring buffer bounded to a trailing window: the initial buffer was sized to the
1450            // loaded (or demo) cloud, and unbounded accumulation would grind the GPU after ~1 min.
1451            let cap = g.n_cap.min(ACQ_RING_POINTS).max(1) as u32;
1452            g.renderer =
1453                PointCloudRenderer::new(&g.device, &g.queue, g.config.format, DEPTH_FORMAT, cap, g.supports_compaction);
1454            g.renderer.enable_ring(); // append wraps + overwrites oldest -> bounded resident
1455            g.renderer.set_draw_count(u32::MAX);
1456            // Stash the static view so Stop / completion can restore it in place (chunk 2d). The
1457            // mem::take also empties cpu_points (acquisition drives the GPU buffer directly); snapshot
1458            // the display params + view that the lines below are about to overwrite.
1459            g.acq_saved = Some(AcqStash {
1460                points: std::mem::take(&mut g.cpu_points),
1461                params: g.params,
1462                view: g.view_mode,
1463            });
1464            g.view_mode = ViewMode::Points; // acquisition is a point cloud; never the volume path
1465            // Acquisition replaces cpu_points/_pad[0], so any clustering is now invalid: a stale
1466            // cluster Run or a clicked cluster row would index the emptied cpu_points and panic.
1467            // Invalidate + clear it (and bump data_gen so an already-posted worker result is dropped).
1468            g.data_gen = g.data_gen.wrapping_add(1);
1469            invalidate_clusters_mut(g);
1470            g.params.color_mode = 3; // colour by _pad[0] = peptide_id + size-by-intensity (acq shader path)
1471            g.params.ms_mask = 0b11; // show BOTH MS1 (precursors) and MS2 (fragments) — the whole point
1472            // Intensity transfer so faint peaks shrink/recede: sqrt over [1, i_p99].
1473            let i_p99 = g.acq_meta.as_ref().map(|m| m.i_p99.max(2.0)).unwrap_or(1000.0);
1474            g.params.transfer = [1.0, 1.0, i_p99, 1.0];
1475            // Sync speed to whatever the selector currently shows (covers browser form-restore where
1476            // the <select> value survives a reload but g.acq_speed reset to its default).
1477            g.acq_speed = by_id::<web_sys::HtmlSelectElement>("acq-speed")
1478                .and_then(|s| s.value().parse::<f64>().ok())
1479                .map_or(1.0, |v| v.max(0.05));
1480            g.acq_cursor = 0;
1481            g.acq_buffer.clear();
1482            g.acq_fetch_next = 0;
1483            g.acq_fetching = false;
1484            g.acq_gen = g.acq_gen.wrapping_add(1); // invalidate any in-flight prefetch from a prior run
1485            g.filter_dirty = true;
1486        }
1487        g.acq_playing
1488    };
1489    if start {
1490        // Tear down the cluster panel/results UI and the save-ready state alongside the state reset.
1491        set_body_class("has-clusters", false);
1492        set_save_ready(false);
1493    }
1494    set_body_class("cluster-color", start); // the intensity colorbar is meaningless while peptide-coloured
1495    set_text("acq-play", if start { "⏸ Stop playback" } else { "▶ Live acquisition" });
1496    if start {
1497        wasm_bindgen_futures::spawn_local(play_loop(gfx.clone()));
1498    } else {
1499        // In-place return to the static dataset cloud (chunk 2d): rebuild from the stash, no reload.
1500        restore_static_view(gfx);
1501        refresh_controls(gfx); // re-enable focus/back/load now that playback is stopped
1502        show_status("playback stopped — returned to the dataset view");
1503    }
1504}

1506/// Stop's in-place return to the static dataset cloud (chunk 2d): rebuild a normal (non-ring)
1507/// renderer from the points stashed at playback start, restore the display params + view that
1508/// acquisition overwrote, and drop any in-flight prefetch. No page reload, no server round-trip.
1509/// Coloring is forced back to intensity (entry destroyed the clustering, so the restored points
1510/// have no cluster ids to color by — mirrors `apply_load`). A no-op if nothing was stashed.
1511fn restore_static_view(gfx: &Rc<RefCell<Gfx>>) {
1512    let mut g = gfx.borrow_mut();
1513    let g = &mut *g;
1514    let Some(stash) = g.acq_saved.take() else { return };
1515    // Rebuild the GPU buffer at the stashed cloud's size (not the trailing-ring cap) and re-upload.
1516    let cap = stash.points.len().min(g.n_cap).max(1) as u32;
1517    let mut renderer =
1518        PointCloudRenderer::new(&g.device, &g.queue, g.config.format, DEPTH_FORMAT, cap, g.supports_compaction);
1519    let _ = renderer.append(&g.queue, &stash.points);
1520    g.renderer = renderer;
1521    g.cpu_points = stash.points;
1522    g.params = stash.params; // restores transfer/exposure, MS mask, and the spatial crops
1523    g.params.color_mode = 0; // intensity (no clusters survive acquisition entry)
1524    g.view_mode = stash.view;
1525    g.vol_needs_grid = true; // the volume density grid is rebuilt from the restored points if shown
1526    g.data_gen = g.data_gen.wrapping_add(1);
1527    g.acq_gen = g.acq_gen.wrapping_add(1); // a still-in-flight prefetch sees a stale gen and drops
1528    g.acq_buffer.clear();
1529    g.acq_fetching = false;
1530    g.filter_dirty = true;
1531}

1533/// Detached prefetch: fetch one parallel batch and push its decoded frames into `acq_buffer`, then
1534/// clear the in-flight flag so the play-loop can request the next. Single-flight (the sidecar
1535/// serializes builds anyway). Forward-only: the streamed frames are consumed as displayed.
1536async fn acq_prefetch(gfx: Rc<RefCell<Gfx>>, start: u32, count: u32, meta: AcqMeta, gen: u32) {
1537    let base = acq_service_url();
1538    let result = fetch_acq_frames(&base, start, count, &meta).await;
1539    let g = &mut *gfx.borrow_mut();
1540    if g.acq_gen != gen {
1541        return; // a newer playback session started while we were in flight — drop this result
1542    }
1543    g.acq_fetching = false;
1544    match result {
1545        Ok(frames) => {
1546            if g.acq_playing {
1547                g.acq_buffer.extend(frames);
1548            }
1549        }
1550        Err(e) => {
1551            log::warn!("acq batch [{start},{}) failed: {e}", start + count);
1552            // Roll the fetch frontier back so this batch is RETRIED next tick instead of leaving a
1553            // permanent hole in the played stream. Only when we're still the current frontier (no
1554            // newer advance/reset owns it). A transient sidecar hiccup self-heals; a dead sidecar
1555            // stalls visibly rather than silently skipping frames.
1556            if g.acq_fetch_next == start + count {
1557                g.acq_fetch_next = start;
1558            }
1559        }
1560    }
1561}

1563/// The async playback loop. Each tick it (1) kicks a prefetch batch if the buffer is running low and
1564/// none is in flight, and (2) pops one ready frame and appends it at the (speed-scaled) device
1565/// cadence. Fetching overlaps display, so playback isn't build-limited — it runs at real time with
1566/// headroom to fast-forward, bounded by the prefetch throughput.
1567async fn play_loop(gfx: Rc<RefCell<Gfx>>) {
1568    let gen = gfx.borrow().acq_gen; // this session's id; a newer toggle supersedes us
1569    loop {
1570        enum Next {
1571            Displayed { cur: u32, n_frames: u32, rt: f64, resident: u32, cadence_ms: i32 },
1572            Wait,
1573            Done(u32),
1574        }
1575        let (prefetch, next) = {
1576            let g = &mut *gfx.borrow_mut();
1577            if !g.acq_playing || g.acq_gen != gen {
1578                return;
1579            }
1580            let Some(meta) = g.acq_meta.clone() else { return };
1581            // (1) keep the buffer filled ahead of the cursor.
1582            let prefetch = if !g.acq_fetching
1583                && g.acq_buffer.len() < ACQ_LOOKAHEAD
1584                && g.acq_fetch_next < meta.n_frames
1585            {
1586                let start = g.acq_fetch_next;
1587                let count = ACQ_BATCH.min(meta.n_frames - start);
1588                g.acq_fetch_next += count;
1589                g.acq_fetching = true;
1590                Some((start, count, meta.clone()))
1591            } else {
1592                None
1593            };
1594            // (2) display one buffered frame.
1595            let next = if let Some(frame) = g.acq_buffer.pop_front() {
1596                if !frame.is_empty() {
1597                    g.renderer.append(&g.queue, &frame);
1598                    g.renderer.set_draw_count(u32::MAX);
1599                    g.filter_dirty = true;
1600                }
1601                g.acq_cursor += 1;
1602                Next::Displayed {
1603                    cur: g.acq_cursor,
1604                    n_frames: meta.n_frames,
1605                    rt: g.acq_cursor as f64 * meta.rt_cycle_length,
1606                    resident: g.renderer.resident(),
1607                    cadence_ms: (meta.rt_cycle_length * 1000.0 / g.acq_speed.max(0.05)) as i32,
1608                }
1609            } else if g.acq_fetch_next >= meta.n_frames && !g.acq_fetching {
1610                Next::Done(meta.n_frames) // buffer drained and nothing left to fetch
1611            } else {
1612                Next::Wait // buffer momentarily empty — prefetch is catching up
1613            };
1614            (prefetch, next)
1615        };
1616        if let Some((start, count, meta)) = prefetch {
1617            wasm_bindgen_futures::spawn_local(acq_prefetch(gfx.clone(), start, count, meta, gen));
1618        }
1619        match next {
1620            Next::Displayed { cur, n_frames, rt, resident, cadence_ms } => {
1621                show_status(&format!(
1622                    "▶ frame {cur} / {n_frames} · RT {rt:.0}s · {} pts",
1623                    group_short(resident as usize),
1624                ));
1625                sleep_ms(cadence_ms).await;
1626            }
1627            Next::Wait => sleep_ms(8).await,
1628            Next::Done(n) => {
1629                gfx.borrow_mut().acq_playing = false;
1630                // Same in-place return as a manual Stop, so completion never strands the user on a
1631                // frozen trailing-window cloud with no way back but a reload.
1632                restore_static_view(&gfx);
1633                refresh_controls(&gfx);
1634                set_body_class("cluster-color", false); // restore the intensity colorbar (manual stop does too)
1635                set_text("acq-play", "▶ Live acquisition");
1636                show_status(&format!("acquisition complete — {n} frames; returned to the dataset view"));
1637                return;
1638            }
1639        }
1640    }
1641}

1738/// POST the (already axis-scaled) flat positions to the Python clustering service and read back the
1739/// per-point labels. Same wire format as the wasm worker: float32 xyz triples in, int32 labels out.
1740async fn fetch_cluster(svc: &str, flat: &[f32], query: &str) -> Result<Vec<i32>, String> {
1741    let window = web_sys::window().ok_or("no window")?;
1742    let arr = js_sys::Float32Array::from(flat);
1743    let opts = web_sys::RequestInit::new();
1744    opts.set_method("POST");
1745    opts.set_body(&arr.buffer());
1746    // Abort after 30s so a hung/half-open cluster service can't pin the Run button forever (the user
1747    // can still Stop sooner). Generous enough for a legitimate sklearn run over the capped point set.
1748    if let Ok(controller) = web_sys::AbortController::new() {
1749        opts.set_signal(Some(&controller.signal()));
1750        let cb = wasm_bindgen::closure::Closure::once_into_js(move || controller.abort());
1751        let _ = window.set_timeout_with_callback_and_timeout_and_arguments_0(cb.unchecked_ref(), 30000);
1752    }
1753    let url = with_query(svc, query);
1754    let resp_val = wasm_bindgen_futures::JsFuture::from(window.fetch_with_str_and_init(&url, &opts))

1756        .map_err(|e| format!("fetch failed: {e:?}"))?;
1757    let resp: web_sys::Response = resp_val.dyn_into().map_err(|_| "not a Response".to_string())?;
1758    if !resp.ok() {
1759        return Err(format!("HTTP {}", resp.status()));
1760    }

1767        if ab.byte_length() % 4 != 0 {
1768            return Err("cluster response not int32-aligned".to_string());
1769        }
1770    }
1771    Ok(js_sys::Int32Array::new(&buf).to_vec())

1774/// Fetch packed `GpuPoint` bytes from the server and reinterpret them (the server already
1775/// normalized positions to the `[-1, 1]` cube, so they upload as-is).
1776async fn fetch_points(url: &str) -> Result<Vec<GpuPoint>, String> {
1777    let window = web_sys::window().ok_or("no window")?;
1778    let resp_val = wasm_bindgen_futures::JsFuture::from(window.fetch_with_str(url))
1779        .await
1780        .map_err(|e| format!("fetch failed: {e:?}"))?;
1781    let resp: web_sys::Response = resp_val.dyn_into().map_err(|_| "not a Response".to_string())?;
1782    if !resp.ok() {
1783        return Err(format!("HTTP {}", resp.status()));
1784    }
1785    let buf = wasm_bindgen_futures::JsFuture::from(
1786        resp.array_buffer().map_err(|e| format!("array_buffer: {e:?}"))?,
1787    )
1788    .await
1789    .map_err(|e| format!("body read failed: {e:?}"))?;
1790
1791    let bytes = js_sys::Uint8Array::new(&buf);
1792    let len = bytes.length();
1793    let stride = std::mem::size_of::<GpuPoint>() as u32;
1794    if len == 0 {
1795        return Ok(Vec::new()); // a valid empty region (the caller decides what to show)
1796    }
1797    if len % stride != 0 {
1798        return Err(format!("body {len} bytes is not a multiple of the {stride}-byte point stride"));
1799    }
1800    let mut pts = vec![GpuPoint::zeroed(); (len / stride) as usize];
1801    // dst is exactly `len` bytes and 4-aligned (it's a Vec<GpuPoint>); copy_to requires equal len.
1802    let dst: &mut [u8] = bytemuck::cast_slice_mut(&mut pts);
1803    bytes.copy_to(dst);
1804    Ok(pts)
1805}

1833/// Derive the `/meta` URL from the `/points` URL: replace only the trailing path endpoint and keep

1845        Some(q) => format!("{meta_path}?{q}"),
1846        None => meta_path,
1847    }

2006/// Build the window-overlay line vertices: each footprint as an `(m/z × 1/K0)` rectangle at
2007/// `DIA_WINDOW_RT_SLICES` RT slices, normalized to the focused region's `bounds` and colored by group.
2008fn build_window_verts(
2009    rects: &[WindowRect],
2010    bounds: [(f64, f64); 3],
2011    max_group: u32,
2012    mask: u32,
2013) -> Vec<LineVertex> {
2014    let norm = |val: f64, axis: usize| -> f32 {
2015        let (lo, hi) = bounds[axis];
2016        (((val - lo) / (hi - lo).max(1e-12)) * 2.0 - 1.0) as f32
2017    };
2018    let mut v: Vec<LineVertex> = Vec::new();
2019    let denom = max_group.max(1);
2020    for r in rects {
2021        // Per-group visibility: groups 1..=32 gate on their mask bit; >32 and group 0 always show.
2022        if r.group >= 1 && r.group <= 32 && (mask & (1u32 << (r.group - 1))) == 0 {
2023            continue;
2024        }
2025        // Order each axis first: a footprint can arrive descending (1/K0 falls as scan rises, so
2026        // the scan_lo/scan_hi -> 1/K0 conversion yields im0 > im1). Without this the clamp below
2027        // produces hi < lo and the next check drops EVERY window — the overlay vanishes while the
2028        // legend (which only needs max_window_group) still shows.
2029        let (rmz0, rmz1) = (r.mz0.min(r.mz1), r.mz0.max(r.mz1));
2030        let (rim0, rim1) = (r.im0.min(r.im1), r.im0.max(r.im1));
2031        // Clamp the axis-aligned footprint to the region so a partly-off-region window clips cleanly
2032        // (the per-vertex shader cull would otherwise leave stubs); drop windows fully outside.
2033        let (mz0r, mz1r) = (rmz0.max(bounds[0].0), rmz1.min(bounds[0].1));
2034        let (im0r, im1r) = (rim0.max(bounds[1].0), rim1.min(bounds[1].1));
2035        if mz1r <= mz0r || im1r <= im0r {
2036            continue;
2037        }
2038        let color = tims_viewer::render::colormap::group_color(r.group, denom);
2039        let (mz0, mz1) = (norm(mz0r, 0), norm(mz1r, 0));
2040        let (im0, im1) = (norm(im0r, 1), norm(im1r, 1));
2041        for i in 0..DIA_WINDOW_RT_SLICES {
2042            let rt_real =
2043                bounds[2].0 + (bounds[2].1 - bounds[2].0) * (i as f64 + 0.5) / DIA_WINDOW_RT_SLICES as f64;
2044            let rt = norm(rt_real, 2);
2045            let corners = [[mz0, im0, rt], [mz1, im0, rt], [mz1, im1, rt], [mz0, im1, rt]];
2046            for e in 0..4 {
2047                v.push(LineVertex::new(corners[e], color));
2048                v.push(LineVertex::new(corners[(e + 1) % 4], color));
2049            }
2050        }
2051    }
2052    v
2053}

2236/// Auto-set the transfer function + exposure from intensity percentiles, mirroring the native
2237/// viewer's `auto_transfer_points`, so the additive cloud is legible (not blown out) on load.
2238fn apply_auto_transfer(params: &mut ParamsUniform, m: &MetaInfo) {
2239    if m.i_p50 <= 0.0 {
2240        return;
2241    }
2242    let p1 = m.i_p1.max(1.0) as f32;
2243    let p50 = m.i_p50.max(m.i_p1) as f32;
2244    let p99 = m.i_p99.max(m.i_p50 * 1.0001) as f32;
2245    params.transfer[0] = 1.0; // sqrt: lifts the mid-range without flattening bright peaks
2246    params.transfer[1] = p1; // i_min
2247    params.transfer[2] = (p99 * 4.0).max(p50 * 1.0001); // i_max — headroom above p99 for peaks
2248    // Exposure solves a single median splat to ~0.35 alpha; weight = downsample stride.
2249    let weight = (m.stride as f32).max(1.0);
2250    params.transfer[3] = (0.35 / (params.opacity.max(0.05) * weight)).clamp(0.02, 4.0);
2251}

2381/// CPU copy, re-apply auto-transfer + bounds, reset crops/floor/Display, and rebind the DOM. Used
2382/// by every reload (Load budget now; Focus/Back next). Returns the resident count.
2383fn apply_load(gfx: &Rc<RefCell<Gfx>>, meta: Option<MetaInfo>, pts: Vec<GpuPoint>) {
2384    let n = {
2385        let mut g = gfx.borrow_mut();

2387        let mut renderer = PointCloudRenderer::new(
2388            &g.device,
2389            &g.queue,
2390            g.config.format,
2391            DEPTH_FORMAT,
2392            cap,
2393            g.supports_compaction,
2394        );
2395        let n = renderer.append(&g.queue, &pts);
2396        g.renderer = renderer;
2397        let mut cpu = pts;
2398        cpu.truncate(cap as usize);
2399        g.cpu_points = cpu;
2400        g.data_gen = g.data_gen.wrapping_add(1); // invalidate any in-flight clustering Run
2401        g.vol_needs_grid = true; // the volume density grid is rebuilt from the new points
2402        g.params.color_mode = 0; // the fresh buffer has no cluster ids
2403        g.clustered = false;
2404        g.cluster_stats = None;
2405        g.cluster_params_json = None;
2406        g.cluster_run_params = None;
2407        set_save_ready(false); // ④ Save: results gone with the clustering

2430            g.params.filter_max[a] = 1.0;

2459    set_body_class("cluster-color", false);

2474#[derive(Clone, Copy)]
2475enum StackOp {

2477    Pop,
2478    Keep,
2479}

2481/// The focus-lens load primitive: re-fetch a region and rebuild the GPU buffer. `budget == None`
2482/// (the root stack entry) clamps to `n_cap` — the request is always a capped region query, never a
2483/// bare `/points`, so Back-to-root can't pull the server's full `--budget` and OOM the wasm heap.
2484/// The stack op is applied only after the load succeeds, so a debounced or failed load never desyncs
2485/// the stack from the displayed view. Debounced via `reloading`.
2486async fn load_region(gfx: Rc<RefCell<Gfx>>, region: Region, budget: Option<usize>, op: StackOp) {
2487    {
2488        let mut g = gfx.borrow_mut();
2489        if g.acq_playing {
2490            // A region load rebuilds the renderer; doing that under the running play-loop would
2491            // desync the two. Make the user stop playback first.
2492            show_status("stop live acquisition before focusing a region");
2493            return;
2494        }
2495        if g.reloading || g.focus_stack.is_empty() {
2496            return;
2497        }
2498        g.reloading = true;
2499    }
2500    refresh_controls(&gfx); // disable focus/back/load while a load is in flight
2501    let (base, n_cap) = {
2502        let g = gfx.borrow();
2503        (g.points_base.clone(), g.n_cap)
2504    };
2505    // Always cap n (None = the root entry) so Back-to-root can't fetch the server's full --budget
2506    // uncapped and OOM the wasm heap, mirroring the initial load.
2507    let q = region_query(&region, budget.unwrap_or(n_cap).min(n_cap));
2508    let purl = with_query(&base, &q);
2509    let murl = with_query(&meta_url(&base), &q);
2510    show_status("loading…");
2511    // Require BOTH meta and points: applying points without their matching meta would leave
2512    // axis_bounds stale while the buffer is normalized to the new region (breaks nested Focus).
2513    let meta = fetch_meta(&murl).await;
2514    let pts = fetch_points(&purl).await;
2515    gfx.borrow_mut().reloading = false;
2516    match (meta, pts) {
2517        (Ok(meta), Ok(pts)) if !pts.is_empty() => {
2518            apply_load(&gfx, Some(meta), pts);
2519            // Maintain the stack from the SERVER-canonical bounds (apply_load set axis_bounds), so
2520            // `stack.last()` always matches the displayed view.
2521            let canonical = gfx.borrow().axis_bounds.map(|b| Region {
2522                mz: b[0],
2523                im: b[1],
2524                rt: b[2],
2525                imin: region.imin,
2526            });
2527            {
2528                let mut g = gfx.borrow_mut();
2529                match op {
2530                    StackOp::Push => {
2531                        if let Some(c) = canonical {
2532                            g.focus_stack.push((c, budget));
2533                        }
2534                    }
2535                    StackOp::Pop => {
2536                        if g.focus_stack.len() > 1 {
2537                            g.focus_stack.pop();
2538                        }
2539                    }
2540                    StackOp::Keep => {
2541                        if let (Some(c), Some(top)) = (canonical, g.focus_stack.last_mut()) {
2542                            *top = (c, budget);
2543                        }
2544                    }
2545                }
2546            }
2547            show_status("");
2548        }
2549        (Ok(_), Ok(_)) => show_status("region is empty — nothing matched"),
2550        (Err(e), _) | (_, Err(e)) => show_status(&format!("load failed: {e}")),
2551    }
2552    refresh_controls(&gfx);
2553}

2555/// The current crop box + intensity floor, mapped from the normalized cube back to real units of
2556/// the current view — the region to focus on. `None` for the demo (no real bounds) or a degenerate
2557/// (zero/negative-width or non-finite) box that would divide-by-zero when re-normalized.
2558fn compute_focus_region(g: &Gfx) -> Option<Region> {
2559    let b = g.axis_bounds?;
2560    let p = &g.params;
2561    let (fmin, fmax, ms) = (p.filter_min, p.filter_max, p.ms_mask);
2562    let limit = (g.renderer.drawn() as usize).min(g.cpu_points.len());
2563    // Fit the box to the points that actually survive the active 4D filter (window + intensity
2564    // floor + MS) — NOT to the crop sliders. Otherwise points removed by the intensity floor still
2565    // "hold the box open" to the full extent. Survivors already lie inside any active spatial crop.
2566    let mut lo = [f32::INFINITY; 3];
2567    let mut hi = [f32::NEG_INFINITY; 3];
2568    let mut n = 0u64;
2569    for pt in &g.cpu_points[..limit] {
2570        let pos = pt.pos;
2571        if pos[0] < fmin[0]
2572            || pos[0] > fmax[0]
2573            || pos[1] < fmin[1]
2574            || pos[1] > fmax[1]
2575            || pos[2] < fmin[2]
2576            || pos[2] > fmax[2]
2577            || pt.intensity < fmin[3]
2578        {
2579            continue;
2580        }
2581        let is_ms2 = (pt.flags & GpuPoint::MS2_FLAG) != 0;
2582        if !(if is_ms2 { ms & 0b10 != 0 } else { ms & 0b01 != 0 }) {
2583            continue;
2584        }
2585        for a in 0..3 {
2586            lo[a] = lo[a].min(pos[a]);
2587            hi[a] = hi[a].max(pos[a]);
2588        }
2589        n += 1;
2590    }
2591    if n == 0 {
2592        return None; // nothing visible to focus on
2593    }
2594    // Normalized survivor bbox -> real units, with a small margin (and a floor so a thin axis can't
2595    // collapse to zero width).
2596    let lerp = |rng: (f64, f64), norm: f32| rng.0 + ((norm as f64 + 1.0) * 0.5) * (rng.1 - rng.0);
2597    let mk = |axis: usize| {
2598        let m = ((hi[axis] - lo[axis]) * 0.02).max(0.01);
2599        let (a, c) = ((lo[axis] - m).max(-1.0), (hi[axis] + m).min(1.0));
2600        (lerp(b[axis], a), lerp(b[axis], c))
2601    };
2602    let (mz, im, rt) = (mk(0), mk(1), mk(2));
2603    let ok = |r: (f64, f64)| r.0.is_finite() && r.1.is_finite() && r.1 > r.0;
2604    if !ok(mz) || !ok(im) || !ok(rt) {
2605        return None;
2606    }
2607    Some(Region { mz, im, rt, imin: p.filter_min[3].max(0.0) })
2608}

2610/// Focus on the current crop+floor box: re-load it at full budget and push it on success.
2611fn do_focus(gfx: &Rc<RefCell<Gfx>>) {
2612    if gfx.borrow().reloading {
2613        return;
2614    }
2615    let region = match compute_focus_region(&gfx.borrow()) {
2616        Some(r) => r,
2617        None => {
2618            show_status("crop a region (or set a floor) before focusing");
2619            return;
2620        }
2621    };
2622    let budget = gfx.borrow().n_cap;
2623    wasm_bindgen_futures::spawn_local(load_region(gfx.clone(), region, Some(budget), StackOp::Push));
2624}

2626/// Re-load the region one level down the stack; pop only after it succeeds.
2627fn do_back(gfx: &Rc<RefCell<Gfx>>) {
2628    let target = {
2629        let g = gfx.borrow();

2634    };
2635    wasm_bindgen_futures::spawn_local(load_region(gfx.clone(), target.0, target.1, StackOp::Pop));
2636}

2813/// Run DBSCAN on the filtered resident points, write cluster ids into the GPU buffer, and colour by
2814/// cluster. Synchronous (blocks the main thread) — Focus first if the filtered set is large.
2815fn run_clustering(gfx: &Rc<RefCell<Gfx>>) {
2816    if gfx.borrow().acq_playing {
2817        show_status("stop live acquisition before clustering");
2818        return;
2819    }
2820    if gfx.borrow().reloading {
2821        show_status("loading — try clustering again in a moment");
2822        return;
2823    }
2824    if gfx.borrow().cluster_pending.is_some() {
2825        show_status("clustering already running…");
2826        return;
2827    }
2828    // MS1 (precursors) and MS2 (fragments) live in different m/z spaces — clustering them together is
2829    // meaningless. Require a single MS level.
2830    if gfx.borrow().params.ms_mask == 0b11 {
2831        show_status("pick MS1 or MS2 to cluster — clustering both levels together isn't meaningful");
2832        return;
2833    }
2834    // Gather the filtered survivors (spatial crop + intensity floor + MS) and their cube positions.
2835    let (idx, positions, eps, min_pts, gen, run_params) = {
2836        let g = gfx.borrow();
2837        let p = &g.params;
2838        let (fmin, fmax, ms) = (p.filter_min, p.filter_max, p.ms_mask);
2839        // Grow on demand — do NOT reserve cpu_points.len(): a cropped/focused cluster keeps only a
2840        // small subset, so reserving the whole (up to 32M-point) cloud would waste hundreds of MB of
2841        // wasm heap and raise OOM risk (codex review). The reallocs are bounded by the survivor count.
2842        let mut idx: Vec<usize> = Vec::new();
2843        let mut positions: Vec<[f32; 3]> = Vec::new();
2844        for (i, pt) in g.cpu_points.iter().enumerate() {
2845            let pos = pt.pos;
2846            if pos[0] < fmin[0]
2847                || pos[0] > fmax[0]
2848                || pos[1] < fmin[1]
2849                || pos[1] > fmax[1]
2850                || pos[2] < fmin[2]
2851                || pos[2] > fmax[2]
2852                || pt.intensity < fmin[3]
2853            {
2854                continue;
2855            }
2856            let is_ms2 = pt.flags & GpuPoint::MS2_FLAG != 0;
2857            if !(if is_ms2 { ms & 0b10 != 0 } else { ms & 0b01 != 0 }) {
2858                continue;
2859            }
2860            idx.push(i);
2861            positions.push(pos);
2862        }
2863        // Per-axis equi-distancing (the shared, unit-tested helper in tims_viewer::cluster): calibrated
2864        // to CLUSTER_EPS_REF so the live eps scales the reach proportionally, and — crucially —
2865        // region-INDEPENDENT, so eps spans a fixed physical reach on every axis (cycles / scans /
2866        // peak-widths) regardless of focus crop. (The metric only; labels/stats use unscaled points.)
2867        let eps = g.cluster_eps;
2868        let scales = match g.axis_bounds {
2869            Some(bounds) => cluster_axis_scales(&ScaleInputs {
2870                bounds,
2871                eps_ref: CLUSTER_EPS_REF,
2872                cycle_duration: g.cycle_duration,
2873                im_per_scan: g.im_per_scan,
2874                mz_resolution: MZ_RESOLUTION,
2875                rt_cycles: g.cluster_rt_cycles,
2876                im_scans: g.cluster_im_scans,
2877                mz_peak_widths: g.cluster_mz_peak_widths,
2878            }),
2879            None => AxisScales { mz: 1.0, im: 1.0, rt: 1.0 },
2880        };
2881        for p in positions.iter_mut() {
2882            p[0] *= scales.mz;
2883            p[1] *= scales.im;
2884            p[2] *= scales.rt;
2885        }
2886        // Snapshot the parameters used (for the export JSON), so a slider edit mid-run can't skew it.
2887        let run_params = ClusterRunParams {
2888            dataset_id: g.dataset_id, // resolved/clamped id, not the raw URL value
2889            method: g.cluster_method,
2890            python: g.use_python_cluster,
2891            eps,
2892            min_pts: g.cluster_min_pts,
2893            min_cluster_size: g.cluster_min_cluster_size,
2894            hdb_min_samples: g.cluster_hdb_min_samples,
2895            selection_eps: g.cluster_selection_eps,
2896            rt_cycles: g.cluster_rt_cycles,
2897            im_scans: g.cluster_im_scans,
2898            mz_peak_widths: g.cluster_mz_peak_widths,
2899            ms_mask: g.params.ms_mask,
2900            floor: g.params.filter_min[3],
2901            region: g.axis_bounds,
2902            cycle_duration: g.cycle_duration,
2903            im_per_scan: g.im_per_scan,
2904        };
2905        (idx, positions, eps, g.cluster_min_pts, g.data_gen, run_params)
2906    };
2907    if idx.is_empty() {
2908        show_status("nothing to cluster — adjust the filter");
2909        return;
2910    }
2911    if idx.len() > CLUSTER_CAP {
2912        show_status(&format!(
2913            "{} points — Focus a region first (cap {})",
2914            group(idx.len()),
2915            group(CLUSTER_CAP)
2916        ));
2917        return;
2918    }
2919    gfx.borrow_mut().cluster_run_params = Some(run_params); // before dispatch, for the export JSON
2920
2921    // Python backend (when the "Python (sklearn)" toggle is on — auto-enabled by the startup probe):
2922    // POST the scaled points to the service and colour by its labels. Else the wasm path below.
2923    if gfx.borrow().use_python_cluster {
2924        let svc = cluster_service_url();
2925        let (job, query) = {
2926            let mut g = gfx.borrow_mut();
2927            g.next_cluster_job = g.next_cluster_job.wrapping_add(1);
2928            let q = match g.cluster_method {
2929                ClusterMethod::Hdbscan => format!(
2930                    "method=hdbscan&mcs={}&ms={}&cse={}",
2931                    g.cluster_min_cluster_size, g.cluster_hdb_min_samples, g.cluster_selection_eps
2932                ),
2933                ClusterMethod::Dbscan => format!("method=dbscan&eps={eps}&min={min_pts}"),
2934            };
2935            g.cluster_pending = Some((idx, gen, g.next_cluster_job));
2936            (g.next_cluster_job, q)
2937        };
2938        set_cluster_running(true);
2939        show_status("clustering (python)…");
2940        let gfx2 = gfx.clone();
2941        wasm_bindgen_futures::spawn_local(async move {
2942            // Reinterpret the [f32;3] positions as a flat f32 slice with no intermediate copy
2943            // (fetch_cluster's Float32Array::from does the one unavoidable copy into JS).
2944            let result = fetch_cluster(&svc, bytemuck::cast_slice(&positions), &query).await;
2945            // Only consume the reply if it's still THIS job — a Stop / invalidate / newer Run
2946            // supersedes it (the un-aborted fetch would otherwise steal the new pending job).
2947            let mut g = gfx2.borrow_mut();
2948            if !matches!(&g.cluster_pending, Some((_, _, pj)) if *pj == job) {
2949                return;
2950            }
2951            let (idx, gen, _) = g.cluster_pending.take().unwrap();
2952            drop(g);
2953            match result {
2954                Ok(labels) if labels.len() == idx.len() => {
2955                    let k = labels.iter().copied().max().map_or(0, |m| (m + 1).max(0) as usize);
2956                    finish_clustering(&gfx2, idx, labels, k, gen);
2957                }
2958                Ok(_) => {
2959                    set_cluster_running(false);
2960                    show_status("python clustering: label/point count mismatch");
2961                }
2962                Err(e) => {
2963                    set_cluster_running(false);
2964                    show_status(&format!("python clustering failed: {e}"));
2965                }
2966            }
2967        });
2968        return;
2969    }
2970
2971    show_status("clustering…");
2972    // Small inputs cluster in well under a second, so run them on the main thread (deferred one
2973    // macrotask so the status paints): instant, with none of the worker's one-time wasm spin-up
2974    // latency. Only large inputs — where DBSCAN would freeze the tab for seconds — go to the worker,
2975    // whose init cost is then negligible against the run and whose progress bar actually has time to
2976    // move.
2977    if idx.len() > WORKER_THRESHOLD {
2978        if let Some(worker) = ensure_cluster_worker(gfx) {
2979            // Zero-copy view of the positions as a flat f32 slice (post_cluster_job's
2980            // Float32Array::from does the one copy into JS).
2981            let flat: &[f32] = bytemuck::cast_slice(&positions);
2982            let job = {
2983                let mut g = gfx.borrow_mut();
2984                g.next_cluster_job = g.next_cluster_job.wrapping_add(1);
2985                g.next_cluster_job
2986            };
2987            if post_cluster_job(&worker, flat, eps, min_pts, job) {
2988                gfx.borrow_mut().cluster_pending = Some((idx, gen, job));
2989                set_cluster_progress(Some(0.0)); // worker reports ticks back; bar advances live
2990                set_cluster_running(true); // button becomes "Stop clustering"
2991                return;
2992            }
2993        }
2994    }
2995    // Main thread (small input, or worker unavailable): defer one macrotask so the status paints,
2996    // then run the (brief) blocking DBSCAN.
2997    let gfx = gfx.clone();
2998    defer(move || {
2999        let (labels, k) = dbscan(&positions, eps, min_pts);
3000        finish_clustering(&gfx, idx, labels, k, gen);
3001    });
3002}

3017/// Return the persistent DBSCAN worker, creating it on first use. `None` (→ main-thread fallback) if
3018/// workers are unavailable or the wasm/glue URLs can't be discovered.
3019fn ensure_cluster_worker(gfx: &Rc<RefCell<Gfx>>) -> Option<web_sys::Worker> {
3020    {
3021        let g = gfx.borrow();
3022        if g.cluster_worker_failed {
3023            return None; // a prior runtime failure latched — stay on the main thread
3024        }
3025        if let Some(w) = g.cluster_worker.clone() {
3026            return Some(w);
3027        }
3028    }
3029    let w = create_cluster_worker(gfx)?;
3030    gfx.borrow_mut().cluster_worker = Some(w.clone());

3034/// Build the DBSCAN worker from a Blob module that re-imports the Trunk-built wasm (URLs read from
3035/// the page's preload links, so Trunk's content hashing is handled).
3036fn create_cluster_worker(gfx: &Rc<RefCell<Gfx>>) -> Option<web_sys::Worker> {
3037    let doc = document()?;
3038    // The glue JS + wasm URLs from Trunk's preload links (absolute via the element `href` property).
3039    let link_href = |sel: &str| -> Option<String> {
3040        doc.query_selector(sel)
3041            .ok()
3042            .flatten()
3043            .and_then(|e| e.dyn_into::<web_sys::HtmlLinkElement>().ok())
3044            .map(|l| l.href())
3045    };
3046    let glue = link_href("link[rel='modulepreload']")?;
3047    let wasm = link_href("link[rel='preload'][as='fetch']")?;
3048    let src = CLUSTER_WORKER_SRC.replace("__GLUE__", &glue).replace("__WASM__", &wasm);
3049
3050    let parts = js_sys::Array::of1(&JsValue::from_str(&src));
3051    let bag = web_sys::BlobPropertyBag::new();
3052    bag.set_type("text/javascript"); // module workers require a JS MIME on the script
3053    let blob = web_sys::Blob::new_with_str_sequence_and_options(&parts, &bag).ok()?;
3054    let url = web_sys::Url::create_object_url_with_blob(&blob).ok()?;
3055    let opts = web_sys::WorkerOptions::new();
3056    opts.set_type(web_sys::WorkerType::Module);
3057    let worker = web_sys::Worker::new_with_options(&url, &opts).ok()?;
3058    let _ = web_sys::Url::revoke_object_url(&url); // the worker has captured the script
3059
3060    // Reply handler: only finish the reply whose job id matches the pending one (a superseded job's
3061    // late reply carries a stale id and is ignored).
3062    let gfx_msg = gfx.clone();
3063    let cb = wasm_bindgen::closure::Closure::<dyn FnMut(web_sys::MessageEvent)>::new(
3064        move |e: web_sys::MessageEvent| {
3065            let data = e.data();
3066            let get = |k: &str| js_sys::Reflect::get(&data, &JsValue::from_str(k)).ok();
3067            let reply_job = get("job").and_then(|v| v.as_f64()).map(|f| f as u64);
3068            let job_matches = |g: &Gfx| {
3069                matches!((&g.cluster_pending, reply_job), (Some((_, _, pj)), Some(rj)) if *pj == rj)
3070            };
3071            // Progress tick (no labels): advance the bar if it's for the current job.
3072            if let Some(p) = get("progress").and_then(|v| v.as_f64()) {
3073                if job_matches(&gfx_msg.borrow()) {
3074                    set_cluster_progress(Some(p as f32));
3075                }
3076                return;
3077            }
3078            let Some(labels) = get("labels")
3079                .and_then(|v| v.dyn_into::<js_sys::Int32Array>().ok())
3080                .map(|a| a.to_vec())
3081            else {
3082                return;
3083            };
3084            if !job_matches(&gfx_msg.borrow()) {
3085                return; // no pending / superseded job — drop this reply
3086            }
3087            // Take the pending job in its OWN statement so the borrow_mut RefMut is dropped before
3088            // finish_clustering (which re-borrows mutably). In edition 2021 an `if let` scrutinee
3089            // temporary lives to the end of the body, so borrowing inline here would double-borrow
3090            // and panic on every worker completion (the Python path drops its borrow for the same
3091            // reason).
3092            let pending = gfx_msg.borrow_mut().cluster_pending.take();
3093            if let Some((idx, gen, _)) = pending {
3094                if labels.len() != idx.len() {
3095                    return; // malformed reply — never index past the survivor set
3096                }
3097                let k = labels.iter().copied().max().map_or(0, |m| (m + 1).max(0) as usize);
3098                finish_clustering(&gfx_msg, idx, labels, k, gen);
3099            }
3100        },
3101    );
3102    worker.set_onmessage(Some(cb.as_ref().unchecked_ref()));
3103    cb.forget(); // the worker lives for the app's lifetime
3104
3105    // A runtime failure (e.g. the worker's wasm can't load, or a message can't deserialize) latches
3106    // `cluster_worker_failed` so every later Run uses the main thread instead of rebuilding a broken
3107    // worker; also release any stuck pending job.
3108    for ev in ["error", "messageerror"] {
3109        let gfx_err = gfx.clone();
3110        add_listener(worker.as_ref(), ev, move |_e: web_sys::Event| {
3111            let mut g = gfx_err.borrow_mut();
3112            g.cluster_pending = None;
3113            g.cluster_worker = None;
3114            g.cluster_worker_failed = true;
3115            drop(g);
3116            set_cluster_progress(None);
3117            set_cluster_running(false);
3118            show_status("clustering worker unavailable — runs on the main thread");
3119        });
3120    }
3121    Some(worker)
3122}

3195/// Aggregate per-cluster stats from the DBSCAN labels (intensity weighted by `weight` so it's
3196/// downsample-independent; extents in real units via `axis_bounds`).
3197fn compute_cluster_stats(
3198    idx: &[usize],
3199    labels: &[i32],
3200    k: usize,
3201    pts: &[GpuPoint],
3202    bounds: Option<[(f64, f64); 3]>,
3203) -> ClusterStats {
3204    let mut sizes = vec![0f64; k];
3205    let mut int_sum = vec![0f64; k];
3206    let mut lo = vec![[f32::INFINITY; 3]; k];
3207    let mut hi = vec![[f32::NEG_INFINITY; 3]; k];
3208    let (mut signal_pts, mut noise_pts) = (0u64, 0u64);
3209    let (mut signal_int, mut noise_int) = (0f64, 0f64);
3210    for (j, &l) in labels.iter().enumerate() {
3211        let pt = &pts[idx[j]];
3212        let contrib = pt.intensity as f64 * pt.weight as f64; // widen before multiply
3213        if l < 0 {
3214            noise_pts += 1;
3215            noise_int += contrib;
3216        } else {
3217            let c = l as usize;
3218            sizes[c] += 1.0;
3219            int_sum[c] += contrib;
3220            signal_pts += 1;
3221            signal_int += contrib;
3222            for a in 0..3 {
3223                lo[c][a] = lo[c][a].min(pt.pos[a]);
3224                hi[c][a] = hi[c][a].max(pt.pos[a]);
3225            }
3226        }
3227    }
3228    // Normalized [-1,1] span -> real-unit extent on each axis.
3229    let extent = |a: usize| -> Vec<f64> {
3230        (0..k)
3231            .map(|c| {
3232                if sizes[c] == 0.0 {
3233                    return 0.0;
3234                }
3235                let span = (hi[c][a] - lo[c][a]).max(0.0) as f64;
3236                match bounds {
3237                    Some(b) => span * 0.5 * (b[a].1 - b[a].0).abs(),
3238                    None => span,
3239                }
3240            })
3241            .collect()
3242    };
3243    ClusterStats {
3244        k,
3245        signal_pts,
3246        noise_pts,
3247        signal_int,
3248        noise_int,
3249        mz_extent: extent(0),
3250        im_extent: extent(1),
3251        rt_extent: extent(2),
3252        sizes,
3253        int_sum,
3254    }
3255}

3319/// Bin non-negative finite values into `n` equal-width bins over `[0, max]`. Everything collapses to
3320/// bin 0 when every value is 0.
3321fn hist_bins(values: &[f64], n: usize) -> Vec<u32> {
3322    let n = n.max(1);
3323    let mut h = vec![0u32; n];
3324    let finite: Vec<f64> = values.iter().copied().filter(|v| v.is_finite() && *v >= 0.0).collect();
3325    let max = finite.iter().copied().fold(0.0f64, f64::max);
3326    if max <= 0.0 {
3327        h[0] = finite.len() as u32;
3328        return h;
3329    }
3330    for &v in &finite {
3331        // equal-width: scale by n then clamp to n-1 (vs *(n-1), which leaves the last bin only for
3332        // exact maxima).
3333        let b = ((v / max * n as f64) as usize).min(n - 1);
3334        h[b] += 1;
3335    }
3336    h
3337}

3339/// Compact display of a (possibly huge / non-finite) f64 sum — avoids the silent `as usize`
3340/// saturation when intensity totals are large.
3341fn fmt_count(x: f64) -> String {
3342    if !x.is_finite() {
3343        return "—".into();
3344    }
3345    if x >= 1e9 {
3346        format!("{:.1}G", x / 1e9)
3347    } else if x >= 1e6 {
3348        format!("{:.1}M", x / 1e6)
3349    } else if x >= 1e4 {
3350        format!("{:.0}k", x / 1e3)
3351    } else {
3352        format!("{:.0}", x)
3353    }
3354}

3805/// Render a 2D density projection to its `<canvas>` with the inferno colormap (sqrt-scaled). The
3806/// projection is row-major `x + bins*y`; y is flipped so the low bin sits at the bottom.
3807fn render_heatmap(canvas_id: &str, data: &[u32], bins: usize) {
3808    if bins == 0 || data.len() < bins * bins {
3809        return;
3810    }
3811    let Some(canvas) = by_id::<web_sys::HtmlCanvasElement>(canvas_id) else {
3812        return;
3813    };
3814    canvas.set_width(bins as u32);
3815    canvas.set_height(bins as u32);
3816    let Some(ctx) = canvas
3817        .get_context("2d")
3818        .ok()
3819        .flatten()
3820        .and_then(|c| c.dyn_into::<web_sys::CanvasRenderingContext2d>().ok())
3821    else {
3822        return;
3823    };
3824    let max = data.iter().copied().max().unwrap_or(1).max(1) as f32;
3825    let mut buf = vec![0u8; bins * bins * 4];
3826    for y in 0..bins {
3827        let row = bins - 1 - y; // flip: low bin at the bottom
3828        for x in 0..bins {
3829            let t = (data[x + bins * y] as f32 / max).sqrt();
3830            let rgb = colormap_sample(1, t); // inferno
3831            let i = (row * bins + x) * 4;
3832            buf[i] = rgb[0];
3833            buf[i + 1] = rgb[1];
3834            buf[i + 2] = rgb[2];
3835            buf[i + 3] = 255;
3836        }
3837    }
3838    if let Ok(img) = web_sys::ImageData::new_with_u8_clamped_array_and_sh(
3839        wasm_bindgen::Clamped(&buf),
3840        bins as u32,
3841        bins as u32,
3842    ) {
3843        let _ = ctx.put_image_data(&img, 0.0, 0.0);
3844    }
3845}

3882/// Map a box on projection `proj` (x,y fractions of the map) to a 4D region of the current view.
3883/// The map renders low-at-bottom, so the y fractions flip; the unselected axis keeps the full view
3884/// range, and the current intensity floor carries through.
3885fn region_from_box(g: &Gfx, proj: usize, x: (f64, f64), y: (f64, f64)) -> Option<Region> {
3886    let b = g.axis_bounds?;
3887    let lerp = |rng: (f64, f64), f: f64| rng.0 + f * (rng.1 - rng.0);
3888    let (yl, yh) = (1.0 - y.1, 1.0 - y.0); // top->bottom px, but low-at-bottom display
3889    let imin = g.params.filter_min[3].max(0.0);
3890    let (mz, im, rt) = match proj {
3891        0 => ((lerp(b[0], x.0), lerp(b[0], x.1)), (lerp(b[1], yl), lerp(b[1], yh)), b[2]),
3892        1 => ((lerp(b[0], x.0), lerp(b[0], x.1)), b[1], (lerp(b[2], yl), lerp(b[2], yh))),
3893        2 => (b[0], (lerp(b[1], x.0), lerp(b[1], x.1)), (lerp(b[2], yl), lerp(b[2], yh))),
3894        _ => return None,
3895    };
3896    Some(Region { mz, im, rt, imin })
3897}

3926/// Wire box-select on the three projection maps: drag a rectangle → Focus that 4D region.
3927fn bind_maps(gfx: &Rc<RefCell<Gfx>>) {

3929        return; // demo fallback: maps render but can't drive a server focus
3930    }

3932        return;
3933    };
3934    // Shared drag state: (proj index, start-x frac, start-y frac).
3935    let drag: Rc<RefCell<Option<(usize, f64, f64)>>> = Rc::new(RefCell::new(None));

3937    for proj in 0..3usize {

3949    {

3953                return;
3954            };

3956            if e.buttons() == 0 {

3959                return;
3960            }
3961            if let Some(mw) = by_id::<web_sys::HtmlElement>(&format!("mw-{proj}")) {
3962                let (nx, ny) = map_frac(&mw, &e);

3972            };

3975                return;
3976            };

3985                if g.reloading || !g.maps_ok {
3986                    return; // mid-load, or maps don't reflect the current view
3987                }
3988            }

3990                let budget = gfx.borrow().n_cap;

3992                    gfx.clone(),

3994                    Some(budget),
3995                    StackOp::Push,
3996                ));
3997            }
3998        });

4078/// Count points surviving the active 4D filter (spatial window + intensity floor + MS mask) among
4079/// the drawn prefix of the resident pool. Camera-independent (frustum culling excluded), so it is
4080/// recomputed only when the filter or draw count changes — not every frame.
4081fn recount_displayed(g: &Gfx) -> usize {
4082    let p = &g.params;
4083    let limit = (g.renderer.drawn() as usize).min(g.cpu_points.len());
4084    let (fmin, fmax, ms) = (p.filter_min, p.filter_max, p.ms_mask);
4085    let mut shown = 0usize;
4086    for pt in &g.cpu_points[..limit] {
4087        let pos = pt.pos;
4088        if pos[0] < fmin[0]
4089            || pos[0] > fmax[0]
4090            || pos[1] < fmin[1]
4091            || pos[1] > fmax[1]
4092            || pos[2] < fmin[2]
4093            || pos[2] > fmax[2]
4094        {
4095            continue;
4096        }
4097        if pt.intensity < fmin[3] {
4098            continue;
4099        }
4100        let is_ms2 = (pt.flags & GpuPoint::MS2_FLAG) != 0;
4101        let pass = if is_ms2 { ms & 0b10 != 0 } else { ms & 0b01 != 0 };
4102        if pass {
4103            shown += 1;
4104        }
4105    }
4106    shown
4107}

4378/// Bind a lo/hi range-slider pair to `filter_min/max[axis]` (normalized cube `[-1, 1]`). The two
4379/// thumbs push each other rather than crossing: dragging `lo` past `hi` bumps `hi` up (and vice
4380/// versa), so the lower bound can never exceed the upper.
4381fn bind_crop(gfx: &Rc<RefCell<Gfx>>, prefix: &str, axis: usize) {
4382    let (lo, hi) = match (
4383        by_id::<web_sys::HtmlInputElement>(&format!("{prefix}-lo")),
4384        by_id::<web_sys::HtmlInputElement>(&format!("{prefix}-hi")),
4385    ) {
4386        (Some(lo), Some(hi)) => (lo, hi),
4387        _ => return,
4388    };
4389    // Lower thumb: if it passes the upper, push the upper up to meet it.
4390    {
4391        let (gfx_c, lo_c, hi_c, p) = (gfx.clone(), lo.clone(), hi.clone(), prefix.to_string());
4392        add_listener(lo.as_ref(), "input", move |_e: web_sys::Event| {
4393            let _ = lo_c.style().set_property("z-index", "5"); // active thumb on top
4394            let _ = hi_c.style().set_property("z-index", "4");
4395            if lo_c.value_as_number() > hi_c.value_as_number() {
4396                hi_c.set_value(&lo_c.value());
4397            }
4398            crop_apply(&gfx_c, axis, &p, lo_c.value_as_number(), hi_c.value_as_number());
4399        });
4400    }
4401    // Upper thumb: if it drops below the lower, push the lower down to meet it.
4402    {
4403        let (gfx_c, lo_c, hi_c, p) = (gfx.clone(), lo.clone(), hi.clone(), prefix.to_string());
4404        add_listener(hi.as_ref(), "input", move |_e: web_sys::Event| {
4405            let _ = hi_c.style().set_property("z-index", "5"); // active thumb on top
4406            let _ = lo_c.style().set_property("z-index", "4");
4407            if hi_c.value_as_number() < lo_c.value_as_number() {
4408                lo_c.set_value(&hi_c.value());
4409            }
4410            crop_apply(&gfx_c, axis, &p, lo_c.value_as_number(), hi_c.value_as_number());
4411        });
4412    }
4413}

4416/// readout, and move the highlighted fill bar to sit between the two thumbs.

4418    let a = (lo_val / 1000.0) as f32 * 2.0 - 1.0;

4420    let bounds = {

4423        g.params.filter_max[axis] = b;
4424        g.filter_dirty = true;
4425        invalidate_clusters_mut(&mut g);
4426        g.axis_bounds
4427    };
4428    set_text(&format!("{prefix}-v"), &crop_label(bounds, axis, a, b));

4430        .and_then(|d| d.get_element_by_id(&format!("{prefix}-fill")))

4433        let st = fill.style();
4434        let _ = st.set_property("left", &format!("{:.2}%", lo_val / 10.0));
4435        let _ = st.set_property("width", &format!("{:.2}%", (hi_val - lo_val).max(0.0) / 10.0));
4436    }
4437}

4479            g.needs_redraw = true;

4482    }
4483    {
4484        let gfx = gfx.clone();
4485        add_listener(window.as_ref(), "mousemove", move |e: web_sys::MouseEvent| {

4490                    g.camera.orbit(dx, dy);
4491                } else {
4492                    g.camera.pan(dx, dy);
4493                }

4551/// A synthetic point cloud in the normalized cube `[-1, 1]^3`: a helix plus two gaussian blobs,
4552/// intensities spanning the default log transfer range. Deterministic (seeded LCG, no `rand`).
4553fn demo_cloud() -> Vec<GpuPoint> {
4554    const N: usize = 250_000;
4555    let mut seed: u32 = 0x9E37_79B9;
4556    let mut rng = move || {
4557        seed = seed.wrapping_mul(1_664_525).wrapping_add(1_013_904_223);
4558        (seed >> 8) as f32 / (1u32 << 24) as f32 // [0, 1)
4559    };
4560    let blob = |c: [f32; 3], s: f32, r: &mut dyn FnMut() -> f32| {
4561        // crude gaussian via the central limit theorem (mean of 3 uniforms).
4562        let g = |r: &mut dyn FnMut() -> f32| (r() + r() + r() - 1.5) * 2.0 * s;
4563        [c[0] + g(r), c[1] + g(r), c[2] + g(r)]
4564    };
4565
4566    let mut pts = Vec::with_capacity(N);
4567    for i in 0..N {
4568        let m = (rng() * 3.0) as u32;
4569        let pos = match m {
4570            0 => {
4571                let t = i as f32 / N as f32;
4572                let ang = t * 40.0;
4573                [0.6 * ang.cos(), 0.6 * ang.sin(), t * 2.0 - 1.0]
4574            }
4575            1 => blob([-0.4, 0.3, -0.2], 0.16, &mut rng),
4576            _ => blob([0.5, -0.4, 0.4], 0.20, &mut rng),
4577        };
4578        let u = rng();
4579        let intensity = 50.0 + u * u * 50_000.0; // skew toward low, span ~[50, 5e4]
4580        pts.push(GpuPoint {
4581            pos,
4582            intensity,
4583            weight: 1.0,
4584            flags: (m == 0) as u32, // helix tagged MS2 for variety; both shown via ms_mask
4585            _pad: [GpuPoint::NO_CLUSTER, 0],
4586        });
4587    }
4588    pts
