1//! Annotation overlay renderer: line-list geometry (DDA precursor crosses / DIA
2//! isolation-window boxes) drawn over the scene, always visible (depth-test Always).
3
4use bytemuck::{Pod, Zeroable};
5use wgpu::util::DeviceExt;
6
7use super::uniforms::CameraUniform;
8
9/// A colored line vertex (pairs form segments). Color is per-vertex so the overlay can
10/// distinguish e.g. DIA window groups.
11#[repr(C)]
12#[derive(Clone, Copy, Debug, Pod, Zeroable)]
13pub struct LineVertex {
14    pub pos: [f32; 3],
15    pub color: [f32; 3],
16}
17
18impl LineVertex {
19    pub fn new(pos: [f32; 3], color: [f32; 3]) -> Self {
20        LineVertex { pos, color }
21    }
22}
23
24/// Normalized-cube window the overlay is clipped to (xyz; w unused).
25#[repr(C)]
26#[derive(Clone, Copy, Debug, Pod, Zeroable)]
27struct FilterUniform {
28    min: [f32; 4],
29    max: [f32; 4],
30}
31
32impl Default for FilterUniform {
33    fn default() -> Self {
34        FilterUniform {
35            min: [-1.0, -1.0, -1.0, 0.0],
36            max: [1.0, 1.0, 1.0, 0.0],
37        }
38    }
39}
40
41pub struct AnnotationRenderer {
42    vbuf: Option<wgpu::Buffer>,
43    vcount: u32,
44    camera_buf: wgpu::Buffer,
45    filter_buf: wgpu::Buffer,
46    bind_group: wgpu::BindGroup,
47    pipeline: wgpu::RenderPipeline,
48}
49
50impl AnnotationRenderer {
51    pub fn new(
52        device: &wgpu::Device,
53        color_format: wgpu::TextureFormat,
54        depth_format: wgpu::TextureFormat,
55    ) -> Self {
56        let camera_buf = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
57            label: Some("annotation-camera"),
58            contents: bytemuck::bytes_of(&CameraUniform::default()),
59            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
60        });
61        let filter_buf = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
62            label: Some("annotation-filter"),
63            contents: bytemuck::bytes_of(&FilterUniform::default()),
64            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
65        });
66        let uniform_entry = |binding| wgpu::BindGroupLayoutEntry {
67            binding,
68            visibility: wgpu::ShaderStages::VERTEX,
69            ty: wgpu::BindingType::Buffer {
70                ty: wgpu::BufferBindingType::Uniform,
71                has_dynamic_offset: false,
72                min_binding_size: None,
73            },
74            count: None,
75        };
76        let bind_layout = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
77            label: Some("annotation-bind-layout"),
78            entries: &[uniform_entry(0), uniform_entry(1)],
79        });
80        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
81            label: Some("annotation-bind-group"),
82            layout: &bind_layout,
83            entries: &[
84                wgpu::BindGroupEntry {
85                    binding: 0,
86                    resource: camera_buf.as_entire_binding(),
87                },
88                wgpu::BindGroupEntry {
89                    binding: 1,
90                    resource: filter_buf.as_entire_binding(),
91                },
92            ],
93        });
94        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
95            label: Some("annotation-shader"),
96            source: wgpu::ShaderSource::Wgsl(include_str!("../shaders/annotation.wgsl").into()),
97        });
98        let pipeline_layout = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
99            label: Some("annotation-pipeline-layout"),
100            bind_group_layouts: &[&bind_layout],
101            push_constant_ranges: &[],
102        });
103        let pipeline = device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
104            label: Some("annotation-pipeline"),
105            layout: Some(&pipeline_layout),
106            vertex: wgpu::VertexState {
107                module: &shader,
108                entry_point: "vs_main",
109                buffers: &[wgpu::VertexBufferLayout {
110                    array_stride: 24,
111                    step_mode: wgpu::VertexStepMode::Vertex,
112                    attributes: &wgpu::vertex_attr_array![0 => Float32x3, 1 => Float32x3],
113                }],
114                compilation_options: Default::default(),
115            },
116            fragment: Some(wgpu::FragmentState {
117                module: &shader,
118                entry_point: "fs_main",
119                targets: &[Some(wgpu::ColorTargetState {
120                    format: color_format,
121                    blend: None,
122                    write_mask: wgpu::ColorWrites::ALL,
123                })],
124                compilation_options: Default::default(),
125            }),
126            primitive: wgpu::PrimitiveState {
127                topology: wgpu::PrimitiveTopology::LineList,
128                ..Default::default()
129            },
130            // Scene pass owns a depth attachment; draw on top without testing/writing.
131            depth_stencil: Some(wgpu::DepthStencilState {
132                format: depth_format,
133                depth_write_enabled: false,
134                depth_compare: wgpu::CompareFunction::Always,
135                stencil: wgpu::StencilState::default(),
136                bias: wgpu::DepthBiasState::default(),
137            }),
138            multisample: wgpu::MultisampleState::default(),
139            multiview: None,
140            cache: None,
141        });
142        AnnotationRenderer {
143            vbuf: None,
144            vcount: 0,
145            camera_buf,
146            filter_buf,
147            bind_group,
148            pipeline,
149        }
150    }
151
152    pub fn update_camera(&self, queue: &wgpu::Queue, cam: &CameraUniform) {
153        queue.write_buffer(&self.camera_buf, 0, bytemuck::bytes_of(cam));
154    }
155
156    /// Set the normalized-cube window the overlay is clipped to. `focus` (0.0/1.0) re-fits
157    /// that window to the full cube — carried in the unused `min.w` slot for the shader.
158    pub fn update_filter(&self, queue: &wgpu::Queue, min: [f32; 4], max: [f32; 4], focus: f32) {
159        let mut m = min;
160        m[3] = focus;
161        queue.write_buffer(
162            &self.filter_buf,
163            0,
164            bytemuck::bytes_of(&FilterUniform { min: m, max }),
165        );
166    }
167
168    /// Upload colored line-list vertices (pairs = segments). Recreates the buffer.
169    pub fn upload(&mut self, device: &wgpu::Device, verts: &[LineVertex]) {
170        if verts.is_empty() {
171            self.vbuf = None;
172            self.vcount = 0;
173            return;
174        }
175        let buf = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
176            label: Some("annotation-vertices"),
177            contents: bytemuck::cast_slice(verts),
178            usage: wgpu::BufferUsages::VERTEX,
179        });
180        self.vbuf = Some(buf);
181        self.vcount = verts.len() as u32;
182    }
183
184    pub fn render(&self, rpass: &mut wgpu::RenderPass<'_>) {
185        let Some(vbuf) = &self.vbuf else { return };
186        if self.vcount == 0 {
187            return;
188        }
189        rpass.set_pipeline(&self.pipeline);
190        rpass.set_bind_group(0, &self.bind_group, &[]);
191        rpass.set_vertex_buffer(0, vbuf.slice(..));
192        rpass.draw(0..self.vcount, 0..1);
193    }
194}
