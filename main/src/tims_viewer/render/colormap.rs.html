1//! Colormap lookup tables baked into a single 2D texture (256 wide × N maps tall).
2//!
3//! The point/volume shaders sample row `colormap_id` at column `t∈[0,1]` to map a
4//! normalized intensity to a perceptual color.
5
6/// Names in row order. Keep in sync with [`anchors`].
7pub const COLORMAP_NAMES: [&str; 4] = ["Viridis", "Inferno", "Turbo", "Grayscale"];
8
9/// Number of texels per colormap row.
10const LUT_WIDTH: usize = 256;
11
12/// Control-point anchors (sRGB 0..255) for each colormap; linearly interpolated.
13fn anchors(map: usize) -> &'static [[u8; 3]] {
14    match map {
15        0 => &[
16            [68, 1, 84],
17            [59, 82, 139],
18            [33, 144, 140],
19            [93, 201, 99],
20            [253, 231, 37],
21        ],
22        1 => &[
23            [0, 0, 4],
24            [87, 16, 110],
25            [188, 55, 84],
26            [249, 142, 9],
27            [252, 255, 164],
28        ],
29        2 => &[
30            [48, 18, 59],
31            [54, 125, 197],
32            [7, 193, 151],
33            [166, 228, 48],
34            [249, 189, 38],
35            [180, 4, 38],
36        ],
37        _ => &[[0, 0, 0], [255, 255, 255]],
38    }
39}
40
41/// Build the RGBA8 LUT pixel data, row-major (`width=256`, `height=N`).
42pub fn build_lut_rgba8() -> (Vec<u8>, u32, u32) {
43    let n = COLORMAP_NAMES.len();
44    let mut data = vec![0u8; LUT_WIDTH * n * 4];
45    for map in 0..n {
46        let a = anchors(map);
47        let segs = a.len() - 1;
48        for x in 0..LUT_WIDTH {
49            let t = x as f32 / (LUT_WIDTH - 1) as f32;
50            let fseg = t * segs as f32;
51            let seg = (fseg.floor() as usize).min(segs - 1);
52            let local = fseg - seg as f32;
53            let c0 = a[seg];
54            let c1 = a[seg + 1];
55            let lerp = |i: usize| {
56                (c0[i] as f32 + (c1[i] as f32 - c0[i] as f32) * local).round() as u8
57            };
58            let idx = (map * LUT_WIDTH + x) * 4;
59            data[idx] = lerp(0);
60            data[idx + 1] = lerp(1);
61            data[idx + 2] = lerp(2);
62            data[idx + 3] = 255;
63        }
64    }
65    (data, LUT_WIDTH as u32, n as u32)
66}

68/// Upload the LUT to a sampled 2D texture and return (texture, view, sampler).
69pub fn create_lut_texture(
70    device: &wgpu::Device,
71    queue: &wgpu::Queue,
72) -> (wgpu::Texture, wgpu::TextureView, wgpu::Sampler) {
73    let (data, width, height) = build_lut_rgba8();
74    let texture = device.create_texture(&wgpu::TextureDescriptor {
75        label: Some("colormap-lut"),
76        size: wgpu::Extent3d {
77            width,
78            height,
79            depth_or_array_layers: 1,
80        },
81        mip_level_count: 1,
82        sample_count: 1,
83        dimension: wgpu::TextureDimension::D2,
84        // sRGB so sampling returns linear color for correct blending.
85        format: wgpu::TextureFormat::Rgba8UnormSrgb,
86        usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
87        view_formats: &[],
88    });
89    queue.write_texture(
90        wgpu::ImageCopyTexture {
91            texture: &texture,
92            mip_level: 0,
93            origin: wgpu::Origin3d::ZERO,
94            aspect: wgpu::TextureAspect::All,
95        },
96        &data,
97        wgpu::ImageDataLayout {
98            offset: 0,
99            bytes_per_row: Some(width * 4),
100            rows_per_image: Some(height),
101        },
102        wgpu::Extent3d {
103            width,
104            height,
105            depth_or_array_layers: 1,
106        },
107    );
108    let view = texture.create_view(&wgpu::TextureViewDescriptor::default());
109    let sampler = device.create_sampler(&wgpu::SamplerDescriptor {
110        label: Some("colormap-sampler"),
111        address_mode_u: wgpu::AddressMode::ClampToEdge,
112        address_mode_v: wgpu::AddressMode::ClampToEdge,
113        address_mode_w: wgpu::AddressMode::ClampToEdge,
114        mag_filter: wgpu::FilterMode::Linear,
115        min_filter: wgpu::FilterMode::Linear,
116        mipmap_filter: wgpu::FilterMode::Nearest,
117        ..Default::default()
118    });
119    (texture, view, sampler)
120}
