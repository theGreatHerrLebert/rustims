1//! Colormap lookup tables baked into a single 2D texture (256 wide × N maps tall).
2//!
3//! The point/volume shaders sample row `colormap_id` at column `t∈[0,1]` to map a
4//! normalized intensity to a perceptual color.
5
6/// Names in row order. Keep in sync with [`anchors`].
7pub const COLORMAP_NAMES: [&str; 4] = ["Viridis", "Inferno", "Turbo", "Grayscale"];
8
9/// Number of texels per colormap row.
10const LUT_WIDTH: usize = 256;
11
12/// Control-point anchors (sRGB 0..255) for each colormap; linearly interpolated.
13fn anchors(map: usize) -> &'static [[u8; 3]] {
14    match map {
15        0 => &[
16            [68, 1, 84],
17            [59, 82, 139],
18            [33, 144, 140],
19            [93, 201, 99],
20            [253, 231, 37],
21        ],
22        1 => &[
23            [0, 0, 4],
24            [87, 16, 110],
25            [188, 55, 84],
26            [249, 142, 9],
27            [252, 255, 164],
28        ],
29        2 => &[
30            [48, 18, 59],
31            [54, 125, 197],
32            [7, 193, 151],
33            [166, 228, 48],
34            [249, 189, 38],
35            [180, 4, 38],
36        ],
37        _ => &[[0, 0, 0], [255, 255, 255]],
38    }
39}

41/// RGB at normalized position `t∈[0,1]` for colormap `map` (CPU mirror of the LUT).
42/// The egui colorbar and the GPU LUT both go through this anchor-lerp so they can't drift.
43pub fn sample(map: usize, t: f32) -> [u8; 3] {
44    let a = anchors(map);
45    let segs = a.len() - 1;
46    let t = t.clamp(0.0, 1.0);
47    let fseg = t * segs as f32;
48    let seg = (fseg.floor() as usize).min(segs - 1);
49    let local = fseg - seg as f32;
50    let c0 = a[seg];
51    let c1 = a[seg + 1];
52    let lerp = |i: usize| (c0[i] as f32 + (c1[i] as f32 - c0[i] as f32) * local).round() as u8;
53    [lerp(0), lerp(1), lerp(2)]
54}

56/// Build the RGBA8 LUT pixel data, row-major (`width=256`, `height=N`).
57pub fn build_lut_rgba8() -> (Vec<u8>, u32, u32) {
58    let n = COLORMAP_NAMES.len();
59    let mut data = vec![0u8; LUT_WIDTH * n * 4];
60    for map in 0..n {
61        for x in 0..LUT_WIDTH {
62            let t = x as f32 / (LUT_WIDTH - 1) as f32;
63            let rgb = sample(map, t);
64            let idx = (map * LUT_WIDTH + x) * 4;
65            data[idx] = rgb[0];
66            data[idx + 1] = rgb[1];
67            data[idx + 2] = rgb[2];
68            data[idx + 3] = 255;
69        }
70    }
71    (data, LUT_WIDTH as u32, n as u32)
72}

74/// Upload the LUT to a sampled 2D texture and return (texture, view, sampler).
75pub fn create_lut_texture(
76    device: &wgpu::Device,
77    queue: &wgpu::Queue,
78) -> (wgpu::Texture, wgpu::TextureView, wgpu::Sampler) {
79    let (data, width, height) = build_lut_rgba8();
80    let texture = device.create_texture(&wgpu::TextureDescriptor {
81        label: Some("colormap-lut"),
82        size: wgpu::Extent3d {
83            width,
84            height,
85            depth_or_array_layers: 1,
86        },
87        mip_level_count: 1,
88        sample_count: 1,
89        dimension: wgpu::TextureDimension::D2,
90        // sRGB so sampling returns linear color for correct blending.
91        format: wgpu::TextureFormat::Rgba8UnormSrgb,
92        usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
93        view_formats: &[],
94    });
95    queue.write_texture(
96        wgpu::ImageCopyTexture {
97            texture: &texture,
98            mip_level: 0,
99            origin: wgpu::Origin3d::ZERO,
100            aspect: wgpu::TextureAspect::All,
101        },
102        &data,
103        wgpu::ImageDataLayout {
104            offset: 0,
105            bytes_per_row: Some(width * 4),
106            rows_per_image: Some(height),
107        },
108        wgpu::Extent3d {
109            width,
110            height,
111            depth_or_array_layers: 1,
112        },
113    );
114    let view = texture.create_view(&wgpu::TextureViewDescriptor::default());
115    let sampler = device.create_sampler(&wgpu::SamplerDescriptor {
116        label: Some("colormap-sampler"),
117        address_mode_u: wgpu::AddressMode::ClampToEdge,
118        address_mode_v: wgpu::AddressMode::ClampToEdge,
119        address_mode_w: wgpu::AddressMode::ClampToEdge,
120        mag_filter: wgpu::FilterMode::Linear,
121        min_filter: wgpu::FilterMode::Linear,
122        mipmap_filter: wgpu::FilterMode::Nearest,
123        ..Default::default()
124    });
125    (texture, view, sampler)
126}

128/// Distinct color per window group. Shared by the annotation overlay (DIA isolation windows) and
129/// the egui group legend. Render-safe (no native deps), so it lives here rather than in the native
130/// loader.
131///
132/// Hue advances by the GOLDEN ANGLE per group rather than a plain `group/n` ramp: a linear ramp
133/// gives adjacent groups nearly-identical hues (indistinguishable when a run has many windows),
134/// whereas the golden angle lands consecutive groups ~137° apart on the wheel while still covering
135/// it evenly. A parity brightness step adds a second cue so neighbours differ in value too. The
136/// color depends only on `group`, so the overlay and legend stay in lockstep regardless of `n_groups`.
137pub fn group_color(group: u32, _n_groups: u32) -> [f32; 3] {
138    let g = group.saturating_sub(1) as f32;
139    let h = (g * 0.618_034).fract(); // golden-ratio conjugate ≈ 137.5° steps
140    let v = if group % 2 == 0 { 0.82 } else { 1.0 };
141    hsv_to_rgb(h, 0.7, v)
142}
143
144/// HSV (all in [0,1] except hue which wraps) to linear-ish RGB for line colors.
145fn hsv_to_rgb(h: f32, s: f32, v: f32) -> [f32; 3] {
146    let i = (h * 6.0).floor();
147    let f = h * 6.0 - i;
148    let p = v * (1.0 - s);
149    let q = v * (1.0 - f * s);
150    let t = v * (1.0 - (1.0 - f) * s);
151    match (i as i32).rem_euclid(6) {
152        0 => [v, t, p],
153        1 => [q, v, p],
154        2 => [p, v, t],
155        3 => [p, q, v],
156        4 => [t, p, v],
157        _ => [v, p, q],
158    }
159}

161#[cfg(test)]
162mod tests {
163    use super::*;
164
165    #[test]
166    fn sample_endpoints_match_anchors() {
167        // Viridis anchors run [68,1,84] -> ... -> [253,231,37].
168        assert_eq!(sample(0, 0.0), [68, 1, 84]);
169        assert_eq!(sample(0, 1.0), [253, 231, 37]);
170        // Out-of-range t clamps to the endpoints.
171        assert_eq!(sample(0, -1.0), [68, 1, 84]);
172        assert_eq!(sample(0, 2.0), [253, 231, 37]);
173    }
