1//! Volume rendering: a CPU max-intensity voxel grid (built incrementally as points
2//! stream in) uploaded to an anisotropic R16Float 3D texture, plus a fullscreen-triangle
3//! raycaster that shares the point cloud's transfer-function + colormap.
4//!
5//! MVP semantics: each voxel stores the MAX intensity of points falling in it (nearest
6//! voxel). Max keeps voxel values in the same range as raw per-point intensity, so the
7//! identical transfer function (i_min/i_max/log) and colormap apply unchanged — one
8//! shared exposure model across points and volume. Trilinear cloud-in-cell *density*
9//! deposition is the documented Phase-3 refinement.
10

12
13use super::colormap;
14use super::uniforms::VolumeUniform;

16/// Anisotropic grid dimensions (m/z, 1/K0, RT). m/z gets the most bins (finest features).
17pub const VOLUME_DIMS: [u32; 3] = [256, 128, 192];
18
19/// Target peak stored f16 value; the grid is normalized so its densest voxel maps
20/// here, well under the f16 ceiling (65504) so nothing is clamped/lost.
21const F16_TARGET_MAX: f32 = 60000.0;
22
23/// CPU-side density voxel grid: trilinear cloud-in-cell accumulation of intensity.
24pub struct VolumeGrid {
25    dims: [u32; 3],
26    /// Accumulated density per voxel, indexed x + nx*(y + ny*z).
27    data: Vec<f32>,
28    /// Largest accumulated voxel density (drives the normalization scale).
29    max_density: f32,
30    dirty: bool,
31    /// Reusable f16 staging buffer for uploads, to avoid a per-frame allocation.
32    scratch: Vec<f16>,
33}
34
35impl VolumeGrid {
36    pub fn new(dims: [u32; 3]) -> Self {
37        assert!(
38            dims.iter().all(|&d| d > 0),
39            "volume dims must be non-zero, got {dims:?}"
40        );
41        let n = (dims[0] as usize)
42            .checked_mul(dims[1] as usize)
43            .and_then(|m| m.checked_mul(dims[2] as usize))
44            .expect("volume dimensions overflow usize");
45        VolumeGrid {
46            dims,
47            data: vec![0.0; n],
48            max_density: 0.0,
49            dirty: false,
50            scratch: Vec::new(),
51        }
52    }
53
54    pub fn dirty(&self) -> bool {
55        self.dirty
56    }
57
58    pub fn clear_dirty(&mut self) {
59        self.dirty = false;
60    }
61
62    /// Zero all density (and the running max) — used when re-streaming a refined region so
63    /// the grid rebuilds from scratch.
64    pub fn clear(&mut self) {
65        self.data.iter_mut().for_each(|v| *v = 0.0);
66        self.max_density = 0.0;
67        self.dirty = true;
68    }
69
70    /// Deposit a point (normalized cube position in [-1,1], raw intensity) using
71    /// trilinear cloud-in-cell weighting: the intensity is split across the 8 nearest
72    /// voxels by their interpolation weights and ADDED. This is conservative (the 8
73    /// weights sum to 1, so total deposited == intensity) and avoids the blocky
74    /// aliasing of nearest-voxel binning. Edges clamp onto the boundary voxels.
75    #[inline]
76    pub fn deposit(&mut self, pos: [f32; 3], intensity: f32) {
77        let [nx, ny, nz] = self.dims;
78        // Cell-centered continuous coordinate in [-0.5, dim-0.5].
79        let g = |norm: f32, dim: u32| (norm * 0.5 + 0.5).clamp(0.0, 1.0) * dim as f32 - 0.5;
80        let (gx, gy, gz) = (g(pos[0], nx), g(pos[1], ny), g(pos[2], nz));
81        let split = |gc: f32, dim: u32| -> (usize, usize, f32) {
82            let base = gc.floor();
83            let frac = gc - base;
84            let lo = (base as i64).clamp(0, dim as i64 - 1) as usize;
85            let hi = ((base as i64) + 1).clamp(0, dim as i64 - 1) as usize;
86            (lo, hi, frac)
87        };
88        let (x0, x1, fx) = split(gx, nx);
89        let (y0, y1, fy) = split(gy, ny);
90        let (z0, z1, fz) = split(gz, nz);
91        let nxu = nx as usize;
92        let nyu = ny as usize;
93        let mut add = |x: usize, y: usize, z: usize, w: f32| {
94            let idx = x + nxu * (y + nyu * z);
95            let v = self.data[idx] + intensity * w;
96            self.data[idx] = v;
97            if v > self.max_density {
98                self.max_density = v;
99            }
100        };
101        add(x0, y0, z0, (1.0 - fx) * (1.0 - fy) * (1.0 - fz));
102        add(x1, y0, z0, fx * (1.0 - fy) * (1.0 - fz));
103        add(x0, y1, z0, (1.0 - fx) * fy * (1.0 - fz));
104        add(x1, y1, z0, fx * fy * (1.0 - fz));
105        add(x0, y0, z1, (1.0 - fx) * (1.0 - fy) * fz);
106        add(x1, y0, z1, fx * (1.0 - fy) * fz);
107        add(x0, y1, z1, (1.0 - fx) * fy * fz);
108        add(x1, y1, z1, fx * fy * fz);
109        self.dirty = true;
110    }
111
112    /// Factor the shader multiplies the sampled (normalized) value by to recover raw
113    /// density, chosen so the densest voxel maps to `F16_TARGET_MAX`.
114    pub fn density_scale(&self) -> f32 {
115        if self.max_density > 0.0 {
116            self.max_density / F16_TARGET_MAX
117        } else {
118            1.0
119        }
120    }
121
122    /// p1/p99 of the non-empty voxel densities — sensible default transfer range for
123    /// the volume (density sums live in a different range than per-point intensity).
124    pub fn density_percentiles(&self) -> (f32, f32) {
125        let mut nz: Vec<f32> = self.data.iter().copied().filter(|&v| v > 0.0).collect();
126        if nz.is_empty() {
127            return (1.0, 2.0);
128        }
129        // Two order statistics (median, p99.9) via partial select (O(n)) instead of a full sort of
130        // up to ~6.3M voxels on every re-voxel. select_nth places the k-th smallest at index k, so
131        // nz[k] equals the sorted value — same result as before. Indices are < len (empty handled).
132        let cmp = |a: &f32, b: &f32| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal);
133        let lo_i = (((nz.len() - 1) as f32) * 0.50) as usize;
134        let hi_i = (((nz.len() - 1) as f32) * 0.999) as usize;
135        nz.select_nth_unstable_by(hi_i, cmp);
136        let hi_raw = nz[hi_i];
137        nz.select_nth_unstable_by(lo_i, cmp); // reorders, but hi_raw is already read
138        let lo = nz[lo_i].max(1e-6); // median: voxels below it are sparse fringe
139        let hi = hi_raw.max(lo * 1.0001);
140        (lo, hi)
141    }
142
143    /// Rebuild this grid as `wa*a + wb*b` of two same-dim source grids. Used to fold the
144    /// separate MS1 / MS2 density grids into the displayed volume per the MS1/MS2 toggle
145    /// (weights are 1.0 or 0.0), so the volume raycaster honors the same filter as points.
146    pub fn combine(&mut self, a: &VolumeGrid, b: &VolumeGrid, wa: f32, wb: f32) {
147        assert!(self.data.len() == a.data.len() && self.data.len() == b.data.len());
148        let mut max = 0.0f32;
149        for (i, d) in self.data.iter_mut().enumerate() {
150            let v = a.data[i] * wa + b.data[i] * wb;
151            *d = v;
152            if v > max {
153                max = v;
154            }
155        }
156        self.max_density = max;
157        self.dirty = true;
158    }
159
160    /// Normalize by `density_scale` and convert to f16 into the reusable scratch buffer
161    /// (avoids a fresh multi-MB allocation per upload). The shader multiplies back by
162    /// `density_scale` before the transfer function.
163    pub fn to_f16_scaled(&mut self) -> &[f16] {
164        let inv = 1.0 / self.density_scale();
165        if self.scratch.len() != self.data.len() {
166            self.scratch = vec![f16::from_f32(0.0); self.data.len()];
167        }
168        for (s, &v) in self.scratch.iter_mut().zip(self.data.iter()) {
169            *s = f16::from_f32((v * inv).min(F16_TARGET_MAX));
170        }
171        &self.scratch
172    }
173}

184impl VolumeRenderer {
185    pub fn new(
186        device: &wgpu::Device,
187        queue: &wgpu::Queue,
188        color_format: wgpu::TextureFormat,
189        depth_format: wgpu::TextureFormat,
190        dims: [u32; 3],
191    ) -> Self {
192        let texture = device.create_texture(&wgpu::TextureDescriptor {
193            label: Some("volume-texture"),
194            size: wgpu::Extent3d {
195                width: dims[0],
196                height: dims[1],
197                depth_or_array_layers: dims[2],
198            },
199            mip_level_count: 1,
200            sample_count: 1,
201            dimension: wgpu::TextureDimension::D3,
202            format: wgpu::TextureFormat::R16Float,
203            usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
204            view_formats: &[],
205        });
206        let view = texture.create_view(&wgpu::TextureViewDescriptor::default());
207        let sampler = device.create_sampler(&wgpu::SamplerDescriptor {
208            label: Some("volume-sampler"),
209            address_mode_u: wgpu::AddressMode::ClampToEdge,
210            address_mode_v: wgpu::AddressMode::ClampToEdge,
211            address_mode_w: wgpu::AddressMode::ClampToEdge,
212            mag_filter: wgpu::FilterMode::Linear,
213            min_filter: wgpu::FilterMode::Linear,
214            mipmap_filter: wgpu::FilterMode::Nearest,
215            ..Default::default()
216        });
217
218        let (lut_texture, lut_view, lut_sampler) = colormap::create_lut_texture(device, queue);
219
220        let uniform_buf = device.create_buffer(&wgpu::BufferDescriptor {
221            label: Some("volume-uniform"),
222            size: std::mem::size_of::<VolumeUniform>() as u64,
223            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
224            mapped_at_creation: false,
225        });
226        queue.write_buffer(&uniform_buf, 0, bytemuck::bytes_of(&VolumeUniform::default()));
227
228        let bind_layout = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
229            label: Some("volume-bind-layout"),
230            entries: &[
231                wgpu::BindGroupLayoutEntry {
232                    binding: 0,
233                    visibility: wgpu::ShaderStages::FRAGMENT,
234                    ty: wgpu::BindingType::Buffer {
235                        ty: wgpu::BufferBindingType::Uniform,
236                        has_dynamic_offset: false,
237                        min_binding_size: None,
238                    },
239                    count: None,
240                },
241                wgpu::BindGroupLayoutEntry {
242                    binding: 1,
243                    visibility: wgpu::ShaderStages::FRAGMENT,
244                    ty: wgpu::BindingType::Texture {
245                        sample_type: wgpu::TextureSampleType::Float { filterable: true },
246                        view_dimension: wgpu::TextureViewDimension::D3,
247                        multisampled: false,
248                    },
249                    count: None,
250                },
251                wgpu::BindGroupLayoutEntry {
252                    binding: 2,
253                    visibility: wgpu::ShaderStages::FRAGMENT,
254                    ty: wgpu::BindingType::Sampler(wgpu::SamplerBindingType::Filtering),
255                    count: None,
256                },
257                wgpu::BindGroupLayoutEntry {
258                    binding: 3,
259                    visibility: wgpu::ShaderStages::FRAGMENT,
260                    ty: wgpu::BindingType::Texture {
261                        sample_type: wgpu::TextureSampleType::Float { filterable: true },
262                        view_dimension: wgpu::TextureViewDimension::D2,
263                        multisampled: false,
264                    },
265                    count: None,
266                },
267                wgpu::BindGroupLayoutEntry {
268                    binding: 4,
269                    visibility: wgpu::ShaderStages::FRAGMENT,
270                    ty: wgpu::BindingType::Sampler(wgpu::SamplerBindingType::Filtering),
271                    count: None,
272                },
273            ],
274        });
275
276        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
277            label: Some("volume-bind-group"),
278            layout: &bind_layout,
279            entries: &[
280                wgpu::BindGroupEntry {
281                    binding: 0,
282                    resource: uniform_buf.as_entire_binding(),
283                },
284                wgpu::BindGroupEntry {
285                    binding: 1,
286                    resource: wgpu::BindingResource::TextureView(&view),
287                },
288                wgpu::BindGroupEntry {
289                    binding: 2,
290                    resource: wgpu::BindingResource::Sampler(&sampler),
291                },
292                wgpu::BindGroupEntry {
293                    binding: 3,
294                    resource: wgpu::BindingResource::TextureView(&lut_view),
295                },
296                wgpu::BindGroupEntry {
297                    binding: 4,
298                    resource: wgpu::BindingResource::Sampler(&lut_sampler),

300            ],
301        });
302
303        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
304            label: Some("volume-shader"),
305            source: wgpu::ShaderSource::Wgsl(include_str!("../shaders/volume.wgsl").into()),
306        });
307        let pipeline_layout = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
308            label: Some("volume-pipeline-layout"),
309            bind_group_layouts: &[&bind_layout],
310            push_constant_ranges: &[],
311        });
312        let pipeline = device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
313            label: Some("volume-pipeline"),
314            layout: Some(&pipeline_layout),
315            vertex: wgpu::VertexState {
316                module: &shader,
317                entry_point: "vs_main",
318                buffers: &[],
319                compilation_options: Default::default(),

321            fragment: Some(wgpu::FragmentState {
322                module: &shader,
323                entry_point: "fs_main",
324                targets: &[Some(wgpu::ColorTargetState {

329                    blend: Some(wgpu::BlendState {

331                            src_factor: wgpu::BlendFactor::One,

333                            operation: wgpu::BlendOperation::Add,
334                        },
335                        alpha: wgpu::BlendComponent {
336                            src_factor: wgpu::BlendFactor::One,
337                            dst_factor: wgpu::BlendFactor::OneMinusSrcAlpha,
338                            operation: wgpu::BlendOperation::Add,
339                        },
340                    }),
341                    write_mask: wgpu::ColorWrites::ALL,
342                })],
343                compilation_options: Default::default(),
344            }),
345            primitive: wgpu::PrimitiveState {
346                topology: wgpu::PrimitiveTopology::TriangleList,
347                ..Default::default()
348            },
349            // Scene pass owns a depth attachment; declare it but neither test nor write.
350            depth_stencil: Some(wgpu::DepthStencilState {
351                format: depth_format,
352                depth_write_enabled: false,
353                depth_compare: wgpu::CompareFunction::Always,
354                stencil: wgpu::StencilState::default(),
355                bias: wgpu::DepthBiasState::default(),
356            }),
357            multisample: wgpu::MultisampleState::default(),
358            multiview: None,
359            cache: None,
360        });
361
362        VolumeRenderer {
363            texture,
364            dims,
365            uniform_buf,
366            bind_group,
367            pipeline,
368            _lut_texture: lut_texture,
369        }
370    }

372    pub fn update_uniform(&self, queue: &wgpu::Queue, u: &VolumeUniform) {
373        queue.write_buffer(&self.uniform_buf, 0, bytemuck::bytes_of(u));
374    }
375
376    /// Upload the (f16) grid into the 3D texture. `data` length must equal the grid.
377    pub fn upload(&self, queue: &wgpu::Queue, data: &[f16]) {
378        let [nx, ny, nz] = self.dims;
379        debug_assert_eq!(data.len(), nx as usize * ny as usize * nz as usize);
380        queue.write_texture(
381            wgpu::ImageCopyTexture {
382                texture: &self.texture,
383                mip_level: 0,
384                origin: wgpu::Origin3d::ZERO,
385                aspect: wgpu::TextureAspect::All,
386            },
387            bytemuck::cast_slice(data),
388            wgpu::ImageDataLayout {
389                offset: 0,
390                bytes_per_row: Some(nx * 2), // R16Float = 2 bytes/texel
391                rows_per_image: Some(ny),
392            },
393            wgpu::Extent3d {
394                width: nx,
395                height: ny,
396                depth_or_array_layers: nz,
397            },
398        );
399    }
400
401    pub fn render(&self, rpass: &mut wgpu::RenderPass<'_>) {
402        rpass.set_pipeline(&self.pipeline);
403        rpass.set_bind_group(0, &self.bind_group, &[]);
404        rpass.draw(0..3, 0..1);
405    }
406}

413    #[test]
414    fn trilinear_deposit_conserves_total() {
415        // Trilinear deposit splits intensity across the 8 nearest voxels with weights
416        // summing to 1, so the total grid mass equals the deposited intensity.
417        let mut g = VolumeGrid::new([8, 8, 8]);
418        g.deposit([0.13, -0.27, 0.61], 1000.0);
419        assert!(g.dirty());
420        let total: f32 = g.data.iter().sum();
421        assert!((total - 1000.0).abs() < 0.5, "total mass {total} != 1000");
422        // A second point at the same place doubles the local mass (additive density).
423        g.deposit([0.13, -0.27, 0.61], 1000.0);
424        let total2: f32 = g.data.iter().sum();
425        assert!((total2 - 2000.0).abs() < 1.0, "total mass {total2} != 2000");
426    }

428    #[test]
429    fn density_scale_and_f16_recover_peak() {
430        let mut g = VolumeGrid::new([4, 4, 4]);
431        // Deposit on an exact voxel center so its full intensity lands in one voxel.
432        g.deposit([0.0, 0.0, 0.0], 1e9);
433        assert!(g.density_scale() > 1.0); // huge peak -> scaled down for f16
434        let f = g.to_f16_scaled();
435        assert_eq!(f.len(), 64);
436        let peak = f.iter().map(|h| h.to_f32()).fold(0.0f32, f32::max);
437        assert!(peak > 0.0 && peak <= 65504.0);
438        // Recovering raw density (stored * density_scale) reconstructs the peak.
439        let recovered = peak * g.density_scale();
440        assert!((recovered - g.max_density).abs() / g.max_density < 0.01);
441    }
