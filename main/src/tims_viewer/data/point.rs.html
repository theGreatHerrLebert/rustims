5/// Number of evenly spaced RT slices each DIA window footprint is drawn at. Shared here (a
6/// both-targets module) so the native loader and the wasm overlay use one value — the geometry
7/// must match on both sides of the `/windows` endpoint.
8pub const DIA_WINDOW_RT_SLICES: usize = 6;

10/// One DIA/MIDIA isolation window's selection footprint in **real units**: an `(m/z × 1/K0)`
11/// rectangle (mobility already converted from scan numbers). Group ids color the interleaved
12/// isolation bands. The native loader produces these for the `/windows` endpoint; the wasm client
13/// re-normalizes them to its region bounds (so they align with the cloud and off-region windows
14/// clip). Shared here so both sides use one struct definition.
15#[derive(Clone, Copy, Debug)]
16pub struct WindowRect {
17    pub group: u32,
18    pub mz0: f64,
19    pub mz1: f64,
20    pub im0: f64,
21    pub im1: f64,
22}

24/// One renderable data point, packed for the GPU as instance data.
25///
26/// Layout is 32 bytes, 16-byte aligned, so a `Vec<GpuPoint>` uploads directly as a
27/// vertex (instance-step) buffer via `bytemuck::cast_slice`.
28///
29/// `weight` is the number of original points this sample represents (`1/p` of the
30/// downsample). Carrying it lets additive-density rendering stay brightness-invariant
31/// regardless of the downsample ratio (see the shared density/exposure model).
32#[repr(C)]
33#[derive(Clone, Copy, Debug, Pod, Zeroable)]
34pub struct GpuPoint {
35    /// Position in the normalized cube `[-1, 1]^3` (x=m/z, y=1/K0, z=RT).
36    pub pos: [f32; 3],
37    /// Raw intensity (counts).
38    pub intensity: f32,
39    /// Sample weight = `1/p` (how many source points this one stands in for).
40    pub weight: f32,
41    /// Bit flags. bit0: 0 = MS1/precursor, 1 = MS2/fragment.
42    pub flags: u32,
43    /// `_pad[0]` doubles as the DBSCAN cluster id for cluster coloring (`NO_CLUSTER` =
44    /// noise/unclustered); `_pad[1]` is reserved padding to keep the 32-byte layout.
45    pub _pad: [u32; 2],
46}
47
48impl GpuPoint {
49    pub const MS2_FLAG: u32 = 1;
50    /// Cluster-id sentinel for noise / not-yet-clustered points (rendered grey).
51    pub const NO_CLUSTER: u32 = u32::MAX;
52
53    /// `wgpu` vertex-buffer layout describing the instance step-mode attributes.
54    pub fn layout() -> wgpu::VertexBufferLayout<'static> {
55        const ATTRS: [wgpu::VertexAttribute; 5] = wgpu::vertex_attr_array![
56            0 => Float32x3, // pos
57            1 => Float32,   // intensity
58            2 => Float32,   // weight
59            3 => Uint32,    // flags
60            4 => Uint32,    // cluster id (_pad[0])
61        ];
62        wgpu::VertexBufferLayout {
63            array_stride: std::mem::size_of::<GpuPoint>() as wgpu::BufferAddress,
64            step_mode: wgpu::VertexStepMode::Instance,
65            attributes: &ATTRS,
66        }
67    }
68}

70/// Affine mapping between a real-unit axis range and the normalized cube `[-1, 1]`.
71#[derive(Clone, Copy, Debug)]
72pub struct AxisTransform {
73    pub min: f64,
74    pub max: f64,
75}
76
77impl AxisTransform {
78    pub fn new(min: f64, max: f64) -> Self {
79        // Guard against a degenerate (zero-width) range.
80        if (max - min).abs() < f64::EPSILON {
81            AxisTransform {
82                min,
83                max: min + 1.0,
84            }
85        } else {
86            AxisTransform { min, max }
87        }
88    }
89
90    /// Map a real value into `[-1, 1]`.
91    #[inline]
92    pub fn normalize(&self, v: f64) -> f32 {
93        (2.0 * (v - self.min) / (self.max - self.min) - 1.0) as f32
94    }
95
96    /// Map a normalized `[-1, 1]` value back to real units.
97    /// Used by axis-tick labeling (and tested); kept ahead of the gizmo work.
98    #[allow(dead_code)]
99    #[inline]
100    pub fn denormalize(&self, n: f32) -> f64 {
101        self.min + ((n as f64) + 1.0) * 0.5 * (self.max - self.min)
102    }
103}

105/// The three spatial axes of the run, in real units, plus their normalizers.
106#[derive(Clone, Copy, Debug)]
107pub struct AxisBounds {
108    /// x-axis: m/z (Th).
109    pub mz: AxisTransform,
110    /// y-axis: inverse ion mobility 1/K0.
111    pub im: AxisTransform,
112    /// z-axis: retention time (seconds).
113    pub rt: AxisTransform,
114}
115
116impl AxisBounds {
117    /// Normalize a `(mz, im, rt)` real-unit triple into cube coordinates.
118    #[inline]
119    pub fn normalize(&self, mz: f64, im: f64, rt: f64) -> [f32; 3] {
120        [self.mz.normalize(mz), self.im.normalize(im), self.rt.normalize(rt)]
121    }
122}

124#[cfg(test)]
125mod tests {
126    use super::*;
127
128    #[test]
129    fn gpu_point_is_32_bytes() {
130        assert_eq!(std::mem::size_of::<GpuPoint>(), 32);
131    }
132
133    #[test]
134    fn axis_transform_round_trip() {
135        let t = AxisTransform::new(100.0, 1700.0);
136        for v in [100.0, 400.0, 900.5, 1700.0] {
137            let n = t.normalize(v);
138            let back = t.denormalize(n);
139            assert!((back - v).abs() < 1e-6, "round-trip failed for {v}: {back}");
140        }
141        assert!((t.normalize(100.0) + 1.0).abs() < 1e-6);
142        assert!((t.normalize(1700.0) - 1.0).abs() < 1e-6);
143    }
144
145    #[test]
146    fn axis_transform_handles_degenerate_range() {
147        let t = AxisTransform::new(5.0, 5.0);
148        // Must not divide by zero / produce NaN.
149        assert!(t.normalize(5.0).is_finite());
150    }
