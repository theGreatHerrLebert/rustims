1//! Background streaming loader.
2//!
3//! Runs on its own thread, streams frames (real or synthetic), downsamples to the
4//! point budget, packs `GpuPoint`s, and sends them to the render thread over a bounded
5//! channel so the event loop never blocks on I/O and the cloud builds progressively.
6//!
7//! Sampling is **stratified**: a global systematic stride provides a count-bounded,
8//! RT-unbiased density base (each kept point weighted `1/p` for brightness-invariant
9//! additive rendering), and a coarse spatial grid additionally keeps the brightest
10//! point in every occupied cell so sparse high-intensity features always survive.
11

29/// Messages from the loader thread to the render thread.
30///
31/// `generation` is carried now but only consumed once region-refinement lands
32/// (Phase 1.5); the single Phase-1 load uses generation 0.
33#[allow(dead_code)]
34pub enum LoadMsg {
35    /// A batch of systematic (density-base) points belonging to `generation`.
36    Chunk { generation: u64, points: Vec<GpuPoint> },
37    /// A batch of peak points (per-cell brightest) — display-only enrichment for the
38    /// point cloud; consumers must NOT add these to the volume density (they would
39    /// double-count) and they are not part of the systematic density base.
40    PeakChunk { points: Vec<GpuPoint> },
41    /// Fractional load progress in `[0, 1]`.
42    Progress(f32),
43    /// Robust intensity percentiles (p1/p50/p99) for transfer-function defaults.
44    Stats { i_p1: f32, i_p99: f32, i_p50: f32 },
45    /// Per-axis distribution histograms (each `HIST_BINS` long) for the levels-style filter
46    /// UI. m/z·1/K0·RT bin linearly over the cube bounds; `intensity` is data-tight log,
47    /// spanning `[i_lo, i_hi]` (the kept sample's min/max intensity).
48    Histograms {
49        mz: Vec<u32>,
50        im: Vec<u32>,
51        rt: Vec<u32>,
52        intensity: Vec<u32>,
53        i_lo: f32,
54        i_hi: f32,
55        /// 2D density projections (each `PROJ_BINS * PROJ_BINS`, row-major `x + PROJ_BINS*y`)
56        /// for the minimaps: m/z×1/K0, m/z×RT, 1/K0×RT.
57        proj_mz_im: Vec<u32>,
58        proj_mz_rt: Vec<u32>,
59        proj_im_rt: Vec<u32>,
60    },
61    /// Annotation overlay geometry: colored line-list vertices (pairs = segments) in the
62    /// normalized cube (DDA precursor crosses / DIA isolation-window footprints). `groups`
63    /// is a parallel per-vertex window-group id (u32::MAX = ungrouped); `n_groups` is the
64    /// number of DIA/MIDIA window groups, for the per-group visibility UI.
65    Annotations { lines: Vec<LineVertex>, groups: Vec<u32>, n_groups: u32 },
66    /// All frames for `generation` have been streamed.
67    Done { generation: u64 },
68    /// Fatal error; loading stopped.
69    Error(String),
70}
71
72/// Commands from the render thread to the loader thread.
73pub enum LoadCmd {
74    /// Abandon the current load and exit.
75    Cancel,
76}
77
78/// Optional real-unit m/z + 1/K0 + intensity cull applied per-point during a region refinement
79/// load (frame selection already handles the RT window). Points outside are skipped at the source,
80/// so the budget concentrates on the focused region (the 4D lens — see FOCUS_LENS_PLAN.md).
81#[derive(Clone, Copy)]
82pub struct RegionFilter {
83    pub mz: (f64, f64),
84    pub im: (f64, f64),
85    /// Per-point intensity floor (real counts); points below are dropped at the source.
86    pub intensity_min: f32,
87}
88
89/// What to stream.
90pub enum LoaderMode {
91    Real { path: String, frame_ids: Vec<u32> },
92    Demo(DemoSource),
93}

95/// Owns the loader thread and the channels to talk to it.
96pub struct LoaderHandle {
97    pub rx: Receiver<LoadMsg>,
98    cmd_tx: Sender<LoadCmd>,
99    handle: Option<JoinHandle<()>>,
100}
101
102impl LoaderHandle {
103    pub fn spawn(
104        mode: LoaderMode,
105        bounds: AxisBounds,
106        total_estimate: u64,
107        budget: usize,
108        filter: Option<RegionFilter>,
109    ) -> Self {
110        let (msg_tx, msg_rx) = bounded::<LoadMsg>(8);
111        let (cmd_tx, cmd_rx) = bounded::<LoadCmd>(4);
112        let handle = std::thread::Builder::new()
113            .name("tims-loader".into())
114            .spawn(move || {
115                run_loader(mode, bounds, total_estimate, budget, filter, &msg_tx, &cmd_rx);
116            })
117            .expect("failed to spawn loader thread");
118        LoaderHandle {
119            rx: msg_rx,
120            cmd_tx,
121            handle: Some(handle),
122        }
123    }
124}
125
126impl Drop for LoaderHandle {
127    fn drop(&mut self) {
128        let _ = self.cmd_tx.send(LoadCmd::Cancel);
129        if let Some(h) = self.handle.take() {
130            let _ = h.join();
131        }
132    }
133}

135const GENERATION: u64 = 0; // single generation in Phase 1
136const CHUNK_POINTS: usize = 300_000;
137const RESERVOIR_CAP: usize = 200_000;
138
139/// Sampling stride derived from the budget.
140///
141/// Uses **ceiling** division so the kept count never exceeds the budget: keeping every
142/// `stride`-th point yields `~total/stride <= budget` points. (Rounding instead would
143/// let e.g. total=149, budget=100 round to stride 1 and emit all 149, biasing the view
144/// toward early frames once the GPU buffer truncates the tail.)
145pub fn stride_for(total_estimate: u64, budget: usize) -> usize {
146    if total_estimate == 0 || budget == 0 {
147        return 1;
148    }
149    let budget = budget as u64;
150    if budget >= total_estimate {
151        1
152    } else {
153        total_estimate.div_ceil(budget).max(1) as usize
154    }
155}

157/// Coarse spatial grid for peak preservation (m/z, 1/K0, RT cells).
158const PEAK_DIMS: [u32; 3] = [128, 64, 128];
159/// Placeholder point for the peak-grid HashMap entry API.
160const ZERO_POINT: GpuPoint = GpuPoint {
161    pos: [0.0; 3],
162    intensity: 0.0,
163    weight: 1.0,
164    flags: 0,
165    _pad: [GpuPoint::NO_CLUSTER, 0],
166};
167
168/// Number of bins in each per-axis distribution histogram (for the levels-style filter UI).
169pub const HIST_BINS: usize = 80;
170/// Fine intermediate bins for the intensity log histogram before rebinning to the data range.
171const LOGI_FINE: usize = 256;
172/// Upper bound of the fixed intensity log range (log10), generous for timsTOF intensities.
173const LOGI_HI: f32 = 8.0;
174
175/// Bin a normalized-cube coordinate in `[-1, 1]` into `[0, bins)`. Equal-width cells: scale by the
176/// bin count and clamp, so every bin covers `1/N` of the range (vs `*(N-1)`, which leaves the last
177/// bin degenerate — only exactly `1.0` — and misaligns the strips/maps from UI fractions). The one
178/// place this cube→bin-index convention lives; shared by hbin/pbin/peak_cell.
179#[inline]
180fn bin(n: f32, bins: usize) -> usize {
181    (((n * 0.5 + 0.5).clamp(0.0, 1.0) * bins as f32) as usize).min(bins.saturating_sub(1))
182}
183
184#[inline]
185fn hbin(n: f32) -> usize {
186    bin(n, HIST_BINS)
187}
188
189/// Side length of each 2D projection minimap (m/z·1/K0·RT plane heatmaps).
190pub const PROJ_BINS: usize = 96;
191/// Bin a normalized-cube coordinate in `[-1, 1]` into `[0, PROJ_BINS)` (equal-width cells).
192#[inline]
193fn pbin(n: f32) -> usize {
194    bin(n, PROJ_BINS)
195}
196
197/// Map a normalized-cube position to a coarse peak-grid cell index.
198#[inline]
199fn peak_cell(pos: [f32; 3]) -> u32 {
200    let axis = |n: f32, dim: u32| -> u32 { bin(n, dim as usize) as u32 };
201    let x = axis(pos[0], PEAK_DIMS[0]);
202    let y = axis(pos[1], PEAK_DIMS[1]);
203    let z = axis(pos[2], PEAK_DIMS[2]);
204    x + PEAK_DIMS[0] * (y + PEAK_DIMS[1] * z)
205}

207#[allow(unused_assignments)] // final flush! reassigns `chunk` which is then dropped
208fn run_loader(
209    mode: LoaderMode,
210    bounds: AxisBounds,
211    total_estimate: u64,
212    budget: usize,
213    filter: Option<RegionFilter>,
214    tx: &Sender<LoadMsg>,
215    cmd_rx: &Receiver<LoadCmd>,
216) {
217    // Stratified sampling: spend ~85% of the budget on a systematic density base, and
218    // reserve the rest for per-cell intensity PEAKS so sparse high-intensity features
219    // always survive (pure systematic sampling can statistically miss them).
220    let systematic_budget = (budget.saturating_mul(85) / 100).max(1);
221    let stride = stride_for(total_estimate, systematic_budget);
222    let weight = stride as f32;
223    let mut systematic_count: u64 = 0;
224    // Flat peak grid indexed by cell: (best intensity, source index, best point) over
225    // ALL points. A direct-indexed Vec (no hashing) is far faster than a HashMap across
226    // the hundreds of millions of per-point updates. Sentinel intensity -1 = unoccupied;
227    // the source index lets us drop peaks the systematic sampler already emitted.
228    let n_cells = (PEAK_DIMS[0] * PEAK_DIMS[1] * PEAK_DIMS[2]) as usize;
229    // Peak grid, struct-of-arrays. The hot per-point compare reads only `best_i` (1M×4 B = 4 MB,
230    // cache-resident) instead of pulling a cache line of the 48 B (f32,u64,GpuPoint) tuple; the cold
231    // `best_gi`/`best_gp` are written only on a new max. `best_i` sentinel -1.0 (== the old tuple .0)
232    // so a real 0-intensity point still occupies an empty cell via strict `> -1.0`.
233    let mut best_i: Vec<f32> = vec![-1.0; n_cells];
234    let mut best_gi: Vec<u64> = vec![0; n_cells];
235    let mut best_gp: Vec<GpuPoint> = vec![ZERO_POINT; n_cells];
236
237    // Subsample intensities into a reservoir for percentile defaults.
238    let mut reservoir: Vec<f32> = Vec::with_capacity(RESERVOIR_CAP);
239    let mut kept_counter: u64 = 0;
240    let sample_every = (budget / RESERVOIR_CAP).max(1) as u64;
241
242    // Distribution histograms (over the kept systematic sample) for the levels-style filter
243    // UI. m/z·1/K0·RT bin over the normalized cube; intensity accumulates into a fine fixed
244    // log range, then rebins to the data's actual range at the end (so it is data-tight).
245    let mut hist_mz = [0u32; HIST_BINS];
246    let mut hist_im = [0u32; HIST_BINS];
247    let mut hist_rt = [0u32; HIST_BINS];
248    let mut logi_fine = [0u32; LOGI_FINE];
249    let mut i_lo_seen = f32::MAX;
250    let mut i_hi_seen = 0.0f32;
251    // 2D density projections onto the coordinate planes (the "you are here" minimaps).
252    let mut proj_mz_im = vec![0u32; PROJ_BINS * PROJ_BINS]; // x = m/z, y = 1/K0
253    let mut proj_mz_rt = vec![0u32; PROJ_BINS * PROJ_BINS]; // x = m/z, y = RT
254    let mut proj_im_rt = vec![0u32; PROJ_BINS * PROJ_BINS]; // x = 1/K0, y = RT
255
256    let mut chunk: Vec<GpuPoint> = Vec::with_capacity(CHUNK_POINTS);
257
258    macro_rules! cancelled {
259        () => {
260            matches!(cmd_rx.try_recv(), Ok(LoadCmd::Cancel))
261        };
262    }
263
264    // Send a message, blocking politely if the channel is full but still watching for
265    // cancellation so the loader can never wedge on a full channel during shutdown.
266    macro_rules! send_cancellable {
267        ($msg:expr) => {{
268            let mut m = $msg;
269            loop {
270                match tx.try_send(m) {
271                    Ok(()) => break,
272                    Err(TrySendError::Full(back)) => {
273                        if cancelled!() {
274                            return;
275                        }
276                        m = back;
277                        std::thread::sleep(Duration::from_millis(1));
278                    }
279                    Err(_) => return, // receiver gone
280                }
281            }
282        }};
283    }
284
285    // Flush a chunk, blocking politely if the channel is full but still watching for
286    // cancellation so the loader can never wedge.
287    macro_rules! flush {
288        () => {{
289            if !chunk.is_empty() {
290                let mut payload = std::mem::take(&mut chunk);
291                loop {
292                    match tx.try_send(LoadMsg::Chunk {
293                        generation: GENERATION,
294                        points: payload,
295                    }) {
296                        Ok(()) => break,
297                        Err(TrySendError::Full(LoadMsg::Chunk { points, .. })) => {
298                            if cancelled!() {
299                                return;
300                            }
301                            payload = points;
302                            std::thread::sleep(Duration::from_millis(1));
303                        }
304                        Err(_) => return, // receiver gone
305                    }
306                }
307                chunk = Vec::with_capacity(CHUNK_POINTS);
308            }
309        }};
310    }
311
312    let n_frames = match &mode {
313        LoaderMode::Real { frame_ids, .. } => frame_ids.len(),
314        LoaderMode::Demo(d) => d.num_frames(),
315    };
316    if n_frames == 0 {
317        let _ = tx.send(LoadMsg::Error("run has no frames".into()));
318        return;
319    }
320
321    // Open the real dataset once (SDK-free, thread-safe converter).
322    let dataset = match &mode {
323        LoaderMode::Real { path, .. } => Some(TimsDataset::new("NO_SDK", path, false, false)),
324        LoaderMode::Demo(_) => None,
325    };
326
327    let mut last_progress = -1.0f32;
328    // Cumulative point index across the whole run; keeping every `stride`-th is global
329    // systematic sampling (count-bounded, RT-unbiased).
330    let mut global_i: u64 = 0;
331    let stride_u64 = stride as u64;
332    // Countdown replacing `global_i % stride_u64 == 0` in the hot per-point gate (a division over
333    // 100M+ points). Starts at 0 so survivor index 0 is kept, then resets to stride-1 — identical
334    // phase to the modulo (stride==1 -> always 0 -> keep all). `global_i` still advances every
335    // survivor (it tags the peak grid + drives the peak-tail dedup `gi % stride`).
336    let mut keep_in: u64 = 0;
337
338    // Per-point handler shared by both sources: cull against the region filter first, then update
339    // the peak grid for every SURVIVING point and emit every stride-th into the systematic base.
340    macro_rules! handle_point {
341        ($mz:expr, $im:expr, $rt:expr, $it:expr, $ms2:expr) => {{
342            let intensity = $it;
343            // Region cull (the 4D lens): drop points outside the m/z·1/K0·intensity window at the
344            // source so the budget concentrates on the focused region. RT is handled by frame
345            // selection. `global_i` is not advanced for culled points, so the systematic stride
346            // counts only survivors.
347            if let Some(f) = &filter {
348                if $mz < f.mz.0 || $mz > f.mz.1 || $im < f.im.0 || $im > f.im.1
349                    || intensity < f.intensity_min
350                {
351                    continue;
352                }
353            }
354            let pos = bounds.normalize($mz, $im, $rt);
355            let flags = if $ms2 { GpuPoint::MS2_FLAG } else { 0 };
356            // Peak: keep the highest-intensity point per coarse cell (SoA — the hot read is best_i
357            // only; strict `>` keeps first-wins on ties).
358            let cell = peak_cell(pos) as usize;
359            if intensity > best_i[cell] {
360                best_i[cell] = intensity;
361                best_gi[cell] = global_i;
362                best_gp[cell] =
363                    GpuPoint { pos, intensity, weight: 1.0, flags, _pad: [GpuPoint::NO_CLUSTER, 0] };
364            }
365            // Systematic density base (every stride-th survivor, via the countdown above).
366            if keep_in == 0 {
367                keep_in = stride_u64 - 1;
368                chunk.push(GpuPoint { pos, intensity, weight, flags, _pad: [GpuPoint::NO_CLUSTER, 0] });
369                systematic_count += 1;
370                reservoir_push(&mut reservoir, &mut kept_counter, sample_every, intensity);
371                // Distribution histograms over the kept (representative) sample.
372                hist_mz[hbin(pos[0])] += 1;
373                hist_im[hbin(pos[1])] += 1;
374                hist_rt[hbin(pos[2])] += 1;
375                logi_fine[((intensity.max(1.0).log10() / LOGI_HI) * LOGI_FINE as f32)
376                    .clamp(0.0, (LOGI_FINE - 1) as f32) as usize] += 1;
377                proj_mz_im[pbin(pos[0]) + PROJ_BINS * pbin(pos[1])] += 1;
378                proj_mz_rt[pbin(pos[0]) + PROJ_BINS * pbin(pos[2])] += 1;
379                proj_im_rt[pbin(pos[1]) + PROJ_BINS * pbin(pos[2])] += 1;
380                i_lo_seen = i_lo_seen.min(intensity);
381                i_hi_seen = i_hi_seen.max(intensity);
382                if chunk.len() >= CHUNK_POINTS {
383                    flush!();
384                }
385            } else {
386                keep_in -= 1;
387            }
388            global_i += 1;
389        }};
390    }
391
392    match &mode {
393        LoaderMode::Real { frame_ids, .. } => {
394            use rayon::prelude::*;
395            let ds = dataset.as_ref().unwrap();
396            // Decode frames in parallel batches (the heavy, CPU-bound part — every core helps),
397            // then fold the decoded points in sequentially so the systematic sampler, peak grid
398            // and histograms stay exactly correct. One batch is in flight to bound memory.
399            const DECODE_BATCH: usize = 64;
400            let mut start = 0;
401            while start < n_frames {
402                if cancelled!() {
403                    return;
404                }
405                let end = (start + DECODE_BATCH).min(n_frames);
406                let frames: Vec<_> = frame_ids[start..end]
407                    .par_iter()
408                    .map(|&fid| ds.get_frame(fid))
409                    .collect();
410                for frame in &frames {
411                    let rt = frame.ims_frame.retention_time;
412                    let is_ms2 = !matches!(frame.ms_type, MsType::Precursor);
413                    let mz = frame.ims_frame.mz.as_slice();
414                    let im = frame.ims_frame.mobility.as_slice();
415                    let it = frame.ims_frame.intensity.as_slice();
416                    // Validate parallel arrays — never zip past the shortest.
417                    let n = mz.len().min(im.len()).min(it.len());
418                    for j in 0..n {
419                        handle_point!(mz[j], im[j], rt, it[j] as f32, is_ms2);
420                    }
421                }
422                start = end;
423                let progress = end as f32 / n_frames as f32;
424                if progress - last_progress >= 0.01 {
425                    let _ = tx.try_send(LoadMsg::Progress(progress));
426                    last_progress = progress;
427                }
428            }
429        }
430        LoaderMode::Demo(d) => {
431            for idx in 0..n_frames {
432                if cancelled!() {
433                    return;
434                }
435                for p in d.frame(idx) {
436                    handle_point!(p.mz, p.im, p.rt, p.intensity as f32, p.is_ms2);
437                }
438                let progress = (idx + 1) as f32 / n_frames as f32;
439                if progress - last_progress >= 0.01 {
440                    let _ = tx.try_send(LoadMsg::Progress(progress));
441                    last_progress = progress;
442                }
443            }
444        }
445    }
446
447    flush!();
448
449    // Emit the brightest peaks to fill the reserved budget headroom, but only peaks the
450    // systematic sampler did NOT already emit (source index not a multiple of stride) so
451    // they add genuinely-new sparse points rather than duplicates. Sort by intensity so
452    // the strongest survive if occupied cells exceed the headroom.
453    let peak_room = budget.saturating_sub(systematic_count as usize);
454    if peak_room > 0 && stride_u64 > 1 {
455        // Occupancy first (best_i >= 0.0, == the old `i >= 0.0`), THEN the systematic-dedup modulo —
456        // never run `% stride` on an empty cell.
457        let mut peak_pts: Vec<GpuPoint> = (0..n_cells)
458            .filter(|&c| best_i[c] >= 0.0 && best_gi[c] % stride_u64 != 0)
459            .map(|c| best_gp[c])
460            .collect();
461        // Keep the strongest `peak_room` peaks. A partial select (O(n)) suffices — order within the
462        // kept set is irrelevant (everything is shuffled downstream). Guard peak_room < len:
463        // select_nth at index == len is out of bounds, and peak_room >= len already keeps all.
464        if peak_room < peak_pts.len() {
465            peak_pts.select_nth_unstable_by(peak_room, |a, b| {
466                b.intensity
467                    .partial_cmp(&a.intensity)
468                    .unwrap_or(std::cmp::Ordering::Equal)
469            });
470            peak_pts.truncate(peak_room);
471        }
472        // Fold peaks into the distributions + intensity range too: they are displayed points
473        // and are often the brightest in the run, so excluding them would make the default
474        // intensity filter (which uses i_lo/i_hi) cull the very peaks preservation kept.
475        for gp in &peak_pts {
476            hist_mz[hbin(gp.pos[0])] += 1;
477            hist_im[hbin(gp.pos[1])] += 1;
478            hist_rt[hbin(gp.pos[2])] += 1;
479            logi_fine[((gp.intensity.max(1.0).log10() / LOGI_HI) * LOGI_FINE as f32)
480                .clamp(0.0, (LOGI_FINE - 1) as f32) as usize] += 1;
481            proj_mz_im[pbin(gp.pos[0]) + PROJ_BINS * pbin(gp.pos[1])] += 1;
482            proj_mz_rt[pbin(gp.pos[0]) + PROJ_BINS * pbin(gp.pos[2])] += 1;
483            proj_im_rt[pbin(gp.pos[1]) + PROJ_BINS * pbin(gp.pos[2])] += 1;
484            i_lo_seen = i_lo_seen.min(gp.intensity);
485            i_hi_seen = i_hi_seen.max(gp.intensity);
486        }
487        for batch in peak_pts.chunks(CHUNK_POINTS) {
488            send_cancellable!(LoadMsg::PeakChunk {
489                points: batch.to_vec(),
490            });
491        }
492    }
493
494    // Annotation overlay (real data only — needs the dataset's scan->mobility converter
495    // and the DDA/DIA metadata tables).
496    if let (LoaderMode::Real { path, .. }, Some(ds)) = (&mode, &dataset) {
497        let (lines, groups, n_groups) = build_annotations(ds, path, &bounds);
498        if !lines.is_empty() {
499            send_cancellable!(LoadMsg::Annotations { lines, groups, n_groups });
500        }
501    }
502
503    // Robust intensity range from the reservoir.
504    if !reservoir.is_empty() {
505        reservoir.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
506        let p = |q: f32| {
507            let i = ((reservoir.len() as f32 - 1.0) * q).round() as usize;
508            reservoir[i.min(reservoir.len() - 1)]
509        };
510        let i_p1 = p(0.01).max(1.0);
511        let i_p50 = p(0.50).max(i_p1);
512        let i_p99 = p(0.99).max(i_p50 * 1.0001);
513        send_cancellable!(LoadMsg::Stats { i_p1, i_p99, i_p50 });
514    }
515
516    // Rebin the fine fixed-range intensity histogram into the data's actual log range so the
517    // levels strip is data-tight, and ship all four distributions to the UI.
518    if i_hi_seen > 0.0 {
519        let i_lo = i_lo_seen.max(1.0);
520        let i_hi = i_hi_seen.max(i_lo * 1.0001);
521        let (llo, lhi) = (i_lo.log10(), i_hi.log10());
522        let span = (lhi - llo).max(1e-6);
523        let mut hist_i = vec![0u32; HIST_BINS];
524        for (src, &c) in logi_fine.iter().enumerate() {
525            if c == 0 {
526                continue;
527            }
528            let src_log = ((src as f32 + 0.5) / LOGI_FINE as f32) * LOGI_HI;
529            let t = (((src_log - llo) / span) * HIST_BINS as f32) as i32;
530            if t >= 0 && (t as usize) < HIST_BINS {
531                hist_i[t as usize] += c;
532            }
533        }
534        send_cancellable!(LoadMsg::Histograms {
535            mz: hist_mz.to_vec(),
536            im: hist_im.to_vec(),
537            rt: hist_rt.to_vec(),
538            intensity: hist_i,
539            i_lo,
540            i_hi,
541            proj_mz_im,
542            proj_mz_rt,
543            proj_im_rt,
544        });
545    }
546
547    send_cancellable!(LoadMsg::Done {
548        generation: GENERATION,
549    });
550}

556/// Read the DIA/MIDIA isolation windows and return their real-unit `(m/z × 1/K0)` footprints plus
557/// the max group id (the `group_color` denominator). Subsamples a fine MIDIA diagonal so it stays
558/// legible, widening each kept rect symmetrically across the skipped scans. Returns empty for
559/// non-DIA or unreadable runs.
560pub fn dia_window_rects(ds: &TimsDataset, path: &str) -> (Vec<WindowRect>, u32) {
561    let (windows, info) = match (read_dia_ms_ms_windows(path), read_dia_ms_ms_info(path)) {
562        (Ok(w), Ok(i)) => (w, i),
563        _ => return (Vec::new(), 0),
564    };
565    if windows.is_empty() {
566        return (Vec::new(), 0);
567    }
568    // A representative frame per window group, for scan->mobility conversion.
569    let mut grp_frame: std::collections::HashMap<u32, u32> = Default::default();
570    for di in &info {
571        grp_frame.entry(di.window_group).or_insert(di.frame_id);
572    }
573    // Subsample so a fine MIDIA diagonal (~15k windows) stays legible; a conventional DIA scheme
574    // (tens of windows) keeps every box (stride == 1).
575    const TARGET_WINDOWS: usize = 800;
576    let stride = (windows.len() / TARGET_WINDOWS).max(1);
577    let max_group = windows.iter().map(|w| w.window_group).max().unwrap_or(1).max(1);
578    let max_scan = windows.iter().map(|w| w.scan_num_end).max().unwrap_or(0);
579    let mut out = Vec::new();
580    for w in windows.iter().step_by(stride) {
581        let fid = grp_frame.get(&w.window_group).copied().unwrap_or(1);
582        // Widen SYMMETRICALLY across the skipped scans so the rect stays a visible window AND stays
583        // centered on the real window's mobility (widening only the high-scan end would shift the
584        // 1/K0 center). stride == 1 (conventional DIA) leaves it exact.
585        let half = stride as u32 / 2;
586        let scan_lo = w.scan_num_begin.saturating_sub(half);
587        let scan_hi = (w.scan_num_end + half).min(max_scan.max(w.scan_num_end));
588        let ims = ds.scan_to_inverse_mobility(fid, &vec![scan_lo, scan_hi]);
589        let im0 = ims.first().copied().unwrap_or(0.0);
590        let im1 = ims.get(1).copied().unwrap_or(im0);
591        out.push(WindowRect {
592            group: w.window_group,
593            mz0: w.isolation_mz - w.isolation_width * 0.5,
594            mz1: w.isolation_mz + w.isolation_width * 0.5,
595            im0,
596            im1,
597        });
598    }
599    (out, max_group)
600}

602/// metadata. DDA precursors -> small 3D crosses; DIA isolation windows -> wireframe
603/// boxes. Returns line-list vertices (consecutive pairs are segments).
604fn build_annotations(
605    ds: &TimsDataset,
606    path: &str,
607    bounds: &AxisBounds,
608) -> (Vec<LineVertex>, Vec<u32>, u32) {
609    // frame_id -> retention time (ids are contiguous 1..=N, validated at load).
610    let meta = match read_meta_data_sql(path) {
611        Ok(m) => m,
612        Err(_) => return (Vec::new(), Vec::new(), 0),
613    };
614    let max_id = meta.iter().map(|m| m.id).max().unwrap_or(0).max(0) as usize;
615    let mut rt_by = vec![0f64; max_id + 1];
616    for m in &meta {
617        if m.id >= 0 {
618            rt_by[m.id as usize] = m.time;
619        }
620    }
621    let rt_of = |fid: u32| rt_by.get(fid as usize).copied().unwrap_or(0.0);
622
623    // `groups` is a per-vertex window-group id, parallel to `lines`, for CPU-side group
624    // toggling. u32::MAX means "ungrouped, always shown" (DDA precursor crosses).
625    let mut lines: Vec<LineVertex> = Vec::new();
626    let mut groups: Vec<u32> = Vec::new();
627    let mut n_groups_out: u32 = 0;
628    match ds.get_acquisition_mode() {
629        AcquisitionMode::DDA | AcquisitionMode::PRECURSOR => {
630            if let Ok(precursors) = read_dda_precursor_meta(path) {
631                for p in precursors {
632                    let fid = p.precursor_frame_id.max(0) as u32;
633                    let scan = p.precursor_average_scan_number.round().max(0.0) as u32;
634                    let im = ds
635                        .scan_to_inverse_mobility(fid, &vec![scan])
636                        .first()
637                        .copied()
638                        .unwrap_or(0.0);
639                    let pos = bounds.normalize(p.precursor_mz_highest_intensity, im, rt_of(fid));
640                    push_cross(&mut lines, &mut groups, pos, 0.012, [0.1, 0.95, 0.95], u32::MAX);
641                }
642            }
643        }
644        AcquisitionMode::DIA => {
645            // The (m/z × 1/K0) selection footprints in real units, shared with the web overlay's
646            // `/windows` endpoint so the two never drift.
647            let (rects, max_group) = dia_window_rects(ds, path);
648            n_groups_out = max_group;
649            let (rt_lo, rt_hi) = (bounds.rt.min, bounds.rt.max);
650            for r in &rects {
651                let color = crate::render::colormap::group_color(r.group, max_group);
652                // Draw the footprint at several evenly spaced RT slices, so the recurring isolation
653                // scheme sits on the precursor signal through the whole run rather than piling onto
654                // the two RT end-walls.
655                for i in 0..DIA_WINDOW_RT_SLICES {
656                    let rt = rt_lo
657                        + (rt_hi - rt_lo) * (i as f64 + 0.5) / DIA_WINDOW_RT_SLICES as f64;
658                    push_rect_mz_im(
659                        &mut lines,
660                        &mut groups,
661                        bounds,
662                        (r.mz0, r.mz1),
663                        (r.im0, r.im1),
664                        rt,
665                        color,
666                        r.group,
667                    );
668                }
669            }
670        }
671        AcquisitionMode::Unknown => {}
672    }
673    (lines, groups, n_groups_out)
674}

676/// Append a 3-segment axis-aligned cross centered at `c` (half-length `h`), in `color`,
677/// tagging each vertex with `group`.
678fn push_cross(
679    lines: &mut Vec<LineVertex>,
680    groups: &mut Vec<u32>,
681    c: [f32; 3],
682    h: f32,
683    color: [f32; 3],
684    group: u32,
685) {
686    let mut v = |p: [f32; 3]| {
687        lines.push(LineVertex::new(p, color));
688        groups.push(group);
689    };
690    v([c[0] - h, c[1], c[2]]);
691    v([c[0] + h, c[1], c[2]]);
692    v([c[0], c[1] - h, c[2]]);
693    v([c[0], c[1] + h, c[2]]);
694    v([c[0], c[1], c[2] - h]);
695    v([c[0], c[1], c[2] + h]);
696}

698/// Append the 4 edges of an axis-aligned rectangle in the m/z x mobility plane at a fixed
699/// retention time — one isolation window's footprint, drawn at a given RT slice.
700fn push_rect_mz_im(
701    lines: &mut Vec<LineVertex>,
702    groups: &mut Vec<u32>,
703    bounds: &AxisBounds,
704    mz: (f64, f64),
705    im: (f64, f64),
706    rt: f64,
707    color: [f32; 3],
708    group: u32,
709) {
710    let corner = |a: f64, b: f64| bounds.normalize(a, b, rt);
711    let c = [
712        corner(mz.0, im.0),
713        corner(mz.1, im.0),
714        corner(mz.1, im.1),
715        corner(mz.0, im.1),
716    ];
717    const EDGES: [(usize, usize); 4] = [(0, 1), (1, 2), (2, 3), (3, 0)];
718    for (a, b) in EDGES {
719        lines.push(LineVertex::new(c[a], color));
720        lines.push(LineVertex::new(c[b], color));
721        groups.push(group);
722        groups.push(group);
723    }
724}

726#[inline]
727fn reservoir_push(reservoir: &mut Vec<f32>, counter: &mut u64, sample_every: u64, v: f32) {
728    if *counter % sample_every == 0 && reservoir.len() < RESERVOIR_CAP {
729        reservoir.push(v);
730    }
731    *counter = counter.wrapping_add(1);
732}

764    #[test]
765    fn cumulative_systematic_sampling_is_bounded() {
766        // The loader keeps every stride-th point by a cumulative global index, which is
767        // global systematic sampling: kept == ceil(total/stride), independent of how the
768        // points are split into frames.
769        for stride in [2u64, 3, 5, 7] {
770            let frame_lens = [1u64, 2, 1, 0, 9, 4, 3, 1, 1, 1];
771            let total: u64 = frame_lens.iter().sum();
772            let mut global_i = 0u64;
773            let mut kept = 0u64;
774            for &n in &frame_lens {
775                for _ in 0..n {
776                    if global_i % stride == 0 {
777                        kept += 1;
778                    }
779                    global_i += 1;
780                }
781            }
782            let bound = total.div_ceil(stride);
783            assert_eq!(kept, bound, "stride={stride}: kept {kept} != ceil bound {bound}");
784        }

858    /// FOCUS_LENS_PLAN.md blocker-1 proof: a region load must spend its budget on the *region*,
859    /// not stay over-strided on the full-run estimate. Drives the systematic-base count.
860    #[test]
861    fn region_estimate_concentrates_budget() {
862        let total = 200_000u64;
863        let demo = || DemoSource::new(30, total);
864        // m/z sub-range; im wide open so only m/z culls (isolates a clean strict subset).
865        let region = Some(RegionFilter { mz: (100.0, 500.0), im: (-10.0, 10.0), intensity_min: 0.0 });
866
867        // True survivors in the region (stride 1: budget exceeds everything).
868        let (survivors, _) =
869            drain(LoaderHandle::spawn(LoaderMode::Demo(demo()), demo_bounds(), total, total as usize * 2, region));
870        assert!(survivors > 1000, "region too small to test: {survivors}");
871        assert!((survivors as u64) < total, "region should be a strict subset of the run");
872
873        let budget = (survivors / 2).max(2000); // below survivors so sampling is actually active
874
875        // Region-relative estimate (= survivors): the systematic base should ~fill the budget.
876        let (sys_region, _) =
877            drain(LoaderHandle::spawn(LoaderMode::Demo(demo()), demo_bounds(), survivors as u64, budget, region));
878        // Full-run estimate (the bug): stride stays coarse -> budget badly under-spent.
879        let (sys_full, _) =
880            drain(LoaderHandle::spawn(LoaderMode::Demo(demo()), demo_bounds(), total, budget, region));
881
882        // Exact: the systematic base keeps every `stride`-th SURVIVOR, where the stride is sized to
883        // the survivor count. This equals ceil(survivors/stride) only if `global_i` advances solely
884        // for survivors — it would fail if culled points still advanced the index.
885        let systematic_budget = (budget * 85 / 100).max(1);
886        let stride = stride_for(survivors as u64, systematic_budget) as u64;
887        let expected = (survivors as u64).div_ceil(stride) as usize;
888        assert_eq!(
889            sys_region, expected,
890            "region systematic count must be exactly ceil(survivors/stride): \
891             survivors={survivors} stride={stride}"
892        );
893        // The full-run estimate sizes the stride to ~total, spending only ~(survivors/total) of the
894        // budget — far less than the region estimate. This is the blocker-1 regression guard.
895        assert!(
896            (sys_full as f64) < 0.5 * sys_region as f64,
897            "full-run estimate did not under-spend (blocker-1 unfixed): full={sys_full} region={sys_region}"
898        );
899    }
