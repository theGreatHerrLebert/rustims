1//! Run metadata read from the TDF SQLite tables — cheap, no binary frame decoding.
2//!
3//! This gives us the axis ranges, a frame-id → retention-time map, per-frame MS type,
4//! and a total-point estimate before touching `analysis.tdf_bin`, so the camera and
5//! axis cube can render instantly while points stream in.
6
7use anyhow::{Context, Result};
8
9use rustdf::data::meta::{read_global_meta_sql, read_meta_data_sql};
10
11use super::point::{AxisBounds, AxisTransform};
12
13/// Per-frame info we keep resident for the whole session.
14/// `is_ms2`/`num_peaks` back per-frame LOD and filtering planned for later phases.
15#[allow(dead_code)]
16#[derive(Clone, Copy, Debug)]
17pub struct FrameInfo {
18    pub id: u32,
19    pub retention_time: f64,
20    /// True for MS2 / fragment frames (`ms_ms_type != 0`).
21    pub is_ms2: bool,
22    pub num_peaks: u64,
23}
24
25/// Lightweight index over a run's metadata.
26#[derive(Clone)]
27pub struct MetaIndex {
28    pub data_path: String,
29    pub frames: Vec<FrameInfo>,
30    pub bounds: AxisBounds,
31    pub total_points_estimate: u64,
32    /// TIMS scans per frame (the mobility ramp length; constant across frames). Used to anchor the
33    /// clustering's 1/K0 reach to a fixed number of scans (run-level, focus-independent).
34    pub num_scans: u32,
35}
36
37impl MetaIndex {
38    /// Load metadata for a real `.d` folder.
39    pub fn load(data_path: &str) -> Result<Self> {
40        let global = read_global_meta_sql(data_path)
41            .map_err(|e| anyhow::anyhow!("read_global_meta_sql failed: {e:?}"))
42            .context("reading global TDF metadata")?;
43        let meta = read_meta_data_sql(data_path)
44            .map_err(|e| anyhow::anyhow!("read_meta_data_sql failed: {e:?}"))
45            .context("reading per-frame TDF metadata")?;
46
47        anyhow::ensure!(!meta.is_empty(), "TDF run contains no frames");
48
49        let mut frames = Vec::with_capacity(meta.len());
50        let mut rt_min = f64::INFINITY;
51        let mut rt_max = f64::NEG_INFINITY;
52        let mut total: u64 = 0;
53        for m in &meta {
54            // Keep FrameMeta.id as the frame id we pass to get_frame. rustdf's get_frame
55            // indexes its own frame_meta_data by position (id - 1) from the same
56            // unordered SQL query, so we require ids to appear in exact 1..=N positional
57            // order (validated below). Checked-convert to avoid an i64->u32 wrap.
58            let id = u32::try_from(m.id)
59                .map_err(|_| anyhow::anyhow!("frame id {} is out of u32 range", m.id))?;
60            let num_peaks = m.num_peaks.max(0) as u64;
61            if num_peaks > 0 {
62                rt_min = rt_min.min(m.time);
63                rt_max = rt_max.max(m.time);
64            }
65            total = total.saturating_add(num_peaks);
66            frames.push(FrameInfo {
67                id,
68                retention_time: m.time,
69                is_ms2: m.ms_ms_type != 0,
70                num_peaks,
71            });
72        }
73
74        // rustdf looks up frames by vector position (id - 1) from the same query, so the
75        // metadata must arrive in exact 1..=N order. Checking the id *set* isn't enough:
76        // an unordered result like [2,1,3] passes a set check but makes get_frame(2) read
77        // position 1 (id 1's data). Require positional order so this can't silently
78        // corrupt the view; timsTOF runs satisfy it.
79        let n = frames.len();
80        let ordered = frames
81            .iter()
82            .enumerate()
83            .all(|(i, f)| f.id == i as u32 + 1);
84        anyhow::ensure!(
85            ordered,
86            "frame metadata is not in contiguous 1..={n} positional order \
87             (first id {:?}, last id {:?}); the viewer relies on rustdf's position-based \
88             frame indexing",
89            frames.first().map(|f| f.id),
90            frames.last().map(|f| f.id),
91        );
92
93        if !rt_min.is_finite() || !rt_max.is_finite() {
94            // No non-empty frames carried RT; fall back to the full frame span.
95            rt_min = frames.first().map(|f| f.retention_time).unwrap_or(0.0);
96            rt_max = frames.last().map(|f| f.retention_time).unwrap_or(1.0);
97        }
98
99        let bounds = AxisBounds {
100            mz: AxisTransform::new(
101                global.mz_acquisition_range_lower,
102                global.mz_acquisition_range_upper,
103            ),
104            im: AxisTransform::new(
105                global.one_over_k0_range_lower,
106                global.one_over_k0_range_upper,
107            ),
108            rt: AxisTransform::new(rt_min, rt_max),
109        };
110
111        // The mobility ramp length (constant across frames). Use the mode (most common value) so a
112        // single corrupt frame — a stray 0 or an absurdly high count — can't skew the anchor.
113        let num_scans = {
114            let mut counts: std::collections::HashMap<u32, usize> = std::collections::HashMap::new();
115            for m in &meta {
116                let s = m.num_scans.max(0) as u32;
117                if s > 0 {
118                    *counts.entry(s).or_default() += 1;
119                }
120            }
121            counts.into_iter().max_by_key(|&(_, c)| c).map(|(s, _)| s).unwrap_or(0)
122        };
123
124        Ok(MetaIndex {
125            data_path: data_path.to_string(),
126            frames,
127            bounds,
128            total_points_estimate: total,
129            num_scans,
130        })
131    }
132
133    /// Synthetic metadata for the `DEMO` path (no Bruker data required).
134    pub fn demo(num_frames: usize, total_points: u64) -> Self {
135        let mut frames = Vec::with_capacity(num_frames);
136        for i in 0..num_frames {
137            let rt = i as f64 * 0.5; // 0.5 s spacing
138            frames.push(FrameInfo {
139                id: (i + 1) as u32,
140                retention_time: rt,
141                is_ms2: i % 10 == 9, // ~10% MS2
142                num_peaks: total_points / num_frames.max(1) as u64,
143            });
144        }
145        let bounds = AxisBounds {
146            mz: AxisTransform::new(100.0, 1700.0),
147            im: AxisTransform::new(0.6, 1.6),
148            rt: AxisTransform::new(0.0, (num_frames.max(1) - 1) as f64 * 0.5),
149        };
150        MetaIndex {
151            data_path: "DEMO".to_string(),
152            frames,
153            bounds,
154            total_points_estimate: total_points,
155            num_scans: 709, // a typical timsTOF mobility ramp length
156        }
157    }
158}
