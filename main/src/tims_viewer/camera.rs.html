1//! Orbit camera around the normalized data cube.
2
3use glam::{Mat3, Mat4, Vec3};
4
5use crate::render::uniforms::CameraUniform;
6
7#[derive(Clone, Copy, Debug, PartialEq, Eq)]
8pub enum Projection {
9    Perspective,
10    /// Hidden from the UI until its bugs are fixed; the projection math is kept.
11    #[allow(dead_code)]
12    Orthographic,
13}
14
15#[derive(Clone, Copy, Debug)]
16pub enum AxisView {
17    /// Look down the m/z axis.
18    Mz,
19    /// Look down the mobility axis.
20    Mobility,
21    /// Look down the retention-time axis.
22    Rt,
23}
24
25pub struct OrbitCamera {
26    pub target: Vec3,
27    pub distance: f32,
28    /// Azimuth around world Y, radians.
29    pub yaw: f32,
30    /// Elevation, radians, clamped away from the poles.
31    pub pitch: f32,
32    pub fovy: f32,
33    pub znear: f32,
34    pub zfar: f32,
35    pub projection: Projection,
36    /// Which data axis points up ("topple the dice"): 0 = 1/K0, 1 = m/z, 2 = RT. Applied as a
37    /// cyclic permutation of the data axes baked into the view-projection.
38    pub roll: u8,
39}
40
41/// Name of the data axis currently pointing up, for the UI.
42pub const ROLL_UP_AXIS: [&str; 3] = ["1/K0", "m/z", "RT"];
43
44impl Default for OrbitCamera {
45    fn default() -> Self {
46        OrbitCamera {
47            target: Vec3::ZERO,
48            distance: 4.0,
49            yaw: 0.7,
50            pitch: 0.5,
51            fovy: 45f32.to_radians(),
52            znear: 0.05,
53            zfar: 100.0,
54            projection: Projection::Perspective,
55            roll: 0,
56        }
57    }
58}
59
60const PITCH_LIMIT: f32 = 1.5533; // ~89 degrees
61
62impl OrbitCamera {
63    pub fn reset(&mut self) {
64        let (proj, roll) = (self.projection, self.roll);
65        *self = OrbitCamera::default();
66        self.projection = proj;
67        self.roll = roll;
68    }
69
70    /// Cycle which data axis points up (0 → 1/K0, 1 → m/z, 2 → RT).
71    pub fn roll_axis(&mut self) {
72        self.roll = (self.roll + 1) % 3;
73    }
74
75    /// Cyclic axis-permutation matrix for the current roll, mapping data (x=m/z, y=1/K0, z=RT)
76    /// to world so the chosen axis lands on world-up (Y). Identity, or a proper rotation.
77    fn perm(&self) -> Mat4 {
78        let m3 = match self.roll % 3 {
79            // 1/K0 (data y) up: identity.
80            0 => Mat3::IDENTITY,
81            // m/z (data x) up: data x→world y, y→z, z→x.
82            1 => Mat3::from_cols(Vec3::Y, Vec3::Z, Vec3::X),
83            // RT (data z) up: data x→world z, y→x, z→y.
84            _ => Mat3::from_cols(Vec3::Z, Vec3::X, Vec3::Y),
85        };
86        Mat4::from_mat3(m3)
87    }
88
89    pub fn set_axis_view(&mut self, axis: AxisView) {
90        self.target = Vec3::ZERO;
91        self.distance = 4.0;
92        // `perm()` maps DATA axes onto WORLD axes under the current roll (m/z→[X,Y,Z], 1/K0→[Y,Z,X],
93        // RT→[Z,X,Y] for roll 0/1/2). The camera yaw/pitch are world-space, so snap to look down the
94        // WORLD axis the requested data axis currently occupies — otherwise, after "Roll up", the m/z
95        // button would look down whatever data axis happens to sit on world-X. roll 0 reduces to the
96        // identity mapping (Mz→X, Mobility→Y, Rt→Z), i.e. the original behavior, unchanged.
97        let world_axis = match (axis, self.roll % 3) {
98            (AxisView::Mz, 0) | (AxisView::Mobility, 2) | (AxisView::Rt, 1) => 0u8, // world X
99            (AxisView::Mz, 1) | (AxisView::Mobility, 0) | (AxisView::Rt, 2) => 1u8, // world Y (up)
100            _ => 2u8,                                                               // world Z
101        };
102        let (yaw, pitch) = match world_axis {
103            0 => (std::f32::consts::FRAC_PI_2, 0.0),
104            1 => (0.0, PITCH_LIMIT),
105            _ => (0.0, 0.0),
106        };
107        self.yaw = yaw;
108        self.pitch = pitch;
109    }
110
111    fn eye(&self) -> Vec3 {
112        let (sp, cp) = self.pitch.sin_cos();
113        let (sy, cy) = self.yaw.sin_cos();
114        let dir = Vec3::new(cp * sy, sp, cp * cy);
115        self.target + dir * self.distance
116    }
117
118    /// World-space right and up vectors of the camera (for billboarding).
119    fn basis(&self) -> (Vec3, Vec3, Vec3) {
120        let eye = self.eye();
121        let forward = (self.target - eye).normalize_or_zero();
122        let right = forward.cross(Vec3::Y).normalize_or_zero();
123        let up = right.cross(forward).normalize_or_zero();
124        (right, up, forward)
125    }
126
127    pub fn orbit(&mut self, dx: f32, dy: f32) {
128        self.yaw -= dx * 0.005;
129        self.pitch = (self.pitch + dy * 0.005).clamp(-PITCH_LIMIT, PITCH_LIMIT);
130    }
131
132    pub fn pan(&mut self, dx: f32, dy: f32) {
133        let (right, up, _) = self.basis();
134        let scale = self.distance * 0.0015;
135        self.target += (-right * dx + up * dy) * scale;
136    }
137
138    pub fn zoom(&mut self, scroll: f32) {
139        self.distance = (self.distance * (-scroll * 0.1).exp()).clamp(0.2, 50.0);
140    }
141
142    pub fn view_proj(&self, aspect: f32) -> Mat4 {
143        let eye = self.eye();
144        let view = Mat4::look_at_rh(eye, self.target, Vec3::Y);
145        let proj = match self.projection {
146            Projection::Perspective => {
147                Mat4::perspective_rh(self.fovy, aspect.max(1e-3), self.znear, self.zfar)
148            }
149            Projection::Orthographic => {
150                let half_h = self.distance * (self.fovy * 0.5).tan();
151                let half_w = half_h * aspect.max(1e-3);
152                Mat4::orthographic_rh(-half_w, half_w, -half_h, half_h, self.znear, self.zfar)
153            }
154        };
155        // Bake the axis permutation in last: data positions are permuted, then viewed. The
156        // volume raycaster reconstructs data-space coords through the inverse, so it stays
157        // consistent automatically.
158        proj * view * self.perm()
159    }
160
161    /// Inverse view-projection (column-major arrays), for reconstructing world-space
162    /// rays in the volume raycaster.
163    pub fn inv_view_proj(&self, aspect: f32) -> [[f32; 4]; 4] {
164        self.view_proj(aspect).inverse().to_cols_array_2d()
165    }
166
167    pub fn to_uniform(&self, aspect: f32, viewport: [f32; 2]) -> CameraUniform {
168        let (right, up, _) = self.basis();
169        // Billboards expand in DATA space (before the baked permutation), so counter-rotate the
170        // camera screen axes by perm⁻¹ — then perm maps them back to the world screen axes.
171        let inv = self.perm().inverse();
172        let right = inv.transform_vector3(right);
173        let up = inv.transform_vector3(up);
174        CameraUniform {
175            view_proj: self.view_proj(aspect).to_cols_array_2d(),
176            right: [right.x, right.y, right.z, 0.0],
177            up: [up.x, up.y, up.z, 0.0],
178            viewport: [viewport[0], viewport[1], 0.0, 0.0],
179        }
180    }
181}
