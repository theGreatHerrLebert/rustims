1//! "Nice" axis tick selection: 1/2/5 × 10^k steps, 4–6 ticks across a range.
2//!
3//! Pure, testable tick math with no egui/wgpu deps. The renderer (`app.rs`) and the
4//! overlay (`ui.rs`) consume these to draw 3D tick marks plus numeric labels.
5
6/// A computed tick: real-unit value plus its normalized cube coordinate in [-1,1].
7#[derive(Clone, Copy, Debug)]
8pub struct Tick {
9    pub value: f64,
10    pub norm: f32,
11}
12
13/// Axis identity, for per-axis label formatting.
14#[derive(Clone, Copy, Debug, PartialEq, Eq)]
15pub enum Axis {
16    Mz,
17    Im,
18    Rt,
19}
20
21/// RT spans wider than this (seconds) are labeled in minutes instead of seconds.
22pub const RT_MINUTES_SPAN: f64 = 600.0;
23
24/// Choose a "nice" step so that [lo,hi] is covered by ~4–6 ticks.
25/// Step ∈ {1,2,5}·10^k. Returns the step size (>0). `target` is the desired tick count.
26pub fn nice_step(lo: f64, hi: f64, target: u32) -> f64 {
27    let span = (hi - lo).abs();
28    if span <= 0.0 || !span.is_finite() {
29        return 1.0; // degenerate guard
30    }
31    let raw = span / target.max(1) as f64; // ideal step if steps were continuous
32    let mag = 10f64.powf(raw.log10().floor()); // power-of-ten below raw
33    let norm = raw / mag; // in [1,10)
34    let nice = if norm < 1.5 {
35        1.0
36    } else if norm < 3.0 {
37        2.0
38    } else if norm < 7.0 {
39        5.0
40    } else {
41        10.0
42    };
43    nice * mag
44}

46/// Generate ticks across [lo,hi] (auto-orders lo/hi). First tick = ceil(lo/step)*step,
47/// stepping by `step` while <= hi (+epsilon). Each tick's `norm` is computed by `to_norm`
48/// (pass the axis's AxisTransform::normalize, or a window-aware normalizer).
49pub fn ticks_for<F: Fn(f64) -> f32>(lo: f64, hi: f64, target: u32, to_norm: F) -> Vec<Tick> {
50    let (lo, hi) = (lo.min(hi), lo.max(hi));
51    let step = nice_step(lo, hi, target);
52    let first = (lo / step).ceil() * step; // first tick >= lo on the grid
53    let eps = step * 1e-6; // avoid dropping the last tick to fp error
54    let mut out = Vec::new();
55    let mut v = first;
56    while v <= hi + eps {
57        // snap near-zero to exactly 0 so formatting shows "0" not "-0"/"1e-13".
58        let vs = if v.abs() < eps { 0.0 } else { v };
59        out.push(Tick {
60            value: vs,
61            norm: to_norm(vs),
62        });
63        v += step;
64        if out.len() > 64 {
65            break; // hard safety cap
66        }
67    }
68    out
69}

71/// Decimal places needed to distinguish adjacent ticks at the given step size.
72/// A step of 200 needs 0; 0.2 needs 1; 0.02 needs 2, etc. Clamped to [0, 6].
73fn decimals_for_step(step: f64) -> usize {
74    if !step.is_finite() || step <= 0.0 {
75        return 0;
76    }
77    // Digits after the point so the step is resolvable (e.g. step 0.2 -> 1, 0.05 -> 2).
78    let d = -step.log10().floor();
79    (d.max(0.0) as usize).min(6)
80}

82/// Format a tick value for the given axis and the visible span (span chooses RT units and,
83/// together with the tick count, the decimal precision so adjacent ticks render distinctly).
84pub fn fmt_tick(axis: Axis, value: f64, span: f64) -> String {
85    // Recover the per-axis step from the span so a narrow focus window (sub-unit steps)
86    // gets enough decimals; ~5 ticks across the span matches `ticks_for`'s target.
87    let step = nice_step(0.0, span, 5);
88    match axis {
89        // m/z: integer Th normally, but a narrow focus window can need sub-Th precision.
90        Axis::Mz => format!("{:.*}", decimals_for_step(step), value),
91        // 1/K0: at least 2 decimals (its native scale), more if the window is very narrow.
92        Axis::Im => format!("{:.*}", decimals_for_step(step).max(2), value),
93        Axis::Rt => {
94            if span <= RT_MINUTES_SPAN {
95                format!("{:.*}", decimals_for_step(step), value)
96            } else {
97                // Minutes: span/60 sets the resolvable precision (floor at 1 decimal).
98                format!("{:.*}", decimals_for_step(step / 60.0).max(1), value / 60.0)
99            }
100        }
101    }
102}

104#[cfg(test)]

108    /// Is `v` of the form {1,2,5}·10^k (within fp tolerance)?
109    fn is_nice(v: f64) -> bool {
110        if v <= 0.0 || !v.is_finite() {
111            return false;
112        }
113        let mag = 10f64.powf(v.log10().floor());
114        let mantissa = v / mag;
115        [1.0, 2.0, 5.0, 10.0]
116            .iter()
117            .any(|m| (mantissa - m).abs() < 1e-6)
118    }
119
120    #[test]
121    fn nice_step_is_one_two_five() {
122        for &(lo, hi) in &[(0.0, 1600.0), (0.0, 0.8), (0.0, 12.0), (100.0, 1700.0)] {
123            let s = nice_step(lo, hi, 5);
124            assert!(is_nice(s), "step {s} for [{lo},{hi}] is not 1/2/5·10^k");
125        }
126        // span 1600 / 5 = 320 -> raw mag 100, norm 3.2 -> 5 -> 500.
127        assert_eq!(nice_step(0.0, 1600.0, 5), 500.0);
128        // span 0.8 / 5 = 0.16 -> raw mag 0.1, norm 1.6 -> 2 -> 0.2.
129        assert!((nice_step(0.0, 0.8, 5) - 0.2).abs() < 1e-9);
130    }
131
132    #[test]
133    fn ticks_span_range_monotonic() {
134        let ticks = ticks_for(100.0, 1700.0, 5, |v| v as f32);
135        // target=5 with the {1,2,5} snap yields ~4–6; a span landing on a step boundary
136        // (here step=500 over [100,1700] -> 500/1000/1500) bottoms out at 3, which the
137        // spec documents as acceptable and the 64-cap bounds the top end.
138        assert!(
139            (3..=7).contains(&ticks.len()),
140            "expected 3–7 ticks, got {}",
141            ticks.len()
142        );
143        let step = nice_step(100.0, 1700.0, 5);
144        let first = (100.0_f64 / step).ceil() * step;
145        assert!((ticks[0].value - first).abs() < 1e-6);
146        for t in &ticks {
147            assert!(t.value >= 100.0 - 1e-6 && t.value <= 1700.0 + 1e-6);
148        }
149        for w in ticks.windows(2) {
150            assert!(w[1].value > w[0].value, "ticks not increasing");
151        }
152    }

154    #[test]
155    fn degenerate_range_does_not_loop() {
156        let ticks = ticks_for(5.0, 5.0, 5, |v| v as f32);
157        assert!(!ticks.is_empty());
158        assert!(ticks.iter().all(|t| t.value.is_finite()));
159        assert!(ticks.len() <= 65);
160    }

162    #[test]
163    fn fmt_tick_per_axis() {
164        assert_eq!(fmt_tick(Axis::Mz, 412.7, 1600.0), "413");
165        assert_eq!(fmt_tick(Axis::Im, 1.234, 1.0), "1.23");
166        // RT seconds branch (span <= 600).
167        assert_eq!(fmt_tick(Axis::Rt, 120.0, 300.0), "120");
168        // RT minutes branch (span > 600): 1800 s -> 30.0 min.
169        assert_eq!(fmt_tick(Axis::Rt, 1800.0, 3600.0), "30.0");
170    }

172    #[test]
173    fn fmt_tick_narrow_window_gains_precision() {
174        // Narrow m/z focus window (~2 Th span -> step 0.5): adjacent ticks must NOT
175        // collapse to the same integer label.
176        let a = fmt_tick(Axis::Mz, 412.4, 2.0);
177        let b = fmt_tick(Axis::Mz, 412.6, 2.0);
178        assert_ne!(a, b, "narrow m/z ticks collapsed to the same label: {a} == {b}");
179        assert_eq!(a, "412.4");
180        assert_eq!(b, "412.6");
181        // Narrow 1/K0 window (~0.04 span -> step 0.01): needs 2 decimals minimum.
182        assert_ne!(fmt_tick(Axis::Im, 0.812, 0.04), fmt_tick(Axis::Im, 0.818, 0.04));
183    }
