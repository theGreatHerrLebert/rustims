95const DEPTH_FORMAT: wgpu::TextureFormat = wgpu::TextureFormat::Depth32Float;
96
97fn create_depth(device: &wgpu::Device, config: &wgpu::SurfaceConfiguration) -> wgpu::TextureView {
98    let tex = device.create_texture(&wgpu::TextureDescriptor {
99        label: Some("depth"),
100        size: wgpu::Extent3d {
101            width: config.width.max(1),
102            height: config.height.max(1),
103            depth_or_array_layers: 1,
104        },
105        mip_level_count: 1,
106        sample_count: 1,
107        dimension: wgpu::TextureDimension::D2,
108        format: DEPTH_FORMAT,
109        usage: wgpu::TextureUsages::RENDER_ATTACHMENT,
110        view_formats: &[],
111    });
112    tex.create_view(&wgpu::TextureViewDescriptor::default())
113}

115impl App {
116    fn init_gfx(&mut self, event_loop: &ActiveEventLoop) -> Result<Gfx> {
117        let plan = self.plan.take().expect("plan already consumed");
118
119        let attrs = Window::default_attributes()
120            .with_title("tims-viewer")
121            .with_inner_size(winit::dpi::LogicalSize::new(1280.0, 800.0));
122        let window = Arc::new(event_loop.create_window(attrs)?);
123
124        let instance = wgpu::Instance::new(wgpu::InstanceDescriptor {
125            backends: wgpu::Backends::PRIMARY | wgpu::Backends::GL,
126            ..Default::default()
127        });
128        let surface = instance.create_surface(window.clone())?;
129
130        let adapter = pollster::block_on(instance.request_adapter(
131            &wgpu::RequestAdapterOptions {
132                power_preference: wgpu::PowerPreference::HighPerformance,
133                force_fallback_adapter: false,
134                compatible_surface: Some(&surface),
135            },
136        ))
137        .ok_or_else(|| anyhow::anyhow!("no suitable GPU adapter found"))?;
138
139        log::info!("adapter: {:?}", adapter.get_info());
140        let limits = adapter.limits();
141        // GPU compaction needs compute + indirect execution; fall back to draw-all
142        // (vertex-shader culling) on backends that lack them (e.g. some GL drivers).
143        let dl_flags = adapter.get_downlevel_capabilities().flags;
144        let supports_compaction = dl_flags.contains(wgpu::DownlevelFlags::COMPUTE_SHADERS)
145            && dl_flags.contains(wgpu::DownlevelFlags::INDIRECT_EXECUTION);
146        let (device, queue) = pollster::block_on(adapter.request_device(
147            &wgpu::DeviceDescriptor {
148                label: Some("tims-viewer-device"),
149                required_features: wgpu::Features::empty(),
150                required_limits: limits.clone(),
151                memory_hints: wgpu::MemoryHints::Performance,
152            },
153            None,
154        ))?;
155
156        let size = window.inner_size();
157        let caps = surface.get_capabilities(&adapter);
158        let format = caps
159            .formats
160            .iter()
161            .copied()
162            .find(|f| f.is_srgb())
163            .unwrap_or(caps.formats[0]);
164        let config = wgpu::SurfaceConfiguration {
165            usage: wgpu::TextureUsages::RENDER_ATTACHMENT,
166            format,
167            width: size.width.max(1),
168            height: size.height.max(1),
169            present_mode: wgpu::PresentMode::Fifo,
170            desired_maximum_frame_latency: 2,
171            alpha_mode: caps.alpha_modes[0],
172            view_formats: vec![],
173        };
174        surface.configure(&device, &config);
175        let depth_view = create_depth(&device, &config);
176
177        // Derive a safe capacity from device limits and the requested budget.
178        let bytes_per_point = std::mem::size_of::<crate::data::point::GpuPoint>() as u64;
179        let max_by_buffer = (limits.max_buffer_size / bytes_per_point) as usize;
180        let total = plan.meta.total_points_estimate;
181        let mut cap = plan.budget.min(max_by_buffer).min((total as usize).max(1));
182        // When compaction is on, the master + compacted buffers are bound as storage,
183        // which is capped by max_storage_buffer_binding_size (often much smaller than
184        // max_buffer_size, e.g. 128 MiB) — exceeding it fails bind-group validation.
185        if supports_compaction {
186            let max_by_storage =
187                (limits.max_storage_buffer_binding_size as u64 / bytes_per_point) as usize;
188            cap = cap.min(max_by_storage);
189        }
190        let capacity = cap.max(1) as u32;
191        log::info!(
192            "point budget={} capacity={} compaction={} (device max_buffer_size={} MiB)",
193            plan.budget,
194            capacity,
195            supports_compaction,
196            limits.max_buffer_size / (1024 * 1024)
197        );
198
199        let points = PointCloudRenderer::new(
200            &device,
201            &queue,
202            format,
203            DEPTH_FORMAT,
204            capacity,
205            supports_compaction,
206        );
207        let volume = VolumeRenderer::new(&device, &queue, format, DEPTH_FORMAT, VOLUME_DIMS);
208        let grid = VolumeGrid::new(VOLUME_DIMS);
209        let annotations = AnnotationRenderer::new(&device, format, DEPTH_FORMAT);
210
211        let n_colormaps = COLORMAP_NAMES.len() as u32;
212        let mut state = AppState::new(plan.meta.bounds, total, n_colormaps);
213        state.capacity = capacity;
214        state.downsample_stride =
215            crate::data::loader::stride_for(total, capacity as usize) as u32;
216
217        // egui plumbing.
218        let egui_ctx = egui::Context::default();
219        let egui_state = egui_winit::State::new(
220            egui_ctx.clone(),
221            egui::ViewportId::ROOT,
222            window.as_ref(),
223            Some(window.scale_factor() as f32),
224            None,
225            None,
226        );
227        let egui_renderer = egui_wgpu::Renderer::new(&device, format, None, 1, true);
228
229        // Spawn the loader.
230        let mode = if plan.is_demo {
231            LoaderMode::Demo(DemoSource::new(plan.meta.frames.len(), total))
232        } else {
233            let frame_ids = plan.meta.frames.iter().map(|f| f.id).collect();
234            LoaderMode::Real {
235                path: plan.meta.data_path.clone(),
236                frame_ids,
237            }
238        };
239        let loader = LoaderHandle::spawn(mode, plan.meta.bounds, total, capacity as usize);
240
241        Ok(Gfx {
242            window,
243            surface,
244            device,
245            queue,
246            config,
247            depth_view,
248            egui_ctx,
249            egui_state,
250            egui_renderer,
251            points,
252            volume,
253            grid,
254            annotations,
255            camera: OrbitCamera::default(),
256            state,
257            loader,
258            modifiers: ModifiersState::empty(),
259            left_down: false,
260            right_down: false,
261            middle_down: false,
262            last_cursor: None,
263            last_frame: Instant::now(),
264            fps_smooth: 0.0,
265            pending_exit: false,
266            last_view_mode: ViewMode::Points,
267        })
268    }
269}

271impl Gfx {
272    fn resize(&mut self, width: u32, height: u32) {
273        if width == 0 || height == 0 {
274            return;
275        }
276        self.config.width = width;
277        self.config.height = height;
278        self.surface.configure(&self.device, &self.config);
279        self.depth_view = create_depth(&self.device, &self.config);
280    }
281
282    /// Drain the loader channel, bounding work per frame.
283    fn pump_loader(&mut self) {
284        // Cap chunks per frame so a fast loader can't stall the UI with uploads.
285        const MAX_CHUNKS_PER_FRAME: usize = 16;
286        for _ in 0..MAX_CHUNKS_PER_FRAME {
287            match self.loader.rx.try_recv() {
288                Ok(LoadMsg::Chunk { points, .. }) => {
289                    // Volume density: deposit intensity*weight so the density is
290                    // independent of the downsample ratio (weight = 1/p = stride).
291                    for p in &points {
292                        self.grid.deposit(p.pos, p.intensity * p.weight);
293                    }
294                    self.points.append(&self.queue, &points);
295                }
296                Ok(LoadMsg::PeakChunk { points }) => {
297                    // Peaks are display-only enrichment; append to the cloud but do NOT
298                    // add to the volume density (would double-count dense cells).
299                    self.points.append(&self.queue, &points);
300                }
301                Ok(LoadMsg::Progress(p)) => self.state.load_progress = p,
302                Ok(LoadMsg::Stats { i_min, i_max }) => {
303                    // Point-intensity range; only relevant to point mode (volume uses a
304                    // density range derived from the grid).
305                    if self.state.view_mode == ViewMode::Points {
306                        self.state.i_min = i_min;
307                        self.state.i_max = i_max;
308                    }
309                }
310                Ok(LoadMsg::Annotations { lines }) => {
311                    self.annotations.upload(&self.device, &lines);
312                }
313                Ok(LoadMsg::Done { .. }) => {
314                    self.state.load_progress = 1.0;
315                    // Recompute the volume range against the now-complete grid.
316                    if self.state.view_mode == ViewMode::Volume {
317                        let (lo, hi) = self.grid.density_percentiles();
318                        self.state.i_min = lo;
319                        self.state.i_max = hi;
320                    }
321                }
322                Ok(LoadMsg::Error(e)) => {
323                    log::error!("loader error: {e}");
324                    self.state.load_failed = true;
325                    self.state.load_progress = 1.0;
326                }
327                Err(crossbeam_channel::TryRecvError::Empty) => break,
328                Err(crossbeam_channel::TryRecvError::Disconnected) => {
329                    // Normal completion drops the sender after Done (progress==1.0).
330                    // A disconnect before that means the loader thread died (panic).
331                    if self.state.load_progress < 1.0 {
332                        log::error!("loader thread ended before completing the load");
333                        self.state.load_failed = true;
334                        self.state.load_progress = 1.0;
335                    }
336                    break;
337                }
338            }
339        }
340        self.state.resident_points = self.points.resident();
341    }
342
343    fn render(&mut self) {
344        self.pump_loader();
345
346        // FPS (exponential smoothing).
347        let now = Instant::now();
348        let dt = now.duration_since(self.last_frame).as_secs_f32().max(1e-4);
349        self.last_frame = now;
350        let inst = 1.0 / dt;
351        self.fps_smooth = if self.fps_smooth == 0.0 {
352            inst
353        } else {
354            self.fps_smooth * 0.9 + inst * 0.1
355        };
356        self.state.fps = self.fps_smooth;
357
358        // Acquire the surface BEFORE any egui texture upload, so a surface error
359        // returns without leaking uploaded-but-never-freed egui textures (texture
360        // `set` and `free` are then always paired within one successful frame).
361        let frame = match self.surface.get_current_texture() {
362            Ok(f) => f,
363            Err(wgpu::SurfaceError::Lost) | Err(wgpu::SurfaceError::Outdated) => {
364                self.surface.configure(&self.device, &self.config);
365                return;
366            }
367            Err(wgpu::SurfaceError::Timeout) => return,
368            Err(wgpu::SurfaceError::OutOfMemory) => {
369                log::error!("surface out of memory; exiting");
370                self.pending_exit = true;
371                return;
372            }
373        };
374        let view = frame
375            .texture
376            .create_view(&wgpu::TextureViewDescriptor::default());
377
378        // egui frame.
379        let raw_input = self.egui_state.take_egui_input(self.window.as_ref());
380        let full_output = self.egui_ctx.run(raw_input, |ctx| {
381            ui::build(ctx, &mut self.state, &mut self.camera);
382        });
383        self.egui_state
384            .handle_platform_output(self.window.as_ref(), full_output.platform_output);
385        let tris = self
386            .egui_ctx
387            .tessellate(full_output.shapes, full_output.pixels_per_point);
388        for (id, delta) in &full_output.textures_delta.set {
389            self.egui_renderer
390                .update_texture(&self.device, &self.queue, *id, delta);
391        }
392
393        // Uniforms — computed AFTER egui so they reflect this frame's UI/camera changes
394        // (otherwise a Points->Volume switch renders one frame with a stale matrix and
395        // an un-uploaded texture).
396        let aspect = self.config.width as f32 / self.config.height.max(1) as f32;
397        let cam = self.camera.to_uniform(
398            aspect,
399            [self.config.width as f32, self.config.height as f32],
400        );
401        let params = self.state.params();
402        self.points.update_camera(&self.queue, &cam);
403        self.points.update_params(&self.queue, &params);
404        self.annotations.update_camera(&self.queue, &cam);
405        self.annotations
406            .update_filter(&self.queue, params.filter_min, params.filter_max);
407
408        // Volume mode: update the raycaster uniform (incl. the density scale) and
409        // (re)upload the grid if it grew.
410        if self.state.view_mode == ViewMode::Volume {
411            // On entering volume mode, auto-range the transfer fn to the density
412            // distribution (density sums differ in range from per-point intensity).
413            if self.last_view_mode != ViewMode::Volume {
414                let (lo, hi) = self.grid.density_percentiles();
415                self.state.i_min = lo;
416                self.state.i_max = hi;
417            }
418            let inv_vp = self.camera.inv_view_proj(aspect);
419            let mut vu = self.state.volume_uniform(inv_vp);
420            vu.density_scale = self.grid.density_scale();
421            self.volume.update_uniform(&self.queue, &vu);
422            if self.grid.dirty() {
423                self.volume.upload(&self.queue, &self.grid.to_f16_scaled());
424                self.grid.clear_dirty();
425            }
426        }
427        self.last_view_mode = self.state.view_mode;
428
429        let mut encoder = self
430            .device
431            .create_command_encoder(&wgpu::CommandEncoderDescriptor {
432                label: Some("frame-encoder"),
433            });
434
435        let screen_desc = egui_wgpu::ScreenDescriptor {
436            size_in_pixels: [self.config.width, self.config.height],
437            pixels_per_point: full_output.pixels_per_point,
438        };
439        let user_bufs =
440            self.egui_renderer
441                .update_buffers(&self.device, &self.queue, &mut encoder, &tris, &screen_desc);
442
443        // Cull + compact visible points (compute) before the scene pass (points mode only).
444        if self.state.view_mode == ViewMode::Points {
445            self.points.prepare(&self.queue, &mut encoder);
446        }
447
448        // 3D pass.
449        {
450            let mut rpass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
451                label: Some("scene-pass"),
452                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
453                    view: &view,
454                    resolve_target: None,
455                    ops: wgpu::Operations {
456                        load: wgpu::LoadOp::Clear(wgpu::Color {
457                            r: 0.02,
458                            g: 0.02,
459                            b: 0.035,
460                            a: 1.0,
461                        }),
462                        store: wgpu::StoreOp::Store,
463                    },
464                })],
465                depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment {
466                    view: &self.depth_view,
467                    depth_ops: Some(wgpu::Operations {
468                        load: wgpu::LoadOp::Clear(1.0),
469                        store: wgpu::StoreOp::Store,
470                    }),
471                    stencil_ops: None,
472                }),
473                timestamp_writes: None,
474                occlusion_query_set: None,
475            });
476            match self.state.view_mode {
477                ViewMode::Points => self.points.render(&mut rpass, self.state.point_mode),
478                ViewMode::Volume => self.volume.render(&mut rpass),
479            }
480            if self.state.show_annotations {
481                self.annotations.render(&mut rpass);
482            }
483        }
484
485        // egui pass (load existing color, no depth).
486        {
487            let mut rpass = encoder
488                .begin_render_pass(&wgpu::RenderPassDescriptor {
489                    label: Some("egui-pass"),
490                    color_attachments: &[Some(wgpu::RenderPassColorAttachment {
491                        view: &view,
492                        resolve_target: None,
493                        ops: wgpu::Operations {
494                            load: wgpu::LoadOp::Load,
495                            store: wgpu::StoreOp::Store,
496                        },
497                    })],
498                    depth_stencil_attachment: None,
499                    timestamp_writes: None,
500                    occlusion_query_set: None,
501                })
502                .forget_lifetime();
503            self.egui_renderer.render(&mut rpass, &tris, &screen_desc);
504        }
505
506        self.queue
507            .submit(user_bufs.into_iter().chain(std::iter::once(encoder.finish())));
508        frame.present();
509
510        for id in &full_output.textures_delta.free {
511            self.egui_renderer.free_texture(id);
512        }
513    }
514
515    fn handle_cursor(&mut self, pos: PhysicalPosition<f64>) {
516        if let Some(last) = self.last_cursor {
517            let dx = (pos.x - last.x) as f32;
518            let dy = (pos.y - last.y) as f32;
519            let shift = self.modifiers.shift_key();
520            if self.middle_down || self.right_down || (self.left_down && shift) {
521                self.camera.pan(dx, dy);
522            } else if self.left_down {
523                self.camera.orbit(dx, dy);
524            }
525        }
526        self.last_cursor = Some(pos);
527    }

530impl ApplicationHandler for App {
531    fn resumed(&mut self, event_loop: &ActiveEventLoop) {
532        if self.gfx.is_some() {
533            return;
534        }
535        match self.init_gfx(event_loop) {
536            Ok(gfx) => self.gfx = Some(gfx),
537            Err(e) => {
538                log::error!("failed to initialize graphics: {e:#}");
539                event_loop.exit();
540            }
541        }
542    }
543
544    fn window_event(
545        &mut self,
546        event_loop: &ActiveEventLoop,
547        _id: WindowId,
548        event: WindowEvent,
549    ) {
550        let Some(gfx) = self.gfx.as_mut() else {
551            return;
552        };
553
554        // egui gets first dibs.
555        let resp = gfx.egui_state.on_window_event(gfx.window.as_ref(), &event);
556        if resp.repaint {
557            gfx.window.request_redraw();
558        }
559        let consumed = resp.consumed;
560
561        match event {
562            WindowEvent::CloseRequested => event_loop.exit(),
563            WindowEvent::Resized(size) => gfx.resize(size.width, size.height),
564            WindowEvent::ModifiersChanged(m) => gfx.modifiers = m.state(),
565            WindowEvent::RedrawRequested => gfx.render(),
566            // Handle button state BEFORE the consumed-guard: a release that egui
567            // consumes (drag ended over a panel) must still clear the button, or the
568            // camera keeps orbiting on the next move. Presses only start a drag when
569            // egui did not consume them (so dragging a slider doesn't orbit).
570            WindowEvent::MouseInput { state, button, .. } => {
571                let down = state == ElementState::Pressed;
572                let set = if down { !consumed } else { false };
573                match button {
574                    MouseButton::Left => gfx.left_down = set,
575                    MouseButton::Right => gfx.right_down = set,
576                    MouseButton::Middle => gfx.middle_down = set,
577                    _ => {}
578                }
579                if !down {
580                    gfx.last_cursor = None;
581                }
582            }
583            _ if consumed => {
584                // egui handled it; don't drive the camera.
585                gfx.last_cursor = None;
586            }
587            WindowEvent::CursorMoved { position, .. } => gfx.handle_cursor(position),
588            WindowEvent::CursorLeft { .. } => gfx.last_cursor = None,
589            WindowEvent::MouseWheel { delta, .. } => {
590                let s = match delta {
591                    MouseScrollDelta::LineDelta(_, y) => y,
592                    MouseScrollDelta::PixelDelta(p) => (p.y as f32) / 50.0,
593                };
594                gfx.camera.zoom(s);
595            }
596            _ => {}
597        }
598    }
599
600    fn about_to_wait(&mut self, event_loop: &ActiveEventLoop) {
601        if let Some(gfx) = self.gfx.as_ref() {
602            if gfx.pending_exit {
603                event_loop.exit();
604                return;
605            }
606            // Drive continuous redraws so streaming data animates in.
607            gfx.window.request_redraw();
608        }
609    }
