1//! Headless one-frame renderer: load a run, render the point cloud or volume to an
2//! offscreen texture, and write a PNG. Works without a display (used for screenshots on
3//! headless machines and to preview the volume raycaster).
4
5use std::path::Path;
6
7use anyhow::{Context, Result};
8
9use crate::app::Plan;
10use crate::camera::OrbitCamera;
11use crate::data::demo::DemoSource;
12use crate::data::loader::{LoadMsg, LoaderHandle, LoaderMode};
13use crate::data::point::GpuPoint;
14use crate::render::colormap::COLORMAP_NAMES;
15use crate::render::annotation::AnnotationRenderer;
16use crate::render::point_cloud::PointCloudRenderer;
17use crate::render::volume::{VolumeGrid, VolumeRenderer, VOLUME_DIMS};
18use crate::state::{AppState, ViewMode, VolStyle};
19
20const COLOR_FORMAT: wgpu::TextureFormat = wgpu::TextureFormat::Rgba8UnormSrgb;
21const DEPTH_FORMAT: wgpu::TextureFormat = wgpu::TextureFormat::Depth32Float;
22
23/// Which MS level to render.
24#[derive(Clone, Copy, Debug, PartialEq, Eq)]
25pub enum MsFilter {
26    All,
27    Ms1,
28    Ms2,
29}
30
31pub struct Options {
32    pub width: u32,
33    pub height: u32,
34    pub volume: bool,
35    pub mip: bool,
36    pub ms: MsFilter,
37    pub annotations: bool,
38    /// Optional transfer-function overrides (else taken from data percentiles).
39    pub i_min: Option<f32>,
40    pub i_max: Option<f32>,
41    pub exposure: Option<f32>,
42}
43
44/// Render one frame of `plan`'s run and write a PNG to `out`.
45pub fn render_png(plan: Plan, out: &Path, opts: &Options) -> Result<()> {
46    let instance = wgpu::Instance::new(wgpu::InstanceDescriptor {
47        backends: wgpu::Backends::PRIMARY | wgpu::Backends::GL,
48        ..Default::default()
49    });
50    let adapter = pollster::block_on(instance.request_adapter(&wgpu::RequestAdapterOptions {
51        power_preference: wgpu::PowerPreference::HighPerformance,
52        force_fallback_adapter: false,
53        compatible_surface: None,
54    }))
55    .ok_or_else(|| anyhow::anyhow!("no GPU adapter available for offscreen render"))?;
56    log::info!("offscreen adapter: {:?}", adapter.get_info());
57
58    let limits = adapter.limits();
59    let (device, queue) = pollster::block_on(adapter.request_device(
60        &wgpu::DeviceDescriptor {
61            label: Some("offscreen-device"),
62            required_features: wgpu::Features::empty(),
63            required_limits: limits.clone(),
64            memory_hints: wgpu::MemoryHints::Performance,
65        },
66        None,
67    ))?;
68
69    // Validate output dimensions: nonzero (else invalid texture / infinite aspect) and
70    // within the device's max texture size (also bounds the row/size arithmetic below).
71    anyhow::ensure!(
72        opts.width > 0 && opts.height > 0,
73        "render dimensions must be > 0 (got {}x{})",
74        opts.width,
75        opts.height
76    );
77    let max_dim = limits.max_texture_dimension_2d;
78    anyhow::ensure!(
79        opts.width <= max_dim && opts.height <= max_dim,
80        "render dimensions {}x{} exceed the device max texture size {max_dim}",
81        opts.width,
82        opts.height
83    );
84
85    let flags = adapter.get_downlevel_capabilities().flags;
86    let supports_compaction = flags.contains(wgpu::DownlevelFlags::COMPUTE_SHADERS)
87        && flags.contains(wgpu::DownlevelFlags::INDIRECT_EXECUTION);
88
89    let bytes_per_point = std::mem::size_of::<GpuPoint>() as u64;
90    let total = plan.meta.total_points_estimate;
91    let mut cap = plan
92        .budget
93        .min((limits.max_buffer_size / bytes_per_point) as usize)
94        .min((total as usize).max(1));
95    if supports_compaction {
96        cap = cap.min((limits.max_storage_buffer_binding_size as u64 / bytes_per_point) as usize);
97    }
98    let capacity = cap.max(1) as u32;
99
100    let mut points =
101        PointCloudRenderer::new(&device, &queue, COLOR_FORMAT, DEPTH_FORMAT, capacity, supports_compaction);
102    let volume = VolumeRenderer::new(&device, &queue, COLOR_FORMAT, DEPTH_FORMAT, VOLUME_DIMS);
103    let mut grid = VolumeGrid::new(VOLUME_DIMS);
104    let mut annotations = AnnotationRenderer::new(&device, COLOR_FORMAT, DEPTH_FORMAT);
105
106    let bounds = plan.meta.bounds;
107    let mut state = AppState::new(bounds, total, COLORMAP_NAMES.len() as u32);
108    state.capacity = capacity;
109    state.downsample_stride =
110        crate::data::loader::stride_for(total, capacity as usize) as u32;
111    state.view_mode = if opts.volume { ViewMode::Volume } else { ViewMode::Points };
112    state.vol_style = if opts.mip { VolStyle::Mip } else { VolStyle::Composite };
113    // MS selection is applied at upload time via `opts.ms`; let the shader pass everything
114    // that was uploaded (don't double-filter with the interactive MS1-only default).
115    state.show_ms2 = true;
116
117    // Load the whole run synchronously by draining the streaming loader.
118    let mode = if plan.is_demo {
119        LoaderMode::Demo(DemoSource::new(plan.meta.frames.len(), total))
120    } else {
121        LoaderMode::Real {
122            path: plan.meta.data_path.clone(),
123            frame_ids: plan.meta.frames.iter().map(|f| f.id).collect(),
124        }
125    };
126    let loader = LoaderHandle::spawn(mode, bounds, total, capacity as usize, None);
127    let keep = |p: &GpuPoint| match opts.ms {
128        MsFilter::All => true,
129        MsFilter::Ms1 => p.flags & GpuPoint::MS2_FLAG == 0,
130        MsFilter::Ms2 => p.flags & GpuPoint::MS2_FLAG != 0,
131    };
132    // Reservoir of RETAINED intensities so the auto transfer range matches what is
133    // actually rendered (the loader's Stats span both MS levels).
134    let mut reservoir: Vec<f32> = Vec::new();
135    let sample_every = (capacity as usize / 200_000).max(1) as u64;
136    let mut seen: u64 = 0;
137    let mut done = false;
138    loop {
139        match loader.rx.recv() {
140            Ok(LoadMsg::Chunk { points: pts, .. }) => {
141                // Filter to the requested MS level at the source so both the volume grid
142                // and the point buffer render only that level.
143                let pts: Vec<GpuPoint> = if opts.ms == MsFilter::All {
144                    pts
145                } else {
146                    pts.into_iter().filter(&keep).collect()
147                };
148                for p in &pts {
149                    // Deposit intensity*weight so density is downsample-invariant.
150                    grid.deposit(p.pos, p.intensity * p.weight);
151                    if seen % sample_every == 0 && reservoir.len() < 200_000 {
152                        reservoir.push(p.intensity);
153                    }
154                    seen += 1;
155                }
156                points.append(&queue, &pts);
157            }
158            Ok(LoadMsg::PeakChunk { points: pts }) => {
159                // Display-only peaks: append to the cloud, never to the volume density.
160                let pts: Vec<GpuPoint> = if opts.ms == MsFilter::All {
161                    pts
162                } else {
163                    pts.into_iter().filter(&keep).collect()
164                };
165                points.append(&queue, &pts);
166            }
167            Ok(LoadMsg::Stats { .. }) => {} // recomputed below from retained points
168            Ok(LoadMsg::Histograms { .. }) => {} // UI-only; headless ignores
169            Ok(LoadMsg::Annotations { lines, .. }) => {
170                if opts.annotations {
171                    annotations.upload(&device, &lines);
172                }
173            }
174            Ok(LoadMsg::Done { .. }) => {
175                done = true;
176                break;
177            }
178            Ok(LoadMsg::Error(e)) => anyhow::bail!("loader error: {e}"),
179            Ok(LoadMsg::Progress(_)) => {}
180            Err(_) => break,
181        }
182    }
183    // A disconnect before Done means the loader thread died (e.g. panicked) — don't
184    // silently write a partial/empty image.
185    anyhow::ensure!(done, "loader thread ended before completing the load");
186
187    // Default transfer range. Volume density sums live in a different range than
188    // per-point intensity, so take it from the voxel density distribution; the point
189    // cloud uses the retained-intensity percentiles.
190    if opts.volume {
191        let (lo, hi) = grid.density_percentiles();
192        state.auto_transfer_volume(lo, hi);
193    } else if !reservoir.is_empty() {
194        reservoir.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
195        let pct = |q: f32| reservoir[(((reservoir.len() - 1) as f32) * q) as usize];
196        // Feed the same auto-exposure heuristic the interactive viewer uses, so the
197        // headless render is legible (median mid-LUT, additive cloud not blown out).
198        state.i_p1 = pct(0.01).max(1.0);
199        state.i_med = pct(0.50).max(state.i_p1);
200        state.i_p99 = pct(0.99).max(state.i_med * 1.0001);
201        state.auto_transfer_points();
202    }
203    // Apply transfer-function overrides (e.g. raise i_min to threshold out noise).
204    if let Some(v) = opts.i_min {
205        state.i_min = v;
206    }
207    if let Some(v) = opts.i_max {
208        state.i_max = v;
209    }
210    if let Some(v) = opts.exposure {
211        state.exposure = v;
212    }
213    log::info!(
214        "loaded {} points (i_min={:.0} i_max={:.0} exposure={})",
215        points.resident(),
216        state.i_min,
217        state.i_max,
218        state.exposure
219    );
220
221    // Camera + uniforms.
222    let camera = OrbitCamera::default();
223    let aspect = opts.width as f32 / opts.height as f32;
224    let cam_uniform = camera.to_uniform(aspect, [opts.width as f32, opts.height as f32]);
225    let params = state.params();
226    points.update_camera(&queue, &cam_uniform);
227    points.update_params(&queue, &params);
228    annotations.update_camera(&queue, &cam_uniform);
229    annotations.update_filter(&queue, params.filter_min, params.filter_max, params.focus);
230    if opts.volume {
231        let mut vu = state.volume_uniform(camera.inv_view_proj(aspect));
232        vu.density_scale = grid.density_scale();
233        volume.update_uniform(&queue, &vu);
234        volume.upload(&queue, grid.to_f16_scaled());
235    }
236
237    // Offscreen targets.
238    let color = device.create_texture(&wgpu::TextureDescriptor {
239        label: Some("offscreen-color"),
240        size: wgpu::Extent3d {
241            width: opts.width,
242            height: opts.height,
243            depth_or_array_layers: 1,
244        },
245        mip_level_count: 1,
246        sample_count: 1,
247        dimension: wgpu::TextureDimension::D2,
248        format: COLOR_FORMAT,
249        usage: wgpu::TextureUsages::RENDER_ATTACHMENT | wgpu::TextureUsages::COPY_SRC,
250        view_formats: &[],
251    });
252    let depth = device.create_texture(&wgpu::TextureDescriptor {
253        label: Some("offscreen-depth"),
254        size: wgpu::Extent3d {
255            width: opts.width,
256            height: opts.height,
257            depth_or_array_layers: 1,
258        },
259        mip_level_count: 1,
260        sample_count: 1,
261        dimension: wgpu::TextureDimension::D2,
262        format: DEPTH_FORMAT,
263        usage: wgpu::TextureUsages::RENDER_ATTACHMENT,
264        view_formats: &[],
265    });
266    let color_view = color.create_view(&Default::default());
267    let depth_view = depth.create_view(&Default::default());
268
269    // Padded readback buffer (bytes_per_row must be 256-aligned for texture->buffer).
270    let unpadded_bpr = opts.width * 4;
271    let padded_bpr = unpadded_bpr.div_ceil(256) * 256;
272    let readback = device.create_buffer(&wgpu::BufferDescriptor {
273        label: Some("readback"),
274        size: (padded_bpr * opts.height) as u64,
275        usage: wgpu::BufferUsages::MAP_READ | wgpu::BufferUsages::COPY_DST,
276        mapped_at_creation: false,
277    });
278
279    let mut encoder =
280        device.create_command_encoder(&wgpu::CommandEncoderDescriptor { label: Some("offscreen") });
281    if !opts.volume {
282        points.prepare(&queue, &mut encoder);
283    }
284    {
285        let mut rpass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
286            label: Some("offscreen-scene"),
287            color_attachments: &[Some(wgpu::RenderPassColorAttachment {
288                view: &color_view,
289                resolve_target: None,
290                ops: wgpu::Operations {
291                    load: wgpu::LoadOp::Clear(wgpu::Color {
292                        r: 0.02,
293                        g: 0.02,
294                        b: 0.035,
295                        a: 1.0,
296                    }),
297                    store: wgpu::StoreOp::Store,
298                },
299            })],
300            depth_stencil_attachment: Some(wgpu::RenderPassDepthStencilAttachment {
301                view: &depth_view,
302                depth_ops: Some(wgpu::Operations {
303                    load: wgpu::LoadOp::Clear(1.0),
304                    store: wgpu::StoreOp::Store,
305                }),
306                stencil_ops: None,
307            }),
308            timestamp_writes: None,
309            occlusion_query_set: None,
310        });
311        if opts.volume {
312            volume.render(&mut rpass);
313        } else {
314            points.render(&mut rpass, state.point_mode);
315        }
316        if opts.annotations {
317            annotations.render(&mut rpass);
318        }
319    }
320    encoder.copy_texture_to_buffer(
321        wgpu::ImageCopyTexture {
322            texture: &color,
323            mip_level: 0,
324            origin: wgpu::Origin3d::ZERO,
325            aspect: wgpu::TextureAspect::All,
326        },
327        wgpu::ImageCopyBuffer {
328            buffer: &readback,
329            layout: wgpu::ImageDataLayout {
330                offset: 0,
331                bytes_per_row: Some(padded_bpr),
332                rows_per_image: Some(opts.height),
333            },
334        },
335        wgpu::Extent3d {
336            width: opts.width,
337            height: opts.height,
338            depth_or_array_layers: 1,
339        },
340    );
341    queue.submit(std::iter::once(encoder.finish()));
342
343    // Map and read back.
344    let slice = readback.slice(..);
345    let (tx, rx) = std::sync::mpsc::channel();
346    slice.map_async(wgpu::MapMode::Read, move |r| {
347        let _ = tx.send(r);
348    });
349    device.poll(wgpu::Maintain::Wait);
350    rx.recv()
351        .context("map_async channel closed")?
352        .context("buffer map failed")?;
353
354    let data = slice.get_mapped_range();
355    let mut rgba = Vec::with_capacity((opts.width * opts.height * 4) as usize);
356    for row in 0..opts.height {
357        let start = (row * padded_bpr) as usize;
358        rgba.extend_from_slice(&data[start..start + unpadded_bpr as usize]);
359    }
360    drop(data);
361    readback.unmap();
362
363    // Encode PNG (Rgba8UnormSrgb bytes are already sRGB).
364    let file = std::fs::File::create(out).with_context(|| format!("creating {}", out.display()))?;
365    let w = std::io::BufWriter::new(file);
366    let mut encoder = png::Encoder::new(w, opts.width, opts.height);
367    encoder.set_color(png::ColorType::Rgba);
368    encoder.set_depth(png::BitDepth::Eight);
369    encoder.set_source_srgb(png::SrgbRenderingIntent::Perceptual);
370    let mut writer = encoder.write_header().context("png header")?;
371    writer.write_image_data(&rgba).context("png write")?;
372
373    Ok(())
374}
