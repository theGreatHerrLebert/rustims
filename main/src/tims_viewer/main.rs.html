1//! tims-viewer — native GPU viewer for Bruker timsTOF raw data.
2//!
3//! Usage:
4//!   tims-viewer <path/to/run.d>      open a real Bruker .d dataset
5//!   tims-viewer DEMO                 synthetic point cloud (no Bruker data needed)
6//!   tims-viewer DEMO --budget 25000000

16use anyhow::Result;

18use winit::event_loop::{ControlFlow, EventLoop};

20use app::{App, Plan};
21use data::meta::MetaIndex;

23/// Command-line arguments.
24#[derive(Parser, Debug)]
25#[command(name = "tims-viewer", about = "Native GPU viewer for Bruker timsTOF raw data")]
26struct Args {
27    /// Path to a Bruker `.d` folder, or the literal `DEMO` for a synthetic cloud.
28    input: String,
29    /// On-GPU point budget (resident capacity). Larger = more detail, more VRAM.
30    #[arg(long)]
31    budget: Option<usize>,
32    /// Number of synthetic frames for DEMO.
33    #[arg(long, default_value_t = 800)]
34    demo_frames: usize,
35    /// Total synthetic points for DEMO (before downsampling to the budget).
36    #[arg(long, default_value_t = 30_000_000)]
37    demo_points: u64,
38    /// Headless: render one frame to this PNG path and exit (no window needed).
39    #[arg(long)]
40    render_png: Option<String>,
41    /// With --render-png: render the volume raycaster instead of the point cloud.
42    #[arg(long)]
43    volume: bool,
44    /// With --render-png --volume: use maximum-intensity projection instead of composite.
45    #[arg(long)]
46    mip: bool,
47    /// MS level to render: all | ms1 | ms2.
48    #[arg(long, default_value = "all")]
49    ms: String,
50    /// With --render-png: overlay DDA precursor / DIA isolation-window annotations.
51    #[arg(long)]
52    annotations: bool,
53    /// Output image width / height for --render-png.
54    #[arg(long, default_value_t = 1280)]
55    width: u32,
56    #[arg(long, default_value_t = 800)]
57    height: u32,
58    /// Transfer-function overrides for --render-png (default from data percentiles).
59    #[arg(long)]
60    i_min: Option<f32>,
61    #[arg(long)]
62    i_max: Option<f32>,
63    #[arg(long)]
64    exposure: Option<f32>,
65}
66
67fn main() -> Result<()> {
68    env_logger::Builder::from_env(env_logger::Env::default().default_filter_or("info")).init();
69    let args = Args::parse();
70
71    let (meta, is_demo) = if args.input.eq_ignore_ascii_case("DEMO") {
72        log::info!(
73            "DEMO mode: {} frames, {} synthetic points",
74            args.demo_frames,
75            args.demo_points
76        );
77        (MetaIndex::demo(args.demo_frames, args.demo_points), true)
78    } else {
79        log::info!("loading metadata from {}", args.input);
80        let meta = MetaIndex::load(&args.input)?;
81        log::info!(
82            "{} frames, ~{} points, RT [{:.1}, {:.1}] s, m/z [{:.1}, {:.1}], 1/K0 [{:.3}, {:.3}]",
83            meta.frames.len(),
84            meta.total_points_estimate,
85            meta.bounds.rt.min,
86            meta.bounds.rt.max,
87            meta.bounds.mz.min,
88            meta.bounds.mz.max,
89            meta.bounds.im.min,
90            meta.bounds.im.max,
91        );
92        (meta, false)
93    };
94
95    let plan = Plan::new(meta, is_demo, args.budget);
96
97    // Headless one-frame render mode (no window).
98    if let Some(path) = args.render_png {
99        let ms = match args.ms.to_ascii_lowercase().as_str() {
100            "ms1" | "1" => offscreen::MsFilter::Ms1,
101            "ms2" | "2" => offscreen::MsFilter::Ms2,
102            "all" => offscreen::MsFilter::All,
103            other => anyhow::bail!("--ms must be all|ms1|ms2, got '{other}'"),
104        };
105        let opts = offscreen::Options {
106            width: args.width,
107            height: args.height,
108            volume: args.volume,
109            mip: args.mip,
110            ms,
111            annotations: args.annotations,
112            i_min: args.i_min,
113            i_max: args.i_max,
114            exposure: args.exposure,
115        };
116        offscreen::render_png(plan, std::path::Path::new(&path), &opts)?;
117        log::info!("wrote {}", path);
118        return Ok(());
119    }
120
121    let event_loop = EventLoop::new()?;
122    event_loop.set_control_flow(ControlFlow::Poll);
123    let mut app = App::new(plan);
124    event_loop.run_app(&mut app)?;
125    Ok(())
126}
