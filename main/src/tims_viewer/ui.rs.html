8/// Build the left control panel; mutates state and camera in place.
9pub fn build(ctx: &egui::Context, state: &mut AppState, camera: &mut OrbitCamera) {
10    egui::SidePanel::left("controls")
11        .resizable(true)
12        .default_width(280.0)
13        .show(ctx, |ui| {
14            ui.heading("tims-viewer");
15            ui.separator();
16
17            // ---- Readouts ----
18            ui.label(format!("FPS: {:.0}", state.fps));
19            ui.label(format!(
20                "Points: {} / {} (cap)",
21                fmt_count(state.resident_points as u64),
22                fmt_count(state.capacity as u64)
23            ));
24            ui.label(format!(
25                "Run estimate: {} (×{} downsample)",
26                fmt_count(state.total_estimate),
27                state.downsample_stride
28            ));
29            if state.load_failed {
30                ui.colored_label(egui::Color32::LIGHT_RED, "Load failed (see log)");
31            } else if state.load_progress < 1.0 {
32                ui.add(egui::ProgressBar::new(state.load_progress).show_percentage());
33            } else {
34                ui.label("Load complete");
35            }
36            ui.separator();
37
38            // ---- Render mode ----
39            ui.label("Render mode");
40            ui.horizontal(|ui| {
41                ui.radio_value(&mut state.view_mode, ViewMode::Points, "Points");
42                ui.radio_value(&mut state.view_mode, ViewMode::Volume, "Volume");
43            });
44            match state.view_mode {
45                ViewMode::Points => {
46                    ui.horizontal(|ui| {
47                        ui.radio_value(
48                            &mut state.point_mode,
49                            PointMode::AdditiveDensity,
50                            "Additive",
51                        );
52                        ui.radio_value(
53                            &mut state.point_mode,
54                            PointMode::StructuralOpaque,
55                            "Opaque",
56                        );
57                    });
58                }
59                ViewMode::Volume => {
60                    ui.horizontal(|ui| {
61                        ui.radio_value(&mut state.vol_style, VolStyle::Composite, "Composite");
62                        ui.radio_value(&mut state.vol_style, VolStyle::Mip, "MIP");
63                    });
64                    ui.add(
65                        egui::Slider::new(&mut state.vol_steps, 32..=1024).text("ray steps"),
66                    );
67                }
68            }
69            ui.separator();
70
71            // ---- Intensity transfer function ----
72            ui.label("Intensity transfer");
73            ui.horizontal(|ui| {
74                ui.radio_value(&mut state.transfer, TransferMode::Linear, "Lin");
75                ui.radio_value(&mut state.transfer, TransferMode::Sqrt, "Sqrt");
76                ui.radio_value(&mut state.transfer, TransferMode::Log, "Log");
77            });
78            ui.add(
79                egui::Slider::new(&mut state.i_min, 1.0..=1e6)
80                    .logarithmic(true)
81                    .text("i_min"),
82            );
83            ui.add(
84                egui::Slider::new(&mut state.i_max, 10.0..=1e8)
85                    .logarithmic(true)
86                    .text("i_max"),
87            );
88            if state.i_max <= state.i_min {
89                state.i_max = state.i_min * 1.0001;
90            }
91            ui.add(
92                egui::Slider::new(&mut state.exposure, 0.001..=5.0)
93                    .logarithmic(true)
94                    .text("exposure"),
95            );
96
97            egui::ComboBox::from_label("colormap")
98                .selected_text(COLORMAP_NAMES[state.colormap_id as usize])
99                .show_ui(ui, |ui| {
100                    for (i, name) in COLORMAP_NAMES.iter().enumerate() {
101                        ui.selectable_value(&mut state.colormap_id, i as u32, *name);
102                    }
103                });
104            ui.separator();
105
106            // ---- Appearance ----
107            ui.add(egui::Slider::new(&mut state.point_size, 0.5..=12.0).text("point size"));
108            ui.add(egui::Slider::new(&mut state.opacity, 0.02..=1.0).text("opacity"));
109            ui.separator();
110
111            // ---- MS filter ----
112            ui.horizontal(|ui| {
113                ui.checkbox(&mut state.show_ms1, "MS1");
114                ui.checkbox(&mut state.show_ms2, "MS2");
115            });
116            ui.checkbox(&mut state.show_annotations, "Annotations (precursors / DIA windows)");
117            ui.separator();
118
119            // ---- Axis windows ----
120            ui.label("Windows (real units)");
121            axis_window(ui, "RT (s)", &mut state.rt_window, state.bounds.rt.min, state.bounds.rt.max);
122            axis_window(ui, "m/z (Th)", &mut state.mz_window, state.bounds.mz.min, state.bounds.mz.max);
123            axis_window(ui, "1/K0", &mut state.im_window, state.bounds.im.min, state.bounds.im.max);
124            if ui.button("Reset windows").clicked() {
125                state.reset_windows();
126            }
127            ui.separator();
128
129            // ---- Camera ----
130            ui.label("Camera");
131            ui.horizontal(|ui| {
132                if ui.button("Reset").clicked() {
133                    camera.reset();
134                }
135                let mut ortho = camera.projection == Projection::Orthographic;
136                if ui.checkbox(&mut ortho, "Ortho").changed() {
137                    camera.projection = if ortho {
138                        Projection::Orthographic
139                    } else {
140                        Projection::Perspective
141                    };
142                }
143            });
144            ui.horizontal(|ui| {
145                if ui.button("m/z view").clicked() {
146                    camera.set_axis_view(AxisView::Mz);
147                }
148                if ui.button("mob view").clicked() {
149                    camera.set_axis_view(AxisView::Mobility);
150                }
151                if ui.button("RT view").clicked() {
152                    camera.set_axis_view(AxisView::Rt);
153                }
154            });
155        });
156}

158fn axis_window(
159    ui: &mut egui::Ui,
160    label: &str,
161    win: &mut crate::state::Window,
162    lo: f64,
163    hi: f64,
164) {
165    ui.label(label);
166    ui.horizontal(|ui| {
167        ui.add(
168            egui::DragValue::new(&mut win.min)
169                .speed((hi - lo) / 500.0)
170                .range(lo..=hi),
171        );
172        ui.label("–");
173        ui.add(
174            egui::DragValue::new(&mut win.max)
175                .speed((hi - lo) / 500.0)
176                .range(lo..=hi),
177        );
178    });
179    if win.min > win.max {
180        win.max = win.min;
181    }
182}

184fn fmt_count(n: u64) -> String {
185    if n >= 1_000_000 {
186        format!("{:.1}M", n as f64 / 1e6)
187    } else if n >= 1_000 {
188        format!("{:.1}k", n as f64 / 1e3)
189    } else {
190        n.to_string()
191    }
192}
