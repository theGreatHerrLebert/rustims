1//! Grid-accelerated DBSCAN over the normalized point cloud, for coloring points by cluster
2//! identity (the same analysis the proteolizard/imspy-vis Voila tool did, in the viewer).
3//!
4//! Points live in the normalized cube `[-1, 1]^3` (x=m/z, y=1/K0, z=RT), so a single `eps`
5//! in cube units is a sensible neighborhood radius. A uniform spatial hash of cell size `eps`
6//! makes each neighborhood query touch only the 27 adjacent cells, so the whole pass is about
7//! O(n) for the moderate point counts the viewer clusters (it is gated to a cap by the caller).
8
9use rustc_hash::FxHashMap;
10use std::collections::VecDeque;
11
12/// Label assigned to noise / unclustered points.
13pub const NOISE: i32 = -1;
14
15/// Run DBSCAN on `points` (normalized cube positions). Returns a per-point label
16/// (`NOISE` for noise, otherwise a 0-based contiguous cluster id) and the cluster count.
17pub fn dbscan(points: &[[f32; 3]], eps: f32, min_pts: usize) -> (Vec<i32>, usize) {
18    dbscan_with_progress(points, eps, min_pts, |_| {})
19}
20
21/// Like [`dbscan`], but invokes `progress(visited_so_far)` roughly every 1% of points (and once at
22/// the end with `points.len()`). The visited count climbs monotonically — even inside one big
23/// cluster's flood-fill — so it's a smooth progress proxy for a long run.
24pub fn dbscan_with_progress(
25    points: &[[f32; 3]],
26    eps: f32,
27    min_pts: usize,
28    mut progress: impl FnMut(usize),
29) -> (Vec<i32>, usize) {
30    let n = points.len();
31    if n == 0 || eps <= 0.0 {
32        return (vec![NOISE; n], 0);
33    }
34    let inv = 1.0 / eps;
35    let cell = |p: &[f32; 3]| -> (i32, i32, i32) {
36        (
37            (p[0] * inv).floor() as i32,
38            (p[1] * inv).floor() as i32,
39            (p[2] * inv).floor() as i32,
40        )
41    };
42    // Spatial hash: cell -> point indices. FxHashMap (not SipHash) — the grid is hashed n times on
43    // build and 27n times in neighbour queries, so the hash is a hot path, especially in wasm.
44    let mut grid: FxHashMap<(i32, i32, i32), Vec<u32>> =
45        FxHashMap::with_capacity_and_hasher(n / 4 + 1, Default::default());
46    for (i, p) in points.iter().enumerate() {
47        grid.entry(cell(p)).or_default().push(i as u32);
48    }
49
50    let eps2 = eps * eps;
51    // Neighbors of point i within eps (inclusive), scanning the 27 surrounding cells.
52    let neighbors = |i: usize, out: &mut Vec<u32>| {
53        out.clear();
54        let p = points[i];
55        let (ci, cj, ck) = cell(&p);
56        for di in -1..=1 {
57            for dj in -1..=1 {
58                for dk in -1..=1 {
59                    if let Some(v) = grid.get(&(ci + di, cj + dj, ck + dk)) {
60                        for &j in v {
61                            let q = points[j as usize];
62                            let d = (p[0] - q[0]).powi(2)
63                                + (p[1] - q[1]).powi(2)
64                                + (p[2] - q[2]).powi(2);
65                            if d <= eps2 {
66                                out.push(j);
67                            }
68                        }
69                    }
70                }
71            }
72        }
73    };
74
75    let mut labels = vec![NOISE; n];
76    let mut visited = vec![false; n];
77    // `queued` dedups the BFS frontier: each point is enqueued at most once across the whole run, so
78    // the queue stays O(n) instead of O(sum of neighbourhood sizes) — which, on dense MS peaks, would
79    // otherwise balloon to billions of (mostly duplicate) entries.
80    let mut queued = vec![false; n];
81    let mut cid: i32 = 0;
82    let mut nb: Vec<u32> = Vec::new();
83    let mut nbj: Vec<u32> = Vec::new();
84    let mut queue: VecDeque<u32> = VecDeque::new();
85
86    // Throttled progress on the visited count (~1% granularity).
87    let report_step = (n / 100).max(1);
88    let mut visited_count = 0usize;
89    let mut since_report = 0usize;
90    let bump = |count: &mut usize, since: &mut usize, progress: &mut dyn FnMut(usize)| {
91        *count += 1;
92        *since += 1;
93        if *since >= report_step {
94            *since = 0;
95            progress(*count);
96        }
97    };
98
99    for i in 0..n {
100        if visited[i] {
101            continue;
102        }
103        visited[i] = true;
104        bump(&mut visited_count, &mut since_report, &mut progress);
105        neighbors(i, &mut nb);
106        if nb.len() < min_pts {
107            continue; // noise (may still be claimed as a border point later)
108        }
109        // Seed a new cluster and flood-fill density-reachable points.
110        labels[i] = cid;
111        queue.clear();
112        for &j in &nb {
113            if !queued[j as usize] {
114                queued[j as usize] = true;
115                queue.push_back(j);
116            }
117        }
118        while let Some(j) = queue.pop_front() {
119            let j = j as usize;
120            if labels[j] == NOISE {
121                labels[j] = cid; // claim border (or previously-noise) point; first cluster wins
122            }
123            if !visited[j] {
124                visited[j] = true;
125                bump(&mut visited_count, &mut since_report, &mut progress);
126                neighbors(j, &mut nbj);
127                if nbj.len() >= min_pts {
128                    // j is a core point -> add its not-yet-queued neighbours to the frontier.
129                    for &m in &nbj {
130                        if !queued[m as usize] {
131                            queued[m as usize] = true;
132                            queue.push_back(m);
133                        }
134                    }
135                }
136            }
137        }
138        cid += 1;
139    }
140    progress(n); // final 100%
141    (labels, cid as usize)
142}

144/// Per-axis multipliers applied to the normalized-cube coordinates (`x=m/z`, `y=1/K0`, `z=RT`) before
145/// DBSCAN, so the clustering metric is equi-distanced *and region-independent*: at the reference eps,
146/// the neighbourhood reach is a fixed physical amount on each axis — `rt_cycles` MS1 cycles,
147/// `im_scans` TIMS scans, `mz_peak_widths` TOF peak-widths — regardless of how the focus region is
148/// cropped. Each axis solves `reach = eps_ref · range / (2·scale)` for its target reach (the cube maps
149/// each real range onto width 2), anchoring the unit to a *run-level* physical constant. An axis whose
150/// anchor is unavailable (`<= 0`) falls back to `1.0` (uncalibrated, region-relative).
151#[derive(Clone, Copy, Debug, PartialEq)]
152pub struct AxisScales {
153    pub mz: f32,
154    pub im: f32,
155    pub rt: f32,
156}
157
158/// Inputs for [`cluster_axis_scales`] — the focus region (real units) plus run-level anchors + knobs.
159#[derive(Clone, Copy, Debug)]
160pub struct ScaleInputs {
161    /// Real-unit focus-region bounds `[(mz_lo,hi), (im_lo,hi), (rt_lo,hi)]`.
162    pub bounds: [(f64, f64); 3],
163    /// Calibration reference eps (the live eps scales the reach proportionally about this).
164    pub eps_ref: f64,
165    /// Run-level seconds per MS1 cycle; `0` = unknown (RT left uncalibrated).
166    pub cycle_duration: f64,
167    /// Run-level 1/K0 per TIMS scan; `0` = unknown (1/K0 left uncalibrated).
168    pub im_per_scan: f64,
169    /// TOF resolution (peak width = m/z / resolution).
170    pub mz_resolution: f64,
171    pub rt_cycles: f64,
172    pub im_scans: f64,
173    pub mz_peak_widths: f64,
174}
175
176/// Compute the per-axis scales (see [`AxisScales`]).
177pub fn cluster_axis_scales(inp: &ScaleInputs) -> AxisScales {
178    // scale so the target real-unit reach maps to exactly `eps_ref` after normalize+scale.
179    let scale = |range: f64, target_reach: f64| -> f32 {
180        if range.is_finite() && range > 0.0 && target_reach.is_finite() && target_reach > 0.0 {
181            let s = (inp.eps_ref * range / (2.0 * target_reach)) as f32;
182            if s.is_finite() && s > 0.0 {
183                s
184            } else {
185                1.0 // guard a malformed eps_ref (NaN / <=0) from producing a degenerate scale
186            }
187        } else {
188            1.0 // anchor unavailable / degenerate range -> normalized (region-relative) fallback
189        }
190    };
191    let mz_range = (inp.bounds[0].1 - inp.bounds[0].0).abs();
192    let im_range = (inp.bounds[1].1 - inp.bounds[1].0).abs();
193    let rt_range = (inp.bounds[2].1 - inp.bounds[2].0).abs();
194    let mz_center = ((inp.bounds[0].0 + inp.bounds[0].1) * 0.5).abs();
195    let peak_width = if inp.mz_resolution > 0.0 { mz_center / inp.mz_resolution } else { 0.0 };
196    AxisScales {
197        mz: scale(mz_range, inp.mz_peak_widths.max(0.0) * peak_width),
198        im: scale(im_range, inp.im_scans.max(0.0) * inp.im_per_scan),
199        rt: scale(rt_range, inp.rt_cycles.max(0.0) * inp.cycle_duration),
200    }
201}
202
203/// The real-unit neighbourhood reach of `eps` on an axis, given its `scale` and the region's real
204/// `range`. Inverse of the calibration in [`cluster_axis_scales`].
205pub fn axis_reach(eps: f64, range: f64, scale: f32) -> f64 {
206    if scale > 0.0 {
207        eps * range / (2.0 * scale as f64)
208    } else {
209        f64::INFINITY
210    }
211}

217    fn base(im: (f64, f64)) -> ScaleInputs {
218        ScaleInputs {
219            bounds: [(500.0, 520.0), im, (30.0, 60.0)],
220            eps_ref: 0.012,
221            cycle_duration: 0.1,  // s / cycle
222            im_per_scan: 0.0008,  // 1/K0 / scan
223            mz_resolution: 50_000.0,
224            rt_cycles: 2.0,
225            im_scans: 10.0,
226            mz_peak_widths: 1.5,
227        }

230    /// Normalize real points into `bounds`, apply the scales, cluster.
231    fn cluster_real(
232        real: &[[f64; 3]],
233        bounds: [(f64, f64); 3],
234        s: AxisScales,
235        eps: f32,
236        min_pts: usize,
237    ) -> Vec<i32> {
238        let pts: Vec<[f32; 3]> = real
239            .iter()
240            .map(|p| {
241                let mut q = [0f32; 3];
242                for a in 0..3 {
243                    let c = (bounds[a].0 + bounds[a].1) * 0.5;
244                    let h = (bounds[a].1 - bounds[a].0) * 0.5;
245                    q[a] = ((p[a] - c) / h) as f32;
246                }
247                [q[0] * s.mz, q[1] * s.im, q[2] * s.rt]
248            })
249            .collect();
250        dbscan(&pts, eps, min_pts).0
251    }
252
253    /// Re-label to first-seen order so cluster-id numbering doesn't matter; noise stays -1.
254    fn canonical(labels: &[i32]) -> Vec<i32> {
255        let mut map = std::collections::HashMap::new();
256        let mut next = 0i32;
257        labels
258            .iter()
259            .map(|&l| {
260                if l < 0 {
261                    -1
262                } else {
263                    *map.entry(l).or_insert_with(|| {
264                        let v = next;
265                        next += 1;
266                        v
267                    })
268                }
269            })
270            .collect()
271    }

273    /// The realized reach equals the configured physical target on every axis — and is invariant to
274    /// the focus-region width (the whole point of the calibration).
275    #[test]
276    fn reach_equals_target_and_is_region_independent() {
277        let eps = 0.012; // == eps_ref
278        for im in [(0.60, 1.50), (0.95, 1.05)] {
279            let inp = base(im);
280            let s = cluster_axis_scales(&inp);
281            let im_reach = axis_reach(eps, im.1 - im.0, s.im);
282            let rt_reach = axis_reach(eps, 60.0 - 30.0, s.rt);
283            let mz_reach = axis_reach(eps, 520.0 - 500.0, s.mz);
284            // Relative tolerance absorbs the f32 scale rounding; still 3 orders tighter than the ~9x
285            // region-dependence bug it guards against.
286            let close = |got: f64, want: f64| (got / want - 1.0).abs() < 1e-4;
287            assert!(close(im_reach, 10.0 * 0.0008), "im_reach={im_reach}");
288            assert!(close(rt_reach, 2.0 * 0.1), "rt_reach={rt_reach}");
289            let pw = 510.0 / 50_000.0;
290            assert!(close(mz_reach, 1.5 * pw), "mz_reach={mz_reach}");
291        }
292    }

294    /// Region-invariance at the clustering level: the SAME real-unit points, normalized into a wide vs
295    /// a tight IM region and scaled, produce identical DBSCAN labels.
296    #[test]
297    fn dbscan_labels_invariant_to_im_focus() {
298        // Four points spaced 0.004 1/K0 along mobility (same m/z + RT). reach = 10·0.0008 = 0.008 >
299        // 0.004, so they chain into one cluster — in BOTH regions, if the calibration is correct.
300        let real = [
301            [510.0, 1.000, 45.0],
302            [510.0, 1.004, 45.0],
303            [510.0, 1.008, 45.0],
304            [510.0, 1.012, 45.0],
305        ];
306        let wide = (0.60, 1.50);
307        let tight = (0.99, 1.02);
308        let lw = cluster_real(&real, [(500.0, 520.0), wide, (30.0, 60.0)], cluster_axis_scales(&base(wide)), 0.012, 2);
309        let lt = cluster_real(&real, [(500.0, 520.0), tight, (30.0, 60.0)], cluster_axis_scales(&base(tight)), 0.012, 2);
310        assert_eq!(canonical(&lw), canonical(&lt), "labels must be region-independent");
311        assert_eq!(*lw.iter().max().unwrap() + 1, 1, "expected exactly one cluster");
312    }

314    /// Documents the bug the calibration fixes: with 1/K0 left uncalibrated (`im` scale = 1, the old
315    /// behavior), the same points give DIFFERENT labels in a wide vs a tight IM region.
316    #[test]
317    fn uncalibrated_im_diverges_with_focus() {
318        let real = [
319            [510.0, 1.000, 45.0],
320            [510.0, 1.004, 45.0],
321            [510.0, 1.008, 45.0],
322            [510.0, 1.012, 45.0],
323        ];
324        let wide = (0.60, 1.50);
325        let tight = (0.99, 1.02);
326        let uncal = AxisScales { mz: 1.0, im: 1.0, rt: 1.0 };
327        let lw = cluster_real(&real, [(500.0, 520.0), wide, (30.0, 60.0)], uncal, 0.012, 2);
328        let lt = cluster_real(&real, [(500.0, 520.0), tight, (30.0, 60.0)], uncal, 0.012, 2);
329        // wide: spacing 2·0.004/0.9 ≈ 0.0089 < eps -> chains; tight: 2·0.004/0.03 ≈ 0.27 > eps -> shatters

333    /// The live eps (not eps_ref) scales the reach proportionally: 2x eps -> 2x reach on every axis.
334    #[test]
335    fn live_eps_scales_reach_proportionally() {
336        let s = cluster_axis_scales(&base((0.60, 1.50)));
337        for (range, scale) in [(0.9, s.im), (30.0, s.rt), (20.0, s.mz)] {
338            let r1 = axis_reach(0.012, range, scale);
339            let r2 = axis_reach(0.024, range, scale);
340            assert!((r2 / r1 - 2.0).abs() < 1e-4, "reach should scale with eps");
341        }
342    }

358    fn two_dense_blobs_plus_noise() {
359        let mut pts = Vec::new();
360        // Blob A around (-0.5,-0.5,-0.5), blob B around (0.5,0.5,0.5).
361        for i in 0..50 {
362            let t = i as f32 * 0.001;
363            pts.push([-0.5 + t, -0.5 - t, -0.5 + t]);
364            pts.push([0.5 - t, 0.5 + t, 0.5 - t]);
365        }
366        // A lone outlier.
367        pts.push([0.95, -0.95, 0.0]);
368        let (labels, k) = dbscan(&pts, 0.05, 4);
369        assert_eq!(k, 2, "expected two clusters");
370        assert_eq!(*labels.last().unwrap(), NOISE, "outlier should be noise");
371    }
