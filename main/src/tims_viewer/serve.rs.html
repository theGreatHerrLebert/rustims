1//! Point-serving HTTP server with on-demand region queries (FOCUS_LENS_PLAN.md A1).
2//!
3//! Holds the full run's metadata and answers region-scoped loads on demand:
4//!
5//! ```text
6//!   GET /points[?n=&mz0=&mz1=&im0=&im1=&rt0=&rt1=&imin=]  -> packed little-endian GpuPoint bytes
7//!   GET /meta  [same query]                               -> JSON (region bounds, percentiles, hists)
8//! ```
9//!
10//! No query => the full run. A query selects a 4D region (RT realized by frame selection; m/z·1/K0·
11//! intensity culled at the source) and the budget concentrates on it. Each built load is cached by
12//! `(region, budget)` and shared between `/points` and `/meta`. Several worker threads serve
13//! concurrently so a multi-second region load can't stall other requests. Localhost only.
14

29/// Worker threads pulling requests off the shared server (so one slow load can't block others).
30const WORKERS: usize = 4;
31const I_HIST_BINS: usize = 80;

33/// A 4D region of the run in real units (RT is realized via frame selection).
34#[derive(Clone, Copy)]
35struct Region4D {
36    mz: (f64, f64),
37    im: (f64, f64),
38    rt: (f64, f64),
39    imin: f32,
40}
41
42/// A built, cached load: shared bytes for `/points` and `/meta`, plus the point count.
43struct LoadResult {
44    points: Arc<[u8]>,
45    meta: Arc<[u8]>,
46    n_points: usize,
47    /// Lazily-populated zstd-compressed `points` (only when `--compress` + a zstd client). The raw
48    /// body is cached; without this the compressed form is re-encoded (level 3 over ~356 MB,
49    /// 0.3–0.9 s) on every request. Encoded once on first compressed request; a race re-encodes once.
50    points_zstd: std::sync::OnceLock<Arc<[u8]>>,
51}
52
53/// Output of a build: the cacheable result plus the intensity reference the full-run build seeds
54/// for later region survivor estimates.
55struct Built {
56    result: LoadResult,
57    i_hist: Vec<u32>,
58    i_p99: f32,
59}

61/// Quantized cache key (regions are f64, so not directly Hash/Eq).
62#[derive(Clone, Copy, PartialEq, Eq, Hash)]
63struct LoadKey {
64    mz0: i64,
65    mz1: i64,
66    im0: i64,
67    im1: i64,
68    rt0: i64,
69    rt1: i64,
70    imin: i64,
71    budget: u64,
72}
73
74/// Per-field quantization grid `[mz, im, rt, imin]`, shared by `LoadKey::of` (key rounding) and
75/// `snap_region` (reported bounds). These MUST use the same scales: if a request rounded to one key
76/// but reported bounds that snap to a different key, `/points` and `/meta` could disagree. 1/K0 spans
77/// ~1 unit so it's quantized finely (1e6); the rest to 1e3.
78const QUANT: [f64; 4] = [1e3, 1e6, 1e3, 1e3];
79
80impl LoadKey {
81    fn of(r: &Region4D, budget: usize) -> Self {
82        let q = |x: f64, s: f64| (x * s).round() as i64;
83        LoadKey {
84            mz0: q(r.mz.0, QUANT[0]),
85            mz1: q(r.mz.1, QUANT[0]),
86            im0: q(r.im.0, QUANT[1]),
87            im1: q(r.im.1, QUANT[1]),
88            rt0: q(r.rt.0, QUANT[2]),
89            rt1: q(r.rt.1, QUANT[2]),
90            imin: q(r.imin as f64, QUANT[3]),
91            budget: budget as u64,
92        }
93    }
94}

96/// Max distinct (region, budget) loads kept resident. Each can be hundreds of MB, so bound it; the
97/// pinned full-run base is never evicted (instant Reset).
98const CACHE_MAX: usize = 8;
99
100/// Bounded FIFO cache of built loads. The pinned key (full run) is never evicted.
101struct Cache {
102    map: HashMap<LoadKey, Arc<LoadResult>>,
103    order: VecDeque<LoadKey>,
104    pinned: LoadKey,
105}
106
107impl Cache {
108    fn get(&self, k: &LoadKey) -> Option<Arc<LoadResult>> {
109        self.map.get(k).cloned()
110    }
111    fn insert(&mut self, k: LoadKey, v: Arc<LoadResult>) {
112        if self.map.insert(k, v).is_none() {
113            self.order.push_back(k);
114        }
115        while self.map.len() > CACHE_MAX {
116            // Evict the oldest non-pinned entry; drop stale/pinned keys from the order queue.
117            let mut evicted = false;
118            while let Some(old) = self.order.pop_front() {
119                if old != self.pinned && self.map.remove(&old).is_some() {
120                    evicted = true;
121                    break;
122                }
123            }
124            if !evicted {
125                break;
126            }
127        }
128    }
129}

156/// Max datasets kept resident at once — each holds a full-run load (hundreds of MB) plus its region
157/// cache, so switching evicts the least-recently-built (it rebuilds in seconds if revisited).
158const MAX_RESIDENT_DATASETS: usize = 2;
159
160/// LRU of built per-dataset `State`s, keyed by dataset id.
161#[derive(Default)]
162struct StateCache {
163    map: HashMap<usize, Arc<State>>,
164    order: VecDeque<usize>,
165}
166
167impl StateCache {
168    fn get(&mut self, id: usize) -> Option<Arc<State>> {
169        let s = self.map.get(&id).cloned();
170        if s.is_some() {
171            self.order.retain(|&x| x != id);
172            self.order.push_back(id); // mark most-recently-used
173        }
174        s
175    }
176    fn insert(&mut self, id: usize, s: Arc<State>) {
177        if self.map.insert(id, s).is_none() {
178            self.order.push_back(id);
179        }
180        while self.map.len() > MAX_RESIDENT_DATASETS {
181            match self.order.pop_front() {
182                Some(old) if old != id => {
183                    self.map.remove(&old);
184                }
185                Some(_) | None => break,
186            }
187        }
188    }
189}

191/// The multi-dataset server: a registry + a lazily-built, LRU-bounded set of per-dataset states.
192struct Hub {
193    source: ServeSource,
194    entries: Vec<DatasetEntry>,
195    budget: Option<usize>,
196    /// zstd-compress the /points payload when the client advertises Accept-Encoding: zstd.
197    compress: bool,
198    states: Mutex<StateCache>,
199    /// Per-dataset build locks: a dataset switch fires `/meta` + `/points` + `/windows` near-
200    /// simultaneously, so serialize concurrent builds of the *same* id (the rest re-check the cache
201    /// and hit it) — without serializing builds of *different* datasets.
202    build_locks: Mutex<HashMap<usize, Arc<Mutex<()>>>>,
203    datasets_json: Arc<[u8]>,
204    /// PROD only: directory the /telemetry + /feedback endpoints append JSONL to. `None` in dev —
205    /// both endpoints report "disabled" and the frontend hides the feedback widget. The lock
206    /// serializes appends across worker threads.
207    telemetry_dir: Option<PathBuf>,
208    telemetry_lock: Mutex<()>,
209}
210
211impl Hub {
212    /// Build the load Plan for dataset `id` (loads its metadata; cheap relative to the full build).
213    fn plan_for(&self, id: usize) -> Result<Plan> {
214        match &self.source {
215            ServeSource::Demo(meta) => Ok(Plan::new(meta.clone(), true, self.budget)),
216            ServeSource::Datasets(paths) => {
217                let p = paths
218                    .get(id)
219                    .ok_or_else(|| anyhow::anyhow!("dataset {id} out of range"))?;
220                let meta = MetaIndex::load(&p.display().to_string())?;
221                Ok(Plan::new(meta, false, self.budget))
222            }
223        }
224    }
225}

227/// Resolve the `State` for dataset `id`, building (and caching, LRU-evicting) it on first request.
228/// The build runs outside the lock so a multi-second load can't block other datasets' requests.
229fn get_or_build_state(hub: &Hub, id: usize) -> Result<Arc<State>> {
230    // Poison-safe throughout (like the cache lock in get_or_build): a build that panics on a corrupt
231    // `.d` must not poison these mutexes and take every later request for this — or any — dataset down.
232    if let Some(s) = hub.states.lock().unwrap_or_else(|p| p.into_inner()).get(id) {
233        return Ok(s);
234    }
235    // Single-flight per dataset: hold this id's build lock so concurrent first-touches (a switch's
236    // /meta + /points + /windows) don't each build the same hundreds-of-MB state.
237    let build_lock = hub
238        .build_locks
239        .lock()
240        .unwrap_or_else(|p| p.into_inner())
241        .entry(id)
242        .or_default()
243        .clone();
244    let _guard = build_lock.lock().unwrap_or_else(|p| p.into_inner());
245    // Re-check: another thread may have built it while we waited on the build lock.
246    if let Some(s) = hub.states.lock().unwrap_or_else(|p| p.into_inner()).get(id) {
247        return Ok(s);
248    }
249    let plan = hub.plan_for(id)?;
250    // Contain a build panic (malformed run → rustdf/loader/bytemuck can panic) as an error instead
251    // of unwinding through the held build_lock (which would poison it and cascade the worker pool
252    // down). Catching here also lets the guard drop cleanly, leaving the lock usable for a retry.
253    let state = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| build_state(plan)))
254        .map_err(|_| anyhow::anyhow!("dataset {id} build panicked (corrupt or unreadable data)"))??;
255    let state = Arc::new(state);
256    hub.states.lock().unwrap_or_else(|p| p.into_inner()).insert(id, state.clone());
257    Ok(state)
258}

260/// Eagerly build one dataset's `State`: the full-run load (first paint + intensity reference), the
261/// DIA window footprints, and a fresh region cache pinned on the full run.
262fn build_state(plan: Plan) -> Result<State> {
263    let Plan { meta, is_demo, budget } = plan;
264    // Snap the eager full region onto the cache grid so a client's explicit full-bounds query
265    // (also snapped in parse_query) resolves to this same cached entry instead of rebuilding.
266    let full = snap_region(full_region(&meta));
267    let built = build_load_result(&meta, is_demo, &full, budget, None)?;
268    log::info!(
269        "built dataset '{}': {} points ({:.1} MB)",
270        meta.data_path,
271        built.result.n_points,
272        built.result.points.len() as f64 / 1e6,
273    );
274    let full_key = LoadKey::of(&full, budget);
275    let mut map = HashMap::new();
276    map.insert(full_key, Arc::new(built.result));
277    let windows_json = build_windows_json(is_demo, &meta.data_path);
278    Ok(State {
279        meta,
280        is_demo,
281        default_budget: budget,
282        full_i_hist: built.i_hist,
283        full_i_p99: built.i_p99,
284        windows_json,
285        cache: Mutex::new(Cache {
286            map,
287            order: VecDeque::new(),
288            pinned: full_key,
289        }),
290    })
291}

293/// Display name for a dataset path: the `.d` folder's stem (e.g. `/data/run5.d` → `run5`).

298        .unwrap_or_else(|| path.display().to_string())

301/// `/datasets` JSON: `[{"id":0,"name":"run5"}, ...]`. Built via serde_json so any folder name
302/// (control chars, quotes, unicode) is escaped correctly.
303fn build_datasets_json(entries: &[DatasetEntry]) -> Arc<[u8]> {
304    let items: Vec<serde_json::Value> = entries
305        .iter()
306        .enumerate()
307        .map(|(i, e)| serde_json::json!({ "id": i, "name": e.name }))
308        .collect();
309    serde_json::to_vec(&items).unwrap_or_else(|_| b"[]".to_vec()).into()
310}

312/// Build the registry, eagerly build the default dataset (fast first paint), then serve region
313/// queries — and dataset switches — on demand from `WORKERS` threads.
314pub fn serve(
315    source: ServeSource,
316    budget: Option<usize>,
317    port: u16,
318    compress: bool,
319    telemetry_dir: Option<PathBuf>,
320) -> Result<()> {
321    anyhow::ensure!(
322        cfg!(target_endian = "little"),
323        "the point wire format is little-endian; --serve is unsupported on this big-endian target"
324    );
325    let entries: Vec<DatasetEntry> = match &source {
326        ServeSource::Demo(_) => vec![DatasetEntry { name: "DEMO".into() }],
327        ServeSource::Datasets(paths) => {
328            anyhow::ensure!(!paths.is_empty(), "no datasets to serve");
329            paths.iter().map(|p| DatasetEntry { name: dataset_name(p) }).collect()
330        }
331    };
332    let datasets_json = build_datasets_json(&entries);
333    let names: Vec<&str> = entries.iter().map(|e| e.name.as_str()).collect();
334    log::info!("registry: {} dataset(s): {}", entries.len(), names.join(", "));
335
336    let hub = Arc::new(Hub {
337        source,
338        entries,
339        budget,
340        compress,
341        states: Mutex::new(StateCache::default()),
342        build_locks: Mutex::new(HashMap::new()),
343        datasets_json,
344        telemetry_dir,
345        telemetry_lock: Mutex::new(()),
346    });
347    if compress {
348        log::info!("over-the-wire zstd compression enabled for /points (Accept-Encoding negotiated)");
349    }
350    if let Some(dir) = &hub.telemetry_dir {
351        if let Err(e) = std::fs::create_dir_all(dir) {
352            log::warn!("telemetry dir {} not writable ({e}); telemetry/feedback will 500", dir.display());
353        } else {
354            log::info!("telemetry + feedback enabled -> {}", dir.display());
355        }
356    }
357
358    // Eagerly build the default dataset so its first paint is instant.
359    get_or_build_state(&hub, 0)?;
360
361    let server = Arc::new(
362        Server::http(("127.0.0.1", port)).map_err(|e| anyhow::anyhow!("bind {port}: {e}"))?,
363    );
364    log::info!(
365        "serving default '{}' + on-demand region/dataset queries on http://localhost:{port} (localhost only)",
366        hub.entries[0].name,
367    );
368
369    let mut handles = Vec::with_capacity(WORKERS);
370    for _ in 0..WORKERS {
371        let (server, hub) = (server.clone(), hub.clone());
372        handles.push(std::thread::spawn(move || worker(&server, &hub)));
373    }
374    for h in handles {
375        let _ = h.join();
376    }
377    Ok(())
378}

380fn worker(server: &Server, hub: &Hub) {
381    while let Ok(req) = server.recv() {
382        if let Err(e) = handle(req, hub) {
383            log::warn!("HTTP respond failed: {e}");
384        }
385    }
386}

388/// Dataset id from `?dataset=N` (default 0), clamped to the registry size.
389fn parse_dataset_id(url: &str, n_datasets: usize) -> usize {
390    let q = url.split('?').nth(1).unwrap_or("");
391    q.split('&')
392        .find_map(|kv| kv.strip_prefix("dataset=").and_then(|v| v.parse::<usize>().ok()))
393        .unwrap_or(0)
394        .min(n_datasets.saturating_sub(1))
395}

397fn handle(req: Request, hub: &Hub) -> std::io::Result<()> {
398    if *req.method() == Method::Options {
399        // CORS preflight (the trunk page is served from a different origin).
400        return req.respond(
401            Response::empty(204)
402                .with_header(cors())
403                .with_header(header("Access-Control-Allow-Methods", "GET, POST, OPTIONS"))
404                .with_header(header("Access-Control-Allow-Headers", "*")),
405        );
406    }
407    let url = req.url().to_string();
408    let path = url.split('?').next().unwrap_or("");
409    if *req.method() == Method::Get && path == "/datasets" {
410        // The selectable-dataset registry (the frontend renders this as a dropdown).
411        return respond_bytes(req, hub.datasets_json.clone(), "application/json");
412    }
413    if *req.method() == Method::Get && path == "/clientconfig" {
414        // Tells the frontend whether to wire error telemetry + show the feedback widget (PROD only).
415        let on = hub.telemetry_dir.is_some();
416        let json = format!("{{\"telemetry\":{on},\"feedback\":{on}}}");
417        return respond_bytes(req, json.into_bytes().into(), "application/json");
418    }
419    if *req.method() == Method::Post && (path == "/telemetry" || path == "/feedback") {
420        let file = if path == "/telemetry" { "telemetry.jsonl" } else { "feedback.jsonl" };
421        return log_client_event(req, hub, file);
422    }
423    let want_windows = path == "/windows";
424    let (want_points, want_meta) = (path == "/points", path == "/meta");
425    if *req.method() == Method::Get && (want_windows || want_points || want_meta) {
426        // Resolve (and lazily build) the selected dataset's state.
427        let id = parse_dataset_id(&url, hub.entries.len());
428        let state = match get_or_build_state(hub, id) {
429            Ok(s) => s,
430            Err(e) => {
431                return req.respond(
432                    Response::from_string(format!("dataset load failed: {e}"))
433                        .with_status_code(StatusCode(500))
434                        .with_header(cors()),
435                );
436            }
437        };
438        if want_windows {
439            // Run-level DIA isolation-window footprints (static JSON; region-independent).
440            return respond_bytes(req, state.windows_json.clone(), "application/json");
441        }
442        let (region, budget) = parse_query(&url, &state);
443        return match get_or_build(&state, &region, budget) {
444            Ok(lr) if want_points => {
445                respond_points(req, lr.clone(), hub.compress)
446            }
447            Ok(lr) => respond_bytes(req, lr.meta.clone(), "application/json"),
448            Err(e) => req.respond(
449                Response::from_string(format!("load failed: {e}"))
450                    .with_status_code(500)
451                    .with_header(cors()),
452            ),
453        };
454    }
455    req.respond(
456        Response::from_string(
457            "tims-viewer point server\n\
458             GET /points|/meta[?n=&mz0=&mz1=&im0=&im1=&rt0=&rt1=&imin=]\n",
459        )
460        .with_header(cors()),
461    )
462}

464/// Parse the region + budget from a query string, defaulting to the full run. Ranges are clamped to
465/// the full bounds and ordered (lo <= hi).
466fn parse_query(url: &str, state: &State) -> (Region4D, usize) {
467    let full = full_region(&state.meta);
468    let mut r = full;
469    let mut budget = state.default_budget;
470    if let Some(q) = url.split('?').nth(1) {
471        for kv in q.split('&') {
472            let Some((k, v)) = kv.split_once('=') else { continue };
473            match k {
474                "n" => {
475                    if let Ok(n) = v.parse::<usize>() {
476                        budget = n;
477                    }
478                }
479                "imin" => {
480                    if let Ok(x) = v.parse::<f32>() {
481                        if x.is_finite() {
482                            r.imin = x.max(0.0);
483                        }
484                    }
485                }
486                _ => {
487                    // Ignore non-finite (NaN/inf) coordinates rather than letting them reach the
488                    // cube transform / cache key / normalization.
489                    if let Some(x) = v.parse::<f64>().ok().filter(|x| x.is_finite()) {
490                        match k {
491                            "mz0" => r.mz.0 = x,
492                            "mz1" => r.mz.1 = x,
493                            "im0" => r.im.0 = x,
494                            "im1" => r.im.1 = x,
495                            "rt0" => r.rt.0 = x,
496                            "rt1" => r.rt.1 = x,
497                            _ => {}
498                        }
499                    }
500                }
501            }
502        }
503    }
504    // The demo loader generates frames 0..N starting at RT 0 and ignores frame ids, so it can't
505    // honor an RT sub-window — use the full RT range for demo sessions (m/z·1/K0·intensity focus
506    // still works). Real data realizes RT via frame selection.
507    if state.is_demo {
508        r.rt = full.rt;
509    }
510    r.mz = clamp_range(r.mz, full.mz);
511    r.im = clamp_range(r.im, full.im);
512    r.rt = clamp_range(r.rt, full.rt);
513    // Snap to the cache-key grid so the reported bounds exactly match the key (no aliasing where
514    // two distinct requests round to the same key but expect different results).
515    r = snap_region(r);
516    // Cap the budget to the server pool (also guards the downstream `budget * 85` from overflow).
517    // `.max(1)` on the ceiling so a `--budget 0` misconfig can't make clamp(1, 0) panic (min > max).
518    (r, budget.clamp(1, state.default_budget.max(1)))
519}

521/// Round a region onto the same grid the cache key quantizes to, so key and bounds stay consistent.
522fn snap_region(mut r: Region4D) -> Region4D {
523    let snap = |x: f64, s: f64| (x * s).round() / s;
524    r.mz = (snap(r.mz.0, QUANT[0]), snap(r.mz.1, QUANT[0]));
525    r.im = (snap(r.im.0, QUANT[1]), snap(r.im.1, QUANT[1]));
526    r.rt = (snap(r.rt.0, QUANT[2]), snap(r.rt.1, QUANT[2]));
527    r.imin = snap(r.imin as f64, QUANT[3]) as f32;
528    r
529}

531/// Look up a cached load or build (and cache) it. The build runs outside the lock so concurrent
532/// requests for *other* regions are not blocked; a rare double-build of the same region is harmless.
533fn get_or_build(state: &State, region: &Region4D, budget: usize) -> Result<Arc<LoadResult>> {
534    let key = LoadKey::of(region, budget);
535    // Poison-safe: a panicked worker must not wedge the others.
536    let lock = || state.cache.lock().unwrap_or_else(|p| p.into_inner());
537    if let Some(lr) = lock().get(&key) {
538        return Ok(lr);
539    }
540    // Build outside the lock (a multi-second load mustn't block other regions). Builds are
541    // deterministic, so a rare concurrent double-build of the same key yields identical bytes —
542    // `/meta` and `/points` stay coherent regardless of which build each request observes.
543    let built = build_load_result(
544        &state.meta,
545        state.is_demo,
546        region,
547        budget,
548        Some((&state.full_i_hist, state.full_i_p99)),
549    )?;
550    let lr = Arc::new(built.result);
551    lock().insert(key, lr.clone());
552    Ok(lr)
553}

555/// Run one region load: select frames, size the stride to the region survivor estimate, stream
556/// through the loader (normalized to the region so it fills the cube), shuffle, and pack the
557/// `/points` bytes + `/meta` JSON.
558/// Build the static `/windows` JSON: `{ max_window_group, windows: [[group, mz0, mz1, im0, im1], …] }`
559/// in real units. Demo or non-DIA runs yield an empty set.
560fn build_windows_json(is_demo: bool, data_path: &str) -> Arc<[u8]> {
561    let json = if is_demo {
562        serde_json::json!({ "max_window_group": 0, "windows": [] })
563    } else {
564        let ds = rustdf::data::dataset::TimsDataset::new("NO_SDK", data_path, false, false);
565        let (rects, max_group) = crate::data::loader::dia_window_rects(&ds, data_path);
566        let windows: Vec<[f64; 5]> = rects
567            .iter()
568            .map(|r| [r.group as f64, r.mz0, r.mz1, r.im0, r.im1])
569            .collect();
570        serde_json::json!({ "max_window_group": max_group, "windows": windows })
571    };
572    Arc::from(json.to_string().into_bytes().into_boxed_slice())
573}

575fn build_load_result(
576    meta: &MetaIndex,
577    is_demo: bool,
578    region: &Region4D,
579    budget: usize,
580    i_ref: Option<(&[u32], f32)>,
581) -> Result<Built> {
582    let full = full_region(meta);
583
584    // RT window -> frame ids + the RT-region source size (all m/z·1/K0, pre-cull).
585    let mut frame_ids: Vec<u32> = Vec::new();
586    let mut rt_total: u64 = 0;
587    for f in &meta.frames {
588        if f.retention_time >= region.rt.0 && f.retention_time <= region.rt.1 {
589            frame_ids.push(f.id);
590            rt_total = rt_total.saturating_add(f.num_peaks);
591        }
592    }
593
594    // A region with no frames in its RT window is valid — return an empty result (the loader would
595    // otherwise error on a zero-frame run). Frames-but-zero-survivors flows through normally below.
596    if frame_ids.is_empty() {
597        return Ok(empty_result(meta, region));
598    }
599
600    // Region-relative survivor estimate -> stride (FOCUS_LENS_PLAN.md blocker-1): scale the RT
601    // source size by the m/z·1/K0 window fraction and the intensity floor's surviving fraction.
602    let span = |r: (f64, f64), f: (f64, f64)| ((r.1 - r.0) / (f.1 - f.0).max(1e-9)).clamp(0.0, 1.0);
603    let i_frac = intensity_frac(region.imin, i_ref);
604    let estimate =
605        (((rt_total as f64) * span(region.mz, full.mz) * span(region.im, full.im) * i_frac).ceil()
606            as u64)
607            .max(1);
608
609    // Re-normalize: the loader maps points to [-1,1] over the region bounds (so the region fills
610    // the view). For the full run this equals the run bounds (unchanged behavior).
611    let bounds = AxisBounds {
612        mz: AxisTransform::new(region.mz.0, region.mz.1),
613        im: AxisTransform::new(region.im.0, region.im.1),
614        rt: AxisTransform::new(region.rt.0, region.rt.1),
615    };
616
617    // No-op spatial culls (the full run) skip the per-point filter entirely.
618    const EPS: f64 = 1e-6;
619    let no_spatial = region.mz.0 <= full.mz.0 + EPS
620        && region.mz.1 >= full.mz.1 - EPS
621        && region.im.0 <= full.im.0 + EPS
622        && region.im.1 >= full.im.1 - EPS;
623    let filter = if no_spatial && region.imin <= 0.0 {
624        None
625    } else {
626        Some(RegionFilter {
627            mz: region.mz,
628            im: region.im,
629            intensity_min: region.imin,
630        })
631    };
632
633    let mode = if is_demo {
634        LoaderMode::Demo(DemoSource::new(frame_ids.len(), rt_total))
635    } else {
636        LoaderMode::Real {
637            path: meta.data_path.clone(),
638            frame_ids,
639        }
640    };
641
642    let (mut points, stats, hist) = collect(mode, bounds, estimate, budget, filter)?;
643    shuffle_points(&mut points);
644    let n_points = points.len();
645    let stride = stride_for(estimate, budget.max(1));
646    let (i_p1, i_p50, i_p99) = stats.unwrap_or_else(|| intensity_percentiles(&points));
647    let i_hist = intensity_linear_hist(&points, i_p99);
648
649    // One copy into the Arc allocation, not two: `Arc::from(&[u8])` copies the byte view directly,
650    // vs the old `.to_vec().into_boxed_slice()` which memcpy'd the whole 356 MB into a Vec first and
651    // then again into the Arc. (Can't reinterpret the Vec<GpuPoint> allocation as Vec<u8> in place —
652    // bytemuck::cast_vec requires equal alignment, and GpuPoint is align-4.)
653    let body: Arc<[u8]> = Arc::from(bytemuck::cast_slice::<GpuPoint, u8>(&points));
654    drop(points);
655
656    // Cycle duration: the run-level mean RT gap between consecutive MS1 (precursor) frames. The web
657    // uses it to cluster RT in cycle units, so precursor points across cycles stay density-connected
658    // regardless of how tightly the RT range is focused. 0 when unknown (demo / no MS1 frames).
659    let ms1_rt: Vec<f64> = meta.frames.iter().filter(|f| !f.is_ms2).map(|f| f.retention_time).collect();
660    let cycle_duration = if ms1_rt.len() >= 2 {
661        (ms1_rt[ms1_rt.len() - 1] - ms1_rt[0]) / (ms1_rt.len() - 1) as f64
662    } else {
663        0.0
664    };
665
666    // Run-level 1/K0 per TIMS scan: the FULL-run mobility span over the ramp length. Run-level (not
667    // region-scoped) so the client's IM clustering reach is focus-independent.
668    let im_per_scan = if meta.num_scans > 1 {
669        (meta.bounds.im.max - meta.bounds.im.min).abs() / (meta.num_scans as f64 - 1.0)
670    } else {
671        0.0
672    };
673    let meta_json = build_meta_json(
674        region, n_points, stride, cycle_duration, im_per_scan, meta.num_scans, i_p1, i_p50, i_p99,
675        &i_hist, hist.as_ref(),
676    );
677
678    Ok(Built {
679        result: LoadResult {
680            points: body,
681            meta: Arc::from(meta_json.into_bytes().into_boxed_slice()),
682            n_points,
683            points_zstd: std::sync::OnceLock::new(),
684        },
685        i_hist,
686        i_p99,
687    })
688}

690/// Surviving fraction of points at or above an intensity floor, estimated from the full-run linear
691/// intensity histogram over `[0, p99]`. `imin <= 0` (or no reference) => everything survives.
692fn intensity_frac(imin: f32, i_ref: Option<(&[u32], f32)>) -> f64 {
693    if imin <= 0.0 {
694        return 1.0;
695    }
696    let Some((hist, p99)) = i_ref else { return 1.0 };
697    let total: u64 = hist.iter().map(|&c| c as u64).sum();
698    if total == 0 || hist.is_empty() {
699        return 1.0;
700    }
701    let bin = ((imin / p99.max(1.0)).clamp(0.0, 1.0) * (hist.len() as f32 - 1.0)) as usize;
702    let above: u64 = hist[bin..].iter().map(|&c| c as u64).sum();
703    // Never 0: a tiny estimate yields stride 1 and we keep all (few) survivors anyway.
704    (above as f64 / total as f64).max(1.0 / total as f64)
705}

707/// Does the request actually accept `zstd`? Token-correct: matches the `zstd` coding exactly (not a
708/// substring like `x-zstd`) across a `gzip, zstd, br` list and rejects an explicit `q=0`.
709fn accepts_zstd(req: &Request) -> bool {
710    req.headers().iter().filter(|h| h.field.equiv("Accept-Encoding")).any(|h| {
711        h.value.as_str().split(',').any(|tok| {
712            let mut parts = tok.split(';').map(str::trim);
713            let coding = parts.next().unwrap_or("");
714            coding.eq_ignore_ascii_case("zstd")
715                // accept unless an explicit q=0 disqualifies it
716                && parts.all(|p| {
717                    p.strip_prefix("q=")
718                        .or_else(|| p.strip_prefix("Q="))
719                        .and_then(|v| v.parse::<f64>().ok())
720                        .map_or(true, |q| q > 0.0)
721                })
722        })
723    })
724}

726/// Serve the `/points` payload, zstd-compressing it when `--compress` is on AND the client advertises
727/// `Accept-Encoding: zstd` (the browser then decompresses `Content-Encoding: zstd` transparently — no
728/// client code). When compression is enabled the response carries `Vary: Accept-Encoding` so a shared
729/// cache/proxy can't hand a zstd body to a client that didn't ask. Falls back to raw on encode error.
730fn respond_points(req: Request, lr: Arc<LoadResult>, compress: bool) -> std::io::Result<()> {
731    if !compress {
732        return respond_bytes(req, lr.points.clone(), "application/octet-stream");
733    }
734    if accepts_zstd(&req) {
735        // Serve (and memoize) the compressed body from the cache — encode at most once per cached
736        // (region,budget) instead of re-zstd'ing ~356 MB on every request.
737        let z = match lr.points_zstd.get() {
738            Some(z) => Some(z.clone()),
739            None => match zstd::encode_all(&lr.points[..], 3) {
740                Ok(v) => {
741                    let z: Arc<[u8]> = Arc::from(v.into_boxed_slice());
742                    let _ = lr.points_zstd.set(z.clone()); // first writer wins; a race re-encodes once
743                    Some(z)
744                }
745                Err(e) => {
746                    log::warn!("zstd encode failed ({e}); sending raw points");
747                    None
748                }
749            },
750        };
751        if let Some(z) = z {
752            let len = z.len();
753            return req.respond(Response::new(
754                StatusCode(200),
755                vec![
756                    cors(),
757                    header("Content-Type", "application/octet-stream"),
758                    header("Content-Encoding", "zstd"),
759                    header("Vary", "Accept-Encoding"),
760                ],
761                Cursor::new(z),
762                Some(len),
763                None,
764            ));
765        }
766    }
767    // Raw, but still Vary — this route negotiates, so caches must key on Accept-Encoding.
768    let body = lr.points.clone();
769    let len = body.len();
770    req.respond(Response::new(
771        StatusCode(200),
772        vec![cors(), header("Content-Type", "application/octet-stream"), header("Vary", "Accept-Encoding")],
773        Cursor::new(body),
774        Some(len),
775        None,
776    ))
777}

779/// Append a client-posted JSON event (error telemetry or user feedback) as one JSONL line to the
780/// configured telemetry dir, stamped with a server-side unix time. PROD only: 404 when no dir is
781/// configured. Body is bounded and must be valid JSON so the log stays one-object-per-line.
782fn log_client_event(mut req: Request, hub: &Hub, file: &str) -> std::io::Result<()> {
783    use std::io::{Read, Write};
784    let Some(dir) = hub.telemetry_dir.clone() else {
785        return req.respond(
786            Response::from_string("telemetry disabled").with_status_code(404).with_header(cors()),
787        );
788    };
789    const MAX: usize = 64 * 1024;
790    if req.body_length().unwrap_or(0) > MAX {
791        return req.respond(
792            Response::from_string("payload too large").with_status_code(413).with_header(cors()),
793        );
794    }
795    let mut buf = String::new();
796    let _ = req.as_reader().take(MAX as u64).read_to_string(&mut buf);
797    let event: serde_json::Value = match serde_json::from_str(&buf) {
798        Ok(v) => v,
799        Err(_) => {
800            return req
801                .respond(Response::from_string("invalid json").with_status_code(400).with_header(cors()));
802        }
803    };
804    let ts = std::time::SystemTime::now()
805        .duration_since(std::time::UNIX_EPOCH)
806        .map(|d| d.as_secs())
807        .unwrap_or(0);
808    let line = serde_json::json!({ "ts_unix": ts, "event": event }).to_string();
809    {
810        let _g = hub.telemetry_lock.lock().unwrap_or_else(|p| p.into_inner());
811        let path = dir.join(file);
812        match std::fs::OpenOptions::new().create(true).append(true).open(&path) {
813            Ok(mut f) => {
814                let _ = writeln!(f, "{line}");
815            }
816            Err(e) => {
817                log::warn!("telemetry append to {} failed: {e}", path.display());
818                return req.respond(Response::empty(500).with_header(cors()));
819            }
820        }
821    }
822    req.respond(Response::empty(204).with_header(cors()))
823}

838/// Per-axis density histograms (each `HIST_BINS` bins over the region range) plus the three 2D
839/// density projections (`PROJ_BINS²`, row-major `x + bins*y`) for the box-select minimaps.
840struct Hist {
841    mz: Vec<u32>,
842    im: Vec<u32>,
843    rt: Vec<u32>,
844    proj_mz_im: Vec<u32>,
845    proj_mz_rt: Vec<u32>,
846    proj_im_rt: Vec<u32>,
847}

849/// Drain the loader (with `total_estimate` already sized to the region) into a bounded point
850/// buffer, capturing the intensity `Stats` and per-axis histograms.
851fn collect(
852    mode: LoaderMode,
853    bounds: AxisBounds,
854    estimate: u64,
855    budget: usize,
856    filter: Option<RegionFilter>,
857) -> Result<(Vec<GpuPoint>, Option<(f32, f32, f32)>, Option<Hist>)> {
858    let capacity = budget.max(1);
859    let loader = LoaderHandle::spawn(mode, bounds, estimate, capacity, filter);
860
861    let mut points: Vec<GpuPoint> = Vec::with_capacity(capacity.min(1 << 20));
862    let mut stats: Option<(f32, f32, f32)> = None;
863    let mut hist: Option<Hist> = None;
864    let mut done = false;
865    // Reservoir-sample the emitted stream to `capacity` rather than truncating its prefix: if the
866    // region estimate under-counts and the loader over-emits, truncation would keep an RT-ordered
867    // prefix (bias). A reservoir keeps a uniform sample regardless of how wrong the estimate was.
868    let mut seen: u64 = 0;
869    let mut rng: u64 = 0xD1B5_4A32_D192_ED03;
870    loop {
871        match loader.rx.recv() {
872            Ok(LoadMsg::Chunk { points: pts, .. }) | Ok(LoadMsg::PeakChunk { points: pts }) => {
873                for p in pts {
874                    seen += 1;
875                    if points.len() < capacity {
876                        points.push(p);
877                    } else {
878                        let j = (xorshift64(&mut rng) % seen) as usize;
879                        if j < capacity {
880                            points[j] = p;
881                        }
882                    }
883                }
884            }
885            Ok(LoadMsg::Stats { i_p1, i_p99, i_p50 }) => stats = Some((i_p1, i_p50, i_p99)),
886            Ok(LoadMsg::Histograms {
887                mz, im, rt, proj_mz_im, proj_mz_rt, proj_im_rt, ..
888            }) => {
889                hist = Some(Hist { mz, im, rt, proj_mz_im, proj_mz_rt, proj_im_rt });
890            }
891            Ok(LoadMsg::Done { .. }) => {
892                done = true;
893                break;
894            }
895            Ok(LoadMsg::Error(e)) => anyhow::bail!("loader error: {e}"),
896            Ok(_) => {}
897            Err(_) => break,
898        }
899    }
900    anyhow::ensure!(done, "loader thread ended before completing the load");
901    Ok((points, stats, hist))
902}

904/// A valid empty load for a region with no points (e.g. an RT window selecting no frames).
905/// Build the `/meta` JSON string. The single source of the meta schema, shared by the normal load
906/// path and the empty-region path so a field can't be added to one and silently omitted from the
907/// other. `hist: None` emits null hist/proj (empty region).
908#[allow(clippy::too_many_arguments)]
909fn build_meta_json(
910    region: &Region4D,
911    n_points: usize,
912    downsample_stride: usize,
913    cycle_duration: f64,
914    im_per_scan: f64,
915    num_scans: u32,
916    i_p1: f32,
917    i_p50: f32,
918    i_p99: f32,
919    i_hist: &[u32],
920    hist: Option<&Hist>,
921) -> String {
922    let fin = |x: f64| if x.is_finite() { x } else { 0.0 };
923    serde_json::json!({
924        "version": 1,
925        "point_stride": std::mem::size_of::<GpuPoint>(),
926        "n_points": n_points,
927        "downsample_stride": downsample_stride,
928        "cycle_duration": fin(cycle_duration),
929        "im_per_scan_1k0": fin(im_per_scan),
930        "num_scans": num_scans,
931        // Region bounds in real units — the client re-normalizes axes/crops/strips to these.
932        "bounds": {
933            "mz": [fin(region.mz.0), fin(region.mz.1)],
934            "im": [fin(region.im.0), fin(region.im.1)],
935            "rt": [fin(region.rt.0), fin(region.rt.1)],
936        },
937        "intensity": {
938            "p1": fin(i_p1 as f64), "p50": fin(i_p50 as f64), "p99": fin(i_p99 as f64),
939            "hist": i_hist,
940        },
941        // Sample-based per-axis density histograms (over the kept systematic sample + peak tail).
942        "hist": hist.map(|h| serde_json::json!({ "mz": h.mz, "im": h.im, "rt": h.rt })),
943        // Sample-based 2D density projections for the box-select minimaps.
944        "proj": hist.map(|h| serde_json::json!({
945            "bins": PROJ_BINS,
946            "mz_im": h.proj_mz_im, "mz_rt": h.proj_mz_rt, "im_rt": h.proj_im_rt,
947        })),
948    })
949    .to_string()
950}

985/// Clamp a range into `full` and order it (lo <= hi).
986fn clamp_range(r: (f64, f64), full: (f64, f64)) -> (f64, f64) {
987    let lo = r.0.clamp(full.0, full.1);
988    let hi = r.1.clamp(full.0, full.1);
989    if lo <= hi {
990        (lo, hi)
991    } else {
992        (hi, lo)
993    }
994}

996/// Intensity (p1, p50, p99), nearest-rank, fallback when the loader Stats are absent.
997fn intensity_percentiles(points: &[GpuPoint]) -> (f32, f32, f32) {
998    let mut v: Vec<f32> = points.iter().map(|p| p.intensity).filter(|x| x.is_finite()).collect();
999    if v.is_empty() {
1000        return (1.0, 1.0, 1.0);
1001    }
1002    v.sort_by(|a, b| a.total_cmp(b));
1003    let pct = |q: f32| {
1004        let idx = ((v.len() as f32 * q).ceil() as usize).saturating_sub(1).min(v.len() - 1);
1005        v[idx].max(1.0)
1006    };
1007    (pct(0.01), pct(0.50), pct(0.99))
1008}

1010/// One step of a xorshift64 PRNG (deterministic, no `rand` dependency). Shared by the reservoir
1011/// sampler in `collect` and the Fisher-Yates shuffle below.
1012#[inline]
1013fn xorshift64(state: &mut u64) -> u64 {
1014    *state ^= *state << 13;
1015    *state ^= *state >> 7;
1016    *state ^= *state << 17;
1017    *state
1018}

1020/// In-place Fisher-Yates shuffle with a seeded xorshift64. Decorrelates the loader's RT-ordered
1021/// emission so a prefix of the array is a uniform subsample.
1022fn shuffle_points(points: &mut [GpuPoint]) {
1023    let mut s: u64 = 0x9E37_79B9_7F4A_7C15;
1024    for i in (1..points.len()).rev() {
1025        points.swap(i, (xorshift64(&mut s) % (i as u64 + 1)) as usize);
1026    }
1027}

1029/// Linear intensity histogram over `[0, hi]` (real counts); values above `hi` clamp to the last bin.
1030fn intensity_linear_hist(points: &[GpuPoint], hi: f32) -> Vec<u32> {
1031    let hi = hi.max(1.0);
1032    let mut h = vec![0u32; I_HIST_BINS];
1033    for p in points {
1034        let i = p.intensity;
1035        let bin = if i.is_finite() && i > 0.0 {
1036            ((i / hi).clamp(0.0, 1.0) * (I_HIST_BINS as f32 - 1.0)) as usize
1037        } else {
1038            0
1039        };
1040        h[bin] += 1;
1041    }
1042    h
1043}
