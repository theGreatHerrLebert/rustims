1pub fn find_sparse_local_maxima_mask(
2    indices: &Vec<u32>,
3    values: &Vec<f64>,
4    window: u32,
5) -> Vec<bool> {
6    let mut local_maxima: Vec<bool> = vec![true; indices.len()];
7    for (index, sparse_index) in indices.iter().enumerate() {
8        let current_intensity: f64 = values[index];
9        for (_next_index, next_sparse_index) in
10            indices[index + 1..].iter().enumerate()
11        {
12            let next_index: usize = _next_index + index + 1;
13            let next_value: f64 = values[next_index];
14            if (next_sparse_index - sparse_index) <= window {
15                if current_intensity < next_value {
16                    local_maxima[index] = false
17                } else {
18                    local_maxima[next_index] = false
19                }
20            } else {
21                break;
22            }
23        }
24    }
25    local_maxima
26}

28pub fn filter_with_mask<T: Copy>(vec: &Vec<T>, mask: &Vec<bool>) -> Vec<T> {
