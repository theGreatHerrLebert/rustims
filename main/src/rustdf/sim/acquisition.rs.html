1//! Vendor-neutral acquisition-writer abstraction.
2//!
3//! [`AcquisitionWriter`] turns vendor-neutral [`ScanDescriptor`]s into a vendor
4//! raw file. The first implementation is [`ThermoRawWriter`] (Thermo `.raw`, via
5//! the `thermo` feature + the `thermorawfile` crate); a Bruker `.d` adapter over
6//! the existing write path is a planned follow-up.
7
8use std::io;
9
10/// MS2 quadrupole isolation window.
11#[derive(Clone, Copy, Debug)]
12pub struct IsolationWindow {
13    pub center_mz: f64,
14    pub width_mz: f64,
15    pub collision_energy: f64,
16}
17
18/// One vendor-neutral scan: a peak list plus acquisition metadata.
19///
20/// Intentionally has **no ion-mobility dimension** — it is the
21/// Bruker∩Thermo∩SCIEX common denominator. IMS-bearing Bruker frames are handled
22/// by the Bruker writer, which can flatten or carry mobility separately.
23#[derive(Clone, Debug)]
24pub struct ScanDescriptor {
25    pub ms_level: u8,
26    pub retention_time: f64,
27    /// Present for MS2 (the precursor isolation window).
28    pub isolation: Option<IsolationWindow>,
29    /// `(m/z, intensity)` pairs; need not be sorted (writers sort).
30    pub peaks: Vec<(f64, f32)>,
31}
32
33impl ScanDescriptor {
34    /// Convert a rendered event into a writer-ready descriptor (P6e-2 glue).
35    ///
36    /// `collision_energy` is the window's applied CE (eV/NCE) — it is NOT carried
37    /// on [`crate::sim::projector::RenderedEvent`]'s isolation (which is m/z-only),
38    /// so the driver supplies it. A [`crate::sim::projector::RenderedEvent::MobilityFrame`]
39    /// is not writable to a scan-based (Thermo) file. Peak m/z and intensity are
40    /// validated finite and the intensity is range-checked into `f32` (claudex):
41    /// a non-finite or `f32`-overflowing value is a hard error rather than a NaN/inf
42    /// silently authored into the packet.
43    pub fn from_rendered_event(
44        ev: &crate::sim::projector::RenderedEvent,
45        collision_energy: f64,
46    ) -> Result<ScanDescriptor, String> {
47        use crate::sim::projector::RenderedEvent;
48        let (ms_level, retention_time, isolation, spectrum) = match ev {
49            RenderedEvent::Scan { ms_level, retention_time_s, isolation, spectrum } => {
50                let iso = isolation.map(|w| IsolationWindow {
51                    center_mz: w.center_mz,
52                    width_mz: w.width_mz,
53                    collision_energy,
54                });
55                (*ms_level, *retention_time_s, iso, spectrum)
56            }
57            RenderedEvent::MobilityFrame { .. } => {
58                return Err("a MobilityFrame cannot be written to a scan-based (Thermo) file; \
59                            it is an IMS instrument's output"
60                    .to_string());
61            }
62        };
63        if spectrum.mz.len() != spectrum.intensity.len() {
64            return Err(format!(
65                "rendered spectrum m/z ({}) and intensity ({}) length mismatch",
66                spectrum.mz.len(),
67                spectrum.intensity.len()
68            ));
69        }
70        let mut peaks = Vec::with_capacity(spectrum.mz.len());
71        for (&m, &i) in spectrum.mz.iter().zip(spectrum.intensity.iter()) {
72            if !m.is_finite() {
73                return Err(format!("non-finite peak m/z {m}"));
74            }
75            if !i.is_finite() {
76                return Err(format!("non-finite peak intensity {i} at m/z {m}"));
77            }
78            let i32 = i as f32;
79            if !i32.is_finite() {
80                return Err(format!("peak intensity {i} at m/z {m} overflows f32"));
81            }
82            peaks.push((m, i32));
83        }
84        Ok(ScanDescriptor { ms_level, retention_time, isolation, peaks })
85    }
86}

88/// A sink that writes vendor-neutral scans into a vendor raw file. Scans are
89/// written in acquisition order; `finalize` flushes the file (and fixes any
90/// integrity checksum).
91pub trait AcquisitionWriter {
92    fn write_scan(&mut self, scan: &ScanDescriptor) -> io::Result<()>;
93    fn finalize(&mut self) -> io::Result<()>;
94}

105    /// How `ThermoRawWriter` combines simulated peaks with the template scan.
106    #[derive(Clone, Copy, Debug)]
107    pub enum WriteMode {
108        /// Replace the template scan's signal with the simulated peaks.
109        Replace,
110        /// Add simulated peaks onto the template's real signal (real⊕sim);
111        /// `merge_tol_ppm` merges near-coincident centroids on MS2.
112        Overlay { merge_tol_ppm: f64 },
113    }
114
115    /// Authors scans into a Thermo `.raw` by **template-mutation**: open a real
116    /// `.raw`, overwrite template scans of matching type in acquisition order
117    /// (MS1 → profile via `author_profile`, MS2 → centroids via
118    /// `author_centroids`), and save on [`AcquisitionWriter::finalize`].
119    ///
120    /// The template supplies the frequency grid + calibration; synthetic peaks
121    /// are placed at their exact m/z within each packet's byte budget. The
122    /// calibration is read once from the first scan (it is ≈constant across a
123    /// run); each scan's own frequency grid is taken from its profile. For MS2,
124    /// a descriptor's [`IsolationWindow`] (center/width/CE) is authored into the
125    /// scan event via `set_isolation`, so the output's MS2 metadata reflects the
126    /// intended scheme rather than only the template's.
127    ///
128    /// Current limitation: this *replaces* the template scan's signal. A future
129    /// **overlay** mode (keep the real template signal and add simulated peaks)
130    /// would support the "reference run = layout + real noise" real⊕sim workflow.
131    pub struct ThermoRawWriter {
132        raw: RawFile,
133        out_path: PathBuf,
134        calib: Calibration,
135        /// Template slots in ACQUISITION (scan) order: `(scan_number, is_profile)`,
136        /// where `is_profile` marks an MS1 FTMS-profile slot (vs an MS2 centroid
137        /// slot). A SINGLE cursor walks this in order, so the caller must feed scans
138        /// in the template's schedule — a level mismatch is rejected (independent
139        /// MS1/MS2 cursors could not catch a reordered stream).
140        slots: Vec<(u32, bool)>,
141        cursor: usize,
142        mode: WriteMode,
143        /// Opt out of the zero-residual completeness check at `finalize` (e.g. a
144        /// smoke test authoring only a few of the template's slots). Off by default
145        /// so a real Replace run that does not fill every slot fails loudly rather
146        /// than saving a file with untouched template signal in the gaps.
147        allow_partial: bool,
148    }
149
150    /// Whether a descriptor's ms-level fits the next template slot type (MS1 ⇒
151    /// profile slot, MS2+ ⇒ centroid slot). Free fn so the ordering contract is
152    /// unit-testable without a template.
153    pub(crate) fn slot_level_matches(is_profile: bool, ms_level: u8) -> bool {
154        (ms_level <= 1) == is_profile
155    }
156
157    impl ThermoRawWriter {
158        /// Open `template` and prepare to author into `out`.
159        pub fn from_template<P: AsRef<Path>, Q: AsRef<Path>>(
160            template: P,
161            out: Q,
162        ) -> io::Result<Self> {
163            let raw = RawFile::open(template)?;
164            let calib = raw
165                .calibration_at_event(raw.scantrailer_addr as usize + 4)
166                .ok_or_else(|| {
167                    io::Error::new(io::ErrorKind::InvalidData, "no MS1 calibration in template")
168                })?;
169
170            // Build the ordered slot manifest: a non-zero profile section marks an
171            // MS1-style (FTMS profile) slot; centroid-only is MS2 (ASTMS). Kept in
172            // scan order so a single cursor enforces the template's schedule.
173            let mut slots = Vec::with_capacity(raw.index.len());
174            for (i, e) in raw.index.iter().enumerate() {
175                let pkt = (raw.data_addr + e.offset) as usize;
176                if pkt + 8 > raw.bytes.len() {
177                    continue;
178                }
179                let profile_size =
180                    u32::from_le_bytes(raw.bytes[pkt + 4..pkt + 8].try_into().unwrap());
181                let scan = raw.first_scan + i as u32;
182                slots.push((scan, profile_size > 0));
183            }
184            Ok(Self {
185                raw,
186                out_path: out.as_ref().to_path_buf(),
187                calib,
188                slots,
189                cursor: 0,
190                mode: WriteMode::Replace,
191                allow_partial: false,
192            })
193        }
194
195        /// Set the write mode (default [`WriteMode::Replace`]). Use
196        /// [`WriteMode::Overlay`] for the reference-run = layout + real-noise
197        /// (real⊕sim) workflow.
198        pub fn with_mode(mut self, mode: WriteMode) -> Self {
199            self.mode = mode;
200            self
201        }
202
203        /// Permit `finalize` to save even when not every template slot was authored
204        /// (default: off). Only for partial-write smoke tests; a real Replace run
205        /// must fill every slot to honour the zero-residual contract.
206        pub fn with_allow_partial(mut self, allow: bool) -> Self {
207            self.allow_partial = allow;
208            self
209        }
210
211        /// Template MS1/MS2 capacity, so callers can check a run fits.
212        pub fn capacity(&self) -> (usize, usize) {
213            let ms1 = self.slots.iter().filter(|s| s.1).count();
214            (ms1, self.slots.len() - ms1)
215        }
216
217        /// Total number of template slots (the exact scan count a complete run
218        /// must author, in order).
219        pub fn slot_count(&self) -> usize {
220            self.slots.len()
221        }
222
223        /// Number of slots authored so far.
224        pub fn position(&self) -> usize {
225            self.cursor
226        }
227
228        /// Whether every template slot has been authored (the zero-residual
229        /// contract: a `Replace`-mode run that does not fill the template leaves
230        /// real template signal in the unconsumed slots).
231        pub fn is_complete(&self) -> bool {
232            self.cursor == self.slots.len()
233        }
234
235        /// The ordered slot manifest `(scan_number, is_profile)` — for preflight
236        /// (matching the run's MS-level sequence to the template before writing).
237        pub fn manifest(&self) -> &[(u32, bool)] {
238            &self.slots
239        }
240    }
241
242    impl AcquisitionWriter for ThermoRawWriter {
243        fn write_scan(&mut self, scan: &ScanDescriptor) -> io::Result<()> {
244            let (t, is_profile) = *self.slots.get(self.cursor).ok_or_else(|| {
245                io::Error::new(
246                    io::ErrorKind::InvalidInput,
247                    format!(
248                        "template exhausted: {} slots, tried to author slot {}",
249                        self.slots.len(),
250                        self.cursor + 1
251                    ),
252                )
253            })?;
254            // Enforce the template's GLOBAL scan order: the descriptor's level must
255            // match the next slot's type. A single cursor catches a reordered stream
256            // (MS1,MS2,MS1,MS2 vs MS1,MS1,MS2,MS2) that independent cursors could not.
257            if !slot_level_matches(is_profile, scan.ms_level) {
258                return Err(io::Error::new(
259                    io::ErrorKind::InvalidInput,
260                    format!(
261                        "scan-order mismatch at slot {} (template scan {}): template \
262                         expects {} but got ms_level {}",
263                        self.cursor,
264                        t,
265                        if is_profile { "MS1/profile" } else { "MS2/centroid" },
266                        scan.ms_level
267                    ),
268                ));
269            }
270            if is_profile {
271                // Advance the cursor only after authoring succeeds, so a failed
272                // write doesn't permanently consume the slot.
273                match self.mode {
274                    WriteMode::Replace => self.raw.author_profile(t, &scan.peaks, &self.calib),
275                    WriteMode::Overlay { .. } => {
276                        self.raw.overlay_profile(t, &scan.peaks, &self.calib)
277                    }
278                }?;
279            } else {
280                match self.mode {
281                    WriteMode::Replace => {
282                        self.raw.author_centroids(t, &scan.peaks)?;
283                        // Author the descriptor's isolation window + CE into the
284                        // scan event so the output's MS2 metadata reflects the
285                        // intended scheme, not just the template's.
286                        if let Some(iso) = scan.isolation {
287                            self.raw.set_isolation(
288                                t,
289                                iso.center_mz,
290                                iso.width_mz,
291                                iso.collision_energy,
292                            )?;
293                        }
294                    }
295                    // Overlay keeps the template's real signal + acquisition
296                    // metadata (the scheme is derived from this template), so the
297                    // isolation/CE are left as the template's.
298                    WriteMode::Overlay { merge_tol_ppm } => {
299                        self.raw.overlay_centroids(t, &scan.peaks, merge_tol_ppm)?
300                    }
301                }
302            }
303            self.cursor += 1;
304            Ok(())
305        }
306
307        fn finalize(&mut self) -> io::Result<()> {
308            // Zero-residual contract: in Replace mode every template slot must have
309            // been authored, else the saved file keeps the template's REAL signal in
310            // the unconsumed slots (a partial/cancelled run masquerading as valid
311            // output). Overlay mode intentionally retains template signal, so it is
312            // exempt; `allow_partial` opts out for partial-write smoke tests.
313            if matches!(self.mode, WriteMode::Replace) && !self.allow_partial && !self.is_complete() {
314                return Err(io::Error::new(
315                    io::ErrorKind::InvalidInput,
316                    format!(
317                        "incomplete Replace run: {}/{} template slots authored — the \
318                         remaining {} slots still hold the template's real signal. \
319                         Author every slot, or use with_allow_partial(true).",
320                        self.cursor,
321                        self.slots.len(),
322                        self.slots.len() - self.cursor
323                    ),
324                ));
325            }
326            let path = self.out_path.clone();
327            self.raw.save(path)
328        }
329    }

355    fn from_rendered_event_maps_and_attaches_ce() {
356        let d = ScanDescriptor::from_rendered_event(&ms2_event(vec![200.0, 300.0], vec![10.0, 20.0]), 27.0)
357            .expect("valid");

363    }

365    #[test]
366    fn from_rendered_event_rejects_nonfinite_and_mobility() {
367        // Non-finite intensity is a hard error (not a NaN authored into the packet).
368        assert!(ScanDescriptor::from_rendered_event(
369            &ms2_event(vec![200.0], vec![f64::NAN]), 27.0
370        ).is_err());

377        assert!(ScanDescriptor::from_rendered_event(&mob, 0.0).is_err());

398    // Gated: set TIMSIM_ASTRAL_TEMPLATE to a real Orbitrap Astral .raw to run.
399    // `cargo test --features thermo -- --nocapture thermo_roundtrip`
400    #[test]
401    fn thermo_roundtrip() {
402        let template = match std::env::var("TIMSIM_ASTRAL_TEMPLATE") {
403            Ok(p) => p,
404            Err(_) => {
405                eprintln!("SKIP thermo_roundtrip: set TIMSIM_ASTRAL_TEMPLATE=<astral .raw>");
406                return;
407            }
408        };
409        let out = std::env::temp_dir().join("rustdf_thermo_roundtrip.raw");
410
411        // Authors only 2 of the template's many slots, so opt into partial finalize.
412        let mut w = ThermoRawWriter::from_template(&template, &out)
413            .expect("open template")
414            .with_allow_partial(true);
415        let (n_ms1, n_ms2) = w.capacity();
416        assert!(n_ms1 > 0 && n_ms2 > 0, "template has no MS1/MS2 scans");
417
418        let ms1 = ScanDescriptor {
419            ms_level: 1,
420            retention_time: 0.0,
421            isolation: None,
422            peaks: vec![(500.0, 1.0e6), (700.0, 5.0e5)],
423        };
424        let ms2 = ScanDescriptor {
425            ms_level: 2,
426            retention_time: 0.01,
427            isolation: Some(IsolationWindow {
428                center_mz: 500.0,
429                width_mz: 2.0,
430                collision_energy: 25.0,
431            }),
432            peaks: vec![(150.1, 3.0e4), (420.2, 8.0e4), (610.3, 5.0e4)],
433        };
434        w.write_scan(&ms1).expect("write MS1");
435        w.write_scan(&ms2).expect("write MS2");
436        w.finalize().expect("finalize");
437
438        // Read back through thermorawfile and confirm the authored peaks.
439        let rf = thermorawfile::RawFile::open(&out).expect("reopen");
440        assert!(rf.checksum_valid(), "checksum invalid");
441
442        // The writer used the first profile scan for MS1 and first centroid scan for MS2.
443        let cal = rf
444            .calibration_at_event(rf.scantrailer_addr as usize + 4)
445            .unwrap();
446        let mut prof_scan = None;
447        let mut cent_scan = None;
448        for i in 0..rf.index.len() {
449            let scan = rf.first_scan + i as u32;
450            let pkt = (rf.data_addr + rf.index[i].offset) as usize;
451            let psize = u32::from_le_bytes(rf.bytes[pkt + 4..pkt + 8].try_into().unwrap());
452            if psize > 0 && prof_scan.is_none() {
453                prof_scan = Some(scan);
454            }
455            if psize == 0 && cent_scan.is_none() {
456                cent_scan = Some(scan);
457            }
458        }
459        let prof = rf.profile(prof_scan.unwrap()).expect("ms1 profile");
460        assert_eq!(prof.chunks.len(), 2, "MS1 peak count");
461        let ms1_mz: Vec<f64> = prof
462            .chunks
463            .iter()
464            .map(|c| prof.mz_of_bin(c.first_bin, &cal))
465            .collect();
466        assert!((ms1_mz[0] - 500.0).abs() < 0.01 && (ms1_mz[1] - 700.0).abs() < 0.01);
467
468        let cents = rf.centroid_peaks(cent_scan.unwrap());
469        assert_eq!(cents.len(), 3, "MS2 peak count");
470        assert!((cents[0].mz - 150.1).abs() < 0.01);
471        assert!((cents[2].mz - 610.3).abs() < 0.01);
472
473        // The authored isolation window + CE must be reflected in the scan event.
474        let ev = rf.scan_event(cent_scan.unwrap()).expect("ms2 scan event");
475        assert!((ev.isolation_center - 500.0).abs() < 0.01, "authored isolation center");
476        assert!((ev.isolation_width - 2.0).abs() < 0.01, "authored isolation width");
477        assert!((ev.collision_energy - 25.0).abs() < 0.1, "authored CE");
478
479        eprintln!(
480            "thermo_roundtrip OK: MS1 {:?}, MS2 {} peaks, MS2 iso {:.2}±{:.2} CE {:.1}",
481            ms1_mz, cents.len(), ev.isolation_center, ev.isolation_width, ev.collision_energy
482        );
483    }

485    // Gated: overlay (real⊕sim) — sim peaks added onto the template's real signal.
486    #[test]
487    fn thermo_overlay() {
488        let template = match std::env::var("TIMSIM_ASTRAL_TEMPLATE") {
489            Ok(p) => p,
490            Err(_) => {
491                eprintln!("SKIP thermo_overlay: set TIMSIM_ASTRAL_TEMPLATE=<astral .raw>");
492                return;
493            }
494        };
495        let out = std::env::temp_dir().join("rustdf_thermo_overlay.raw");
496
497        // Baseline counts from the untouched template.
498        let base = thermorawfile::RawFile::open(&template).unwrap();
499        let (mut prof_scan, mut cent_scan) = (None, None);
500        for i in 0..base.index.len() {
501            let scan = base.first_scan + i as u32;
502            let pkt = (base.data_addr + base.index[i].offset) as usize;
503            let psize = u32::from_le_bytes(base.bytes[pkt + 4..pkt + 8].try_into().unwrap());
504            if psize > 0 && prof_scan.is_none() { prof_scan = Some(scan); }
505            if psize == 0 && cent_scan.is_none() { cent_scan = Some(scan); }
506        }
507        let base_pts = base.profile(prof_scan.unwrap()).unwrap().point_count();
508        let base_cents = base.centroid_peaks(cent_scan.unwrap()).len();
509
510        let mut w = ThermoRawWriter::from_template(&template, &out)
511            .expect("open")
512            .with_mode(WriteMode::Overlay { merge_tol_ppm: 10.0 });
513        w.write_scan(&ScanDescriptor {
514            ms_level: 1, retention_time: 0.0, isolation: None,
515            peaks: vec![(555.5, 9.9e5), (744.4, 7.7e5)],
516        }).unwrap();
517        w.write_scan(&ScanDescriptor {
518            ms_level: 2, retention_time: 0.01,
519            isolation: Some(IsolationWindow { center_mz: 490.0, width_mz: 2.0, collision_energy: 25.0 }),
520            peaks: vec![(333.33, 5.0e5), (888.88, 4.0e5)],
521        }).unwrap();
522        w.finalize().unwrap();
523
524        let rf = thermorawfile::RawFile::open(&out).unwrap();
525        assert!(rf.checksum_valid());
526        let cal = rf.calibration_at_event(rf.scantrailer_addr as usize + 4).unwrap();
527        let prof = rf.profile(prof_scan.unwrap()).unwrap();
528        // Real signal retained (point count grew, not replaced) + sim m/z present.
529        assert!(prof.point_count() >= base_pts, "real profile points dropped");
530        let has = |mz: f64| prof.chunks.iter().any(|c|
531            (0..c.signal.len()).any(|j| (prof.mz_of_bin(c.first_bin + j as u32, &cal) - mz).abs() < 0.02));
532        assert!(has(555.5) && has(744.4), "sim MS1 peaks missing");
533        let cents = rf.centroid_peaks(cent_scan.unwrap());
534        assert!(cents.len() >= base_cents + 1, "MS2 real centroids not retained / sim not added");
535        eprintln!("thermo_overlay OK: MS1 {}->{} pts (+sim), MS2 {}->{} centroids",
536            base_pts, prof.point_count(), base_cents, cents.len());
537    }
