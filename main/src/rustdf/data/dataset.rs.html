7pub struct TimsDataset {
8    pub loader: TimsDataLoader,
9}

11impl TimsDataset {
12    pub fn new(
13        bruker_lib_path: &str,
14        data_path: &str,
15        in_memory: bool,
16        use_bruker_sdk: bool,
17    ) -> Self {
18        // TODO: error handling
19        let global_meta_data = read_global_meta_sql(data_path).unwrap();
20        let meta_data = read_meta_data_sql(data_path).unwrap();
21
22        let scan_max_index = meta_data.iter().map(|x| x.num_scans).max().unwrap() as u32;
23        let im_lower = global_meta_data.one_over_k0_range_lower;
24        let im_upper = global_meta_data.one_over_k0_range_upper;
25
26        let tof_max_index = global_meta_data.tof_max_index;
27        let mz_lower = global_meta_data.mz_acquisition_range_lower;
28        let mz_upper = global_meta_data.mz_acquisition_range_upper;
29
30        let loader = match in_memory {
31            true => TimsDataLoader::new_in_memory(
32                bruker_lib_path,
33                data_path,
34                use_bruker_sdk,
35                scan_max_index,
36                im_lower,
37                im_upper,
38                tof_max_index,
39                mz_lower,
40                mz_upper,
41            ),
42            false => TimsDataLoader::new_lazy(
43                bruker_lib_path,
44                data_path,
45                use_bruker_sdk,
46                scan_max_index,
47                im_lower,
48                im_upper,
49                tof_max_index,
50                mz_lower,
51                mz_upper,
52            ),
53        };
54
55        TimsDataset { loader }
56    }
57}

59impl TimsData for TimsDataset {
60    // Get a frame by its id
61    fn get_frame(&self, frame_id: u32) -> TimsFrame {
62        self.loader.get_frame(frame_id)
63    }
64    // Get a raw frame by its id
65    fn get_raw_frame(&self, frame_id: u32) -> RawTimsFrame {
66        self.loader.get_raw_frame(frame_id)
67    }
68    // Get a collection of frames by their ids
69    fn get_slice(&self, frame_ids: Vec<u32>, num_threads: usize) -> TimsSlice {
70        self.loader.get_slice(frame_ids, num_threads)
71    }
72    // Get the acquisition mode, DDA or DIA
73    fn get_acquisition_mode(&self) -> AcquisitionMode {
74        self.loader.get_acquisition_mode().clone()
75    }
76    // Get total number of frames in the dataset
77    fn get_frame_count(&self) -> i32 {
78        self.loader.get_frame_count()
79    }
80    // Get the path to the data
81    fn get_data_path(&self) -> &str {
82        &self.loader.get_data_path()
83    }
84}
